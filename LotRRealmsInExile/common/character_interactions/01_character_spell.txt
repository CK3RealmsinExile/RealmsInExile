
spell_saruman_influence = {
	auto_accept = yes
	use_diplomatic_range = yes
	ignores_pending_interaction_block = yes

	desc = spell_saruman_influence_desc
	
	is_shown = {
		scope:actor = {
			can_cast_spells_trigger = yes
			NOT = { has_character_modifier = spell_saruman_influence_cooldown }
		}
		scope:recipient = {
			NOT = {
				this = scope:actor
			}
		}
	}

	is_valid_showing_failures_only = {
		scope:actor = {
			has_capacity = { COST = 1 }
			NOT = { has_strong_hook = scope:recipient }
		}
	}

	on_accept = {
		scope:actor = {
			add_character_modifier = {
				modifier = spell_saruman_influence_cooldown
				days = 30
			}
			send_interface_toast = {
				title = fabricate_hook_gained_loyalty_title
				left_icon = scope:recipient
				add_hook = {
					type = loyalty_hook
					target = scope:recipient
				}
			}
		}
		scope:recipient = {
			
		}
	}

	ai_targets = {
		ai_recipients = liege
		ai_recipients = vassals
	}

	ai_targets = {
		ai_recipients = neighboring_rulers
		ai_recipients = peer_vassals
		max = 10
	}
	
	ai_targets = {
		ai_recipients = courtiers
		max = 5
	}
	
	ai_targets = {
		ai_recipients = realm_characters
		max = 30
	}
	
	ai_target_quick_trigger = {
		adult = yes
	}

	ai_frequency = 2

	ai_will_do = {
		base = 50

		
		modifier = {
			add = -40
			
			opinion = {
				target = scope:recipient
				value >= very_high_positive_opinion
			}
		}
		
		modifier = {
			add = -40
			
			scope:recipient = {
				is_landed = no
			}
		}
		
		modifier = {
			add = 50
			
			scope:recipient = {
				is_landed = yes
				OR = {
					liege = scope:actor
					this = scope:actor.liege
					liege = scope:actor.liege
				}
			}
		}
		
		modifier = {
			add = 50
			
			scope:recipient = {
				is_councillor_of = scope:actor
			}
		}
		
		start_hostile_scheme_ai_base_modifiers = yes
	}
	
}

spell_explosives = {
	auto_accept = yes
	use_diplomatic_range = yes
	ignores_pending_interaction_block = yes

	desc = spell_explosives_desc
	
	is_shown = {
		scope:actor = {
			can_cast_spells_trigger = yes
			NOT = { has_character_modifier = spell_preparing_explosives_modifier }
		}
		scope:recipient = {
			OR = {
				is_courtier_of = scope:actor
				AND = {
					exists = scope:recipient.liege
					scope:recipient.liege = scope:actor
				}
				this = scope:actor
			}
		}
	}

	is_valid_showing_failures_only = {
		scope:actor = {
			has_capacity = { COST = 1 }
		}
		OR = {
			scope:actor = { is_ai = no }
			scope:recipient = { is_commanding_army = yes }
		}
	}

	on_accept = {
		scope:actor = {
			add_character_modifier = {
				modifier = spell_preparing_explosives_modifier
			}
			set_variable = { name = spell_explosives_target value = scope:recipient }
		}
		scope:recipient = {
			add_character_modifier = {
				modifier = spell_explosives_modifier
			}
		}
	}

	ai_targets = {
		ai_recipients = vassals
	}
	
	ai_targets = {
		ai_recipients = courtiers
		max = 5
	}
	
	ai_target_quick_trigger = {
		adult = yes
	}
	
	ai_potential = {
		can_cast_spells_trigger = yes
		is_at_war = yes
	}

	ai_frequency = 12

	ai_will_do = {
		base = 50
		
		modifier = {
			add = 30
			
			scope:recipient = {
				number_of_commander_traits >= 1
			}
		}
		
		modifier = {
			add = 10
			
			scope:recipient = {
				OR = {
					is_knight = yes
					is_councillor = yes
				}
			}
		}
		
		modifier = {
			add = -30
			
			scope:recipient = {
				martial < mediocre_skill_rating
			}
		}
	}
}

spell_explosives_stop = {
	auto_accept = yes
	use_diplomatic_range = yes

	desc = spell_explosives_desc
	
	is_shown = {
		scope:actor = {
			can_cast_spells_trigger = yes
			has_character_modifier = spell_preparing_explosives_modifier
		}
		scope:recipient = {
			OR = {
				this = scope:actor
				has_character_modifier = spell_explosives_modifier
			}
		}
	}

	is_valid_showing_failures_only = {
	}

	on_accept = {
		if = {
			limit = {
				scope:recipient = scope:actor
				scope:actor = { exists = var:spell_explosives_target }
			}
			scope:actor = {
				var:spell_explosives_target = {
					remove_character_modifier = spell_explosives_modifier
				}
			}
		}
		else = {
			scope:recipient = {
				remove_character_modifier = spell_explosives_modifier
			}
		}


		scope:actor = {
			remove_character_modifier = spell_preparing_explosives_modifier
			remove_variable = spell_explosives_target
		}
	}

	ai_targets = {
		ai_recipients = self
	}

	ai_targets = {
		ai_recipients = vassals
		ai_recipients = courtiers
	}
	
	ai_target_quick_trigger = {
		adult = yes
	}
	
	ai_potential = {
		can_cast_spells_trigger = yes
		is_at_war = no
		has_character_modifier = spell_preparing_explosives_modifier
	}

	ai_frequency = 36

	ai_will_do = {
		base = 60

		
		modifier = {
			add = 40
			
			scope:recipient = {
				is_commanding_army = no
			}
		}
		
		modifier = {
			add = 20
			
			scope:recipient = {
				number_of_commander_traits < 1
			}
		}
		
		modifier = {
			add = 20
			
			scope:recipient = {
				NOT = {
					is_knight = yes
					is_councillor = yes
				}
			}
		}
		
		modifier = {
			add = 20
			
			scope:recipient = {
				martial < mediocre_skill_rating
			}
		}
	}
	
}

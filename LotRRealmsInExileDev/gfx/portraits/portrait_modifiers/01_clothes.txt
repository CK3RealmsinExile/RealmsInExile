			}
		}	
		weight = {
			base = 0
			modifier = {
				add = 20

					AND = {
						exists = primary_spouse

							highest_held_title_tier > tier_county
						}
					}

						has_government = republic_government
						primary_title.tier > 1
					}
					AND = { is_ruler = yes has_government = theocracy_government }
				}

                    portrait_adunai_clothing_trigger = yes

	mena_nobility_legwear = {

		dna_modifiers = {
			accessory = {
				mode = add
				gene = legwear
				template = mena_nobility_legwear
				range = { 0 1 } # For the randomness to work correctly
			}
		}	
		weight = {
			base = 0
			modifier = {
				add = 50
				trigger_if = {
					limit = { exists = this }
					OR = {
						AND = {
							OR = {
								highest_held_title_tier > tier_county
								AND = {
									has_government = republic_government
									primary_title.tier > 1
								}
								AND = { is_ruler = yes has_government = theocracy_government }
							}
							NOT = { AND = { is_ruler = yes has_government = mercenary_government  } } # Blocked for mercenaries
							OR = {
								portrait_rayan_clothing_trigger = yes
								portrait_north_haradrim_clothing_trigger = yes
								portrait_south_haradrim_clothing_trigger = yes
								portrait_east_haradrim_clothing_trigger = yes
							}
						}
						AND = {
							exists = primary_spouse
							OR = {
								is_ruler = no
								highest_held_title_tier = tier_barony
							}
							primary_spouse = {
								OR = {
									highest_held_title_tier > tier_county
									AND = {
										has_government = republic_government
										primary_title.tier > 1
									}
									AND = { is_ruler = yes has_government = theocracy_government }
								}
								OR = {
									portrait_rayan_clothing_spouse_trigger = yes
									portrait_north_haradrim_clothing_spouse_trigger = yes
									portrait_south_haradrim_clothing_spouse_trigger = yes
									portrait_east_haradrim_clothing_spouse_trigger = yes
								}
							}
						}
					}
				}
				trigger_else = {
					scope:highest_held_title_tier > tier_county
					OR = {
						portrait_rayan_clothing_trigger = yes
						portrait_north_haradrim_clothing_trigger = yes
						portrait_south_haradrim_clothing_trigger = yes
						portrait_east_haradrim_clothing_trigger = yes
					}
				}
			}
		}
	}

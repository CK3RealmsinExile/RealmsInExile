namespace = lotr_dynamic_dragon_attack

scripted_effect collect_title_for_attack = {
	trigger_if = {
		limit = {
			primary_title = { is_dwarven_dynasty_title = yes }
			trigger_if = {
				limit = { primary_title = title:e_durins_folk }
				NOT = {
					is_target_in_global_variable_list = {
						name = unavailable_unique_decisions
						target = flag:decision_unify_durins_folk
					}
				}
			}
			trigger_else = { always = no }
		}
		OR = {
			has_title = capital_county.kingdom
			has_title = capital_county.empire
		}
		trigger_if = {
			limit = { primary_title.tier = tier_kingdom }
			capital_county.kingdom = { save_scope_as = title_to_be_attacked }
		}
		trigger_else = { capital_county.empire = { save_scope_as = title_to_be_attacked } }
	}
	trigger_else = { primary_title = { save_scope_as = title_to_be_attacked } }
}

scripted_trigger title_is_eligible_for_dragon_attack = {
	scope:title_to_be_attacked.title_province = { terrain = halls }
	scope:title_to_be_attacked = {
		any_de_jure_county = {
			percent > 0.51
			holder = { any_liege_or_above = { this = root } }
		}
	}
	scope:title_to_be_attacked = {
		if = {
			limit = { dragon_attack_minimum_value >= 300 }
			set_variable = eligible_dragon_title
		}
	}
}

scripted_effect list_generation = {
	every_in_list = {
		list = dragon_amusement_park
		limit = { title_province = { has_building_or_higher = $DWARVEN_BUILDING$_01 } }
		title_province = { add_to_list = $DWARVEN_BUILDING$ }
	}
}

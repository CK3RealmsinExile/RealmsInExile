#Somewhat awkwardly put this here: For Dúnedain culture spouses - assign widow trait to AI on death of spouse where children exist 

numenoreanblood.1000 = { #Check for dead character
	content_source = realms_dlc
	type = character_event
	hidden = yes
	
	immediate = {
		save_scope_as = dead_character
		every_spouse = {
			limit = { 
				NOT = { this = root }
				culture = { has_cultural_pillar = heritage_dunedain }
				any_child = { count > 0 }
			}
			trigger_event = {
				id = numenoreanblood.1001
				days = 1
			}
		}
		every_spouse = {
			limit = { 
				NOT = { this = root }
				is_elf = yes
			}
			add_trait = widowed
			add_piety = 250
		}
	}

}

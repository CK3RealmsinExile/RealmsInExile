istari.0002 = { # Find the appropriate target for istari.0003
	hidden = yes
	immediate = {
		root = { save_scope_as = departing_istar }
		if = {
			limit = { exists = root.employer }
			root.employer = {
				trigger_event = {
					id = istari.0003
					days = 1
				}
			}
		}
		else_if = { 
			limit = { exists = root.host }
			root.host = {
				trigger_event = {
					id = istari.0003
					days = 1
				}
			}
		}
		else_if = { 
			limit = { exists = root.liege }
			root.liege = {
				trigger_event = {
					id = istari.0003
					days = 1
				}
			}
		} 
	}
}

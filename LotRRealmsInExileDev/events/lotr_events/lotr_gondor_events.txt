gondor.0305 = { # The Princedom of Ithilien
	content_source = realms_dlc
	type = character_event
	title = gondor.0305.t
	desc = gondor.0305.desc
	theme = realm
	override_background = {
		reference = minas_tirith
	}

	left_portrait = {
		character = root
		animation = thinking
	}

	right_portrait = {
		character = scope:steward_of_gondor
		animation = personality_compassionate
	}
	
	lower_left_portrait = {
		character = scope:earnur
	}
	
	lower_right_portrait = {
		character = scope:mardil
	}
	
	immediate = {
		title:k_gondor_steward.holder = { save_scope_as = steward_of_gondor }
		character:lineofanarion51 = { save_scope_as = earnur }
		character:linesteward4 = { save_scope_as = mardil }
	}

	option = { #The Stewards shall henceforth be Princes
		name = gondor.0305.a
		scope:steward_of_gondor = {
			add_opinion = {
				modifier = pleased_opinion
				opinion = 50
				target = root
			}
		}
		create_title_and_vassal_change = {
			type = granted
			save_scope_as = change
			add_claim_on_loss = no
		}
		scope:steward_of_gondor = {
			if = {
				limit = {
					NOT = {
						has_title = title:k_ithilien
					}
				}
				title:k_ithilien = {
					change_title_holder = {
						holder = scope:steward_of_gondor
						change = scope:change
					}
				}
			}
			if = {
				limit = {
					NOT = {
						has_title = title:d_emyn_arnen
					}
				}
				title:d_emyn_arnen = {
					change_title_holder = {
						holder = scope:steward_of_gondor
						change = scope:change
					}
				}
			}
			if = {
				limit = {
					NOT = {
						has_title = title:c_emyn_arnen
					}
				}
				title:c_emyn_arnen = {
					change_title_holder = {
						holder = scope:steward_of_gondor
						change = scope:change
					}
				}
			}
		}
		hidden_effect = {
			every_held_title = {
				limit = {
					kingdom = title:k_ithilien
					OR = {
						tier = tier_county
						tier = tier_duchy
					}
				}
				this = {
					change_title_holder = {
						holder = scope:steward_of_gondor
						change = scope:change
					}
				}
			}
		}
		if = {
			limit = {
				NOT = {
					has_title = title:c_osgiliath
				}
			}
			title:c_osgiliath = {
				change_title_holder = {
					holder = root
					change = scope:change
				}
			}
		}
		resolve_title_and_vassal_change = scope:change
		dynasty:dynasty_ostomir ?= { add_dynasty_modifier = princely_house_of_ithilien }
		ai_chance = {
			base = 0 # LotR: This should be tied to WotR mode.

			modifier = { # Game Rule
				has_game_rule = default_ai_behavior
				add = 100
			}

			modifier = { # Game Rule
				has_game_rule = weighted_ai_behavior
				add = 70
			}

			modifier = { # Game Rule
				has_game_rule = random_ai_behavior
				add = 50
			}
		}
	}

	option = { #Ithilien is my birthright
		name = gondor.0305.b
		ai_chance = {
			base = 0 # LotR: This should be tied to WotR mode.

			modifier = { # Game Rule

				add = 0
			}

gondor.0500 = {
	content_source = realms_dlc
	type = character_event
	title = gondor.0500.t
	desc = gondor.0500.desc
	theme = diplomacy_majesty_focus
	trigger = { NOT = { has_character_flag = aragorn_has_returned } }# Don't fire if he's already king through the gondorian liberation
	override_background = {
		reference = minas_tirith
	}
	right_portrait = {
		character = root
		animation = personality_honorable
	}
	
	option = { # It is time...
		name = gondor.0500.a 
		trigger = {
			custom_tooltip = { 
				text = gondor.0500.a.tt
				AND = {
					exists = title:e_gondor_steward.holder
					title:e_gondor_steward.holder = { faith_is_good = yes }
				}
			}
		}
		show_as_unavailable = {
			NOT = {
				exists = title:e_gondor_steward.holder
				title:e_gondor_steward.holder = { faith_is_good = yes }

druedain.0201 = { # Arrival in Bru-Kha
	content_source = realms_dlc
	type = character_event
	title = druedain.0201.t
	desc = druedain.0201.desc
	theme = migration
	override_background = { reference = jungle_pyramid_ruins }

	left_portrait = {
		character = root
		animation = survey
	}
	immediate = {
        hidden_effect = {
            current_travel_plan ?= { abort_travel_plan = yes }
        }
		every_held_title = {
			limit = { tier >= tier_county }
			add_to_list = titles_to_shift
		}
		create_character = {
			template = laamp_inheritor_template
			culture = root.capital_county.culture
			faith = root.capital_county.faith
			dynasty = generate
			location = root.capital_province
			after_creation = { save_scope_as = honnin_inheritor }
		}
	}
	option = { # Here we are 
		name = druedain.0201.a
		if = { 
			limit = { has_character_flag = honnin_give_up_ray }
			get_title = title:c_bru_kha
			title:c_bru_kha = {
				set_county_faith = root.faith
				set_county_culture = root.culture
				generate_coa = yes
				every_county_province = {
					barony = { generate_coa = yes }
				}
			}
			if = { 
				limit = { has_government = tribal_government }
				title:c_bru_kha = {
					every_county_province = {
						limit = {
							has_holding = yes
							has_holding_type = wastelands_holding
						}
						set_holding_type = tribal_holding
					}
					title_province = {
						add_building = tribe_02
						add_building = longhouses_01
						add_building = market_villages_01
					}
				}
			}
			else = {
				title:c_bru_kha = {
					every_county_province = {
						limit = {
							has_holding = yes
							has_holding_type = wastelands_holding
						}
						set_holding_type = castle_holding
						add_building = castle_02 
						add_building = logging_camps_01
						add_building = ramparts_01
					}
				}
			}
			create_title_and_vassal_change = {
				type = granted
				save_scope_as = change
				add_claim_on_loss = no
			}
			
			every_in_list = {
				list = titles_to_shift
				change_title_holder = {
					holder = scope:honnin_inheritor
					change = scope:change
					take_baronies = yes
				}
			}
			resolve_title_and_vassal_change = scope:change
		}
		else = {
			ai_colonisation_effect = { WASTELANDS = title:c_bru_kha.title_province ROOT_SCOPE = root }
		}
	}
}

			}

			create_character = {
				location = title:c_dru_bhuta.title_province
				culture = culture:druedain
				faith = faith:faith_druedain
				dynasty = generate
				gender = male
				age = { 25 45 }

				trait = education_martial_2
				trait = brave

				prowess = { 6 12 }

				save_scope_as = druedain_warrior_1
			}

			create_character = {
				location = title:c_dru_bhuta.title_province
				culture = culture:druedain
				faith = faith:faith_druedain
				dynasty = generate
				gender = male
				age = { 20 40 }

				trait = education_stewardship_2
				trait = diligent
				trait = content

				diplomacy = { 4 8 }
				stewardship = { 6 10 }
				martial = { 4 8 }
				intrigue = { 3 6 }
				learning = { 6 10 }
				prowess = { 4 8 }

				save_scope_as = druedain_elder
			}

			create_character = {
				location = title:c_dru_bhuta.title_province
				culture = culture:druedain
				faith = faith:faith_druedain
				dynasty = dynasty:dynasty_aghanghu
				gender = female
				age = { 18 30 }

				trait = education_learning_2
				trait = patient
				trait = compassionate

				diplomacy = { 4 8 }
				stewardship = { 4 8 }
				martial = { 3 6 }
				intrigue = { 4 8 }
				learning = { 8 12 }
				prowess = { 2 6 }

				save_scope_as = druedain_woman
			}

			# Set up marriage so dynasty can continue
			scope:druedain_chief = {
				marry = scope:druedain_woman
			}

			# Create 1-3 children so the dynasty survives (only if spouse scope is valid)
			if = {
				limit = { exists = scope:druedain_woman }
				# First child - guaranteed
				random_list = {
					50 = {
						create_character = {
							location = title:c_dru_bhuta.title_province
							culture = culture:druedain
							faith = faith:faith_druedain

								culture = culture:druedain
								faith = faith:faith_druedain
								dynasty = dynasty:dynasty_aghanghu

								culture = culture:druedain
								faith = faith:faith_druedain
								dynasty = dynasty:dynasty_aghanghu
								father = scope:druedain_chief
								mother = scope:druedain_woman
								gender = female
								age = { 1 5 }
							}
						}
					}
				}
				# Third child - 30% chance
				random = {
					chance = 30
					random_list = {
						50 = {
							create_character = {
								location = title:c_dru_bhuta.title_province
								culture = culture:druedain
								faith = faith:faith_druedain
								dynasty = dynasty:dynasty_aghanghu
								father = scope:druedain_chief
								mother = scope:druedain_woman
								gender = male
								age = { 1 3 }
							}
						}
						50 = {
							create_character = {
								location = title:c_dru_bhuta.title_province
								culture = culture:druedain
								faith = faith:faith_druedain
								dynasty = dynasty:dynasty_aghanghu
								father = scope:druedain_chief
								mother = scope:druedain_woman
								gender = female
								age = { 1 3 }
							}
						}
					}
				}
			}

			# Make warrior and elder courtiers of the chief
			scope:druedain_warrior_1 = {
				set_employer = scope:druedain_chief
			}
			scope:druedain_elder = {
				set_employer = scope:druedain_chief
			}
		}
	}

	# Restore as a Protectorate Tributary (as at game start)
	option = {
		name = druedain.1000.a

		# Convert holdings back to tribal
		title:c_dru_bhuta = {
			every_county_province = {
				limit = { has_holding = yes }
				set_holding_type = tribal_holding
			}
		}
		title:c_dhurghud = {
			every_county_province = {
				limit = { has_holding = yes }
				set_holding_type = tribal_holding
			}
		}

		# Set county culture and faith back to Druedain
		title:c_dru_bhuta = {
			set_county_culture = culture:druedain
			set_county_faith = faith:faith_druedain
		}
		title:c_dhurghud = {
			set_county_culture = culture:druedain
			set_county_faith = faith:faith_druedain
		}

		# Step 1: Grant the counties to the chief (makes them our vassal)
		create_title_and_vassal_change = {
			type = granted
			save_scope_as = county_change
			add_claim_on_loss = no
		}
		title:c_dru_bhuta = {
			change_title_holder = {
				holder = scope:druedain_chief
				change = scope:county_change
			}
		}
		title:c_dhurghud = {
			change_title_holder = {
				holder = scope:druedain_chief
				change = scope:county_change
			}
		}
		resolve_title_and_vassal_change = scope:county_change

		# Step 2: Grant the duchy title
		create_title_and_vassal_change = {
			type = created
			save_scope_as = duchy_change
			add_claim_on_loss = no
		}
		title:d_druadan_forest = {
			change_title_holder = {
				holder = scope:druedain_chief
				change = scope:duchy_change
			}
		}
		resolve_title_and_vassal_change = scope:duchy_change

		# Step 3: Make the chief independent
		scope:druedain_chief = {
			create_title_and_vassal_change = {
				type = independency
				save_scope_as = indie_change
				add_claim_on_loss = no
			}
			becomes_independent = {
				change = scope:indie_change
			}
			resolve_title_and_vassal_change = scope:indie_change
		}

		# Step 4: Establish the protectorate tributary relationship
		scope:druedain_chief = {
			start_tributary = {
				contract_group = tributary_protectorate
				suzerain = root
			}
		}

		# Give the chief starting resources and opinion
		scope:druedain_chief = {
			add_opinion = {
				target = root
				modifier = grateful_opinion
				opinion = 50
			}
			hidden_effect = {
				add_gold = 100
				add_prestige = 300
			}
		}

		# Reward the player
		add_piety = 150
		add_prestige = 200

		stress_impact = {
			generous = minor_stress_impact_loss
			just = minor_stress_impact_loss
			compassionate = minor_stress_impact_loss
			greedy = minor_stress_impact_gain
			callous = minor_stress_impact_gain
		}
	}
}

							if = {
								limit = {
									has_building = wonder_ettenmoors_01

							}

							uses_supply = no
							inheritable = no

								type = hill_troll

							}

						}
					}
				}
				30 = { # Failure
					compare_modifier = {
						value = scope:duel_value
						multiplier = -2
						min = -25
					}
					desc = mtgram.0100.a.duel_failure
					root = {
						send_interface_toast = {
							left_icon = root
							title = mtgram.0100.failure.toast.t
							add_trait = maimed
							add_prestige = medium_prestige_loss	
							stress_impact = {
								greedy = medium_stress_impact_gain
								wrathful = medium_stress_impact_gain
								vengeful = medium_stress_impact_gain
								sadistic = minor_stress_impact_gain
								stubborn = minor_stress_impact_gain

						}
					}					
				}
			}
		}
	}

						value = scope:duel_value
						multiplier = -2
						min = -25
					}
					desc = mtgram.0100.b.duel_failure
					root = {
						send_interface_toast = {
							left_icon = root
							title = mtgram.0100.failure.toast.t		
						
							remove_short_term_gold = monumental_gold_value
							stress_impact = {
								greedy = medium_stress_impact_gain
								wrathful = medium_stress_impact_gain
								vengeful = medium_stress_impact_gain
								sadistic = minor_stress_impact_gain
								stubborn = minor_stress_impact_gain

mtgram.0300 = {
	content_source = realms_dlc
	type = character_event
	title = mtgram.0300.t
	desc = mtgram.0300.desc
	theme = war

	left_portrait = {
		character = root
		animation = aggressive_axe
	}

	lower_center_portrait = {
		character = scope:golfimbul
		animation = admiration
	}

	right_portrait = {
		character = scope:bandobras
		animation = severelywounded
	}
	
	immediate = {
		character:lineoftook4 = { save_scope_as = bandobras }
		character:lineofgolfimbul2 = { save_scope_as = golfimbul }
		
		create_title_and_vassal_change = {
			type = usurped
			save_scope_as = change
			add_claim_on_loss = yes
		}
		every_county = {				
					limit = {
						duchy = title:d_shire
						tier = tier_county
						is_wastelands = no
					}
					
	
					save_scope_as = current_target_county
					create_character = {
						location = root.capital_province
						culture = root.culture
						faith = root.faith
						gender = male
						age = { 30 150 }
						random_traits = yes
						save_scope_as = shire_orc
					}
					scope:current_target_county = {
						change_title_holder = {
							holder = scope:shire_orc
							change = scope:change
						}
						#resolve_title_and_vassal_change = scope:title_change
					}
					scope:shire_orc = {
						becomes_independent = {
							change = scope:change
						}						
					}		
			add_county_modifier = {
				modifier = orc_enslavement_free_folk_county_modifier
				years = 10
			}
			random = {
				chance = 33
				set_county_culture = root.culture
				set_county_faith = root.faith
			}
		}	
			resolve_title_and_vassal_change = scope:change							
	}
	option = {
		name = mtgram.0300.a
	}
}

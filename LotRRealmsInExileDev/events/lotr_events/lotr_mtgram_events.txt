mtgram.0100 = { # Subjugate the Ettenmoors
	content_source = realms_dlc
	type = character_event
	title = mtgram.0100.t
	desc = mtgram.0100.desc
	theme = trolls_from_the_ettenmoors

	left_portrait = {
		character = root
		animation = personality_bold
	}

	right_portrait = {
		character = scope:troll
		animation = anger
	}

	lower_center_portrait = {
		character = scope:marshal
		animation = worry
	}

	lower_left_portrait = {
		character = scope:steward
		animation = worry
	}

	lower_right_portrait = {
		character = scope:chancellor
		animation = worry
	}
	
	immediate = {
		cp:councillor_marshal = {
			save_scope_as = marshal
		}
		cp:councillor_steward = {
			save_scope_as = steward
		}
		cp:councillor_chancellor = {			
			save_scope_as = chancellor
		}
		create_character = { # Create a new character to hold the title
			culture = culture:hill_troll
			location = root.location
			gender_female_chance = 0
			save_scope_as = troll
		}
	}

	option = { # Marshal tries to tame the trolls by force
		name = mtgram.0100.a
		skill = martial
		trigger = { 
			any_martial_councillor = {
				count > 0
			}
		}
		scope:marshal = {
			duel = {
				skill = martial 
				value = 25
				30 = { # Success
					compare_modifier = {
						value = scope:duel_value
						multiplier = 2
						min = -10
					}

					desc = mtgram.0100.a.duel_success
					root = {
						send_interface_toast = {
							left_icon = root
							title = mtgram.0100.success.toast.t
						}
						ai_colonisation_effect = { WASTELANDS = title:b_cennmoin }
						title:b_cennmoin.title_province = {
							if = {
								limit = {
									has_building = wonder_ettenmoors_01
								}
								remove_building = wonder_ettenmoors_01
								add_building = wonder_ettenmoors_02
							}
						}

						spawn_army = {		  
							uses_supply = no
							inheritable = no
							name = hill_troll
							men_at_arms = {
								type = hill_troll
								stacks = 1
							}
							location = title:b_cennmoin.title_province
						}
					}
				}
				30 = { # Failure
					compare_modifier = {
						value = scope:duel_value
						multiplier = -2
						min = -25
					}
					desc = mtgram.0100.a.duel_failure
					root = {
						send_interface_toast = {
							left_icon = root
							title = mtgram.0100.failure.toast.t
							add_trait = maimed
							add_prestige = medium_prestige_loss	
							stress_impact = {
								greedy = medium_stress_impact_gain
								wrathful = medium_stress_impact_gain
								vengeful = medium_stress_impact_gain
								sadistic = minor_stress_impact_gain
								stubborn = minor_stress_impact_gain							
							}
						}
					}					
				}
			}
		}
	}
	option = { # Bribe the trolls
		name = mtgram.0100.b
		skill = stewardship
		trigger = { 
			any_stewardship_councillor = {
				count > 0
			}
		}
		scope:steward = {
			duel = {
				skill = stewardship 
				value = 25
				30 = { # Success
					compare_modifier = {
						value = scope:duel_value
						multiplier = 2
						min = -10
					}

					desc = mtgram.0100.b.duel_success					
					root = {						
						send_interface_toast = {
							left_icon = root
							title = mtgram.0100.success.toast.t
							root = {
								remove_short_term_gold = monumental_gold_value
							}						
						}
						ai_colonisation_effect = { WASTELANDS = title:b_cennmoin }
						
					}
					title:b_cennmoin.title_province = {
						if = {
							limit = {
								has_building = wonder_ettenmoors_01
							}
							remove_building = wonder_ettenmoors_01
							add_building = wonder_ettenmoors_02
						}
					}
					spawn_army = {		  
						uses_supply = no
						inheritable = no
						name = hill_troll
						men_at_arms = {
							type = hill_troll
							stacks = 1
						}
						location = title:b_cennmoin.title_province
					}

				}
				30 = { # Failure
					compare_modifier = {
						value = scope:duel_value
						multiplier = -2
						min = -25
					}
					desc = mtgram.0100.b.duel_failure
					root = {
						send_interface_toast = {
							left_icon = root
							title = mtgram.0100.failure.toast.t		
						
							remove_short_term_gold = monumental_gold_value
							stress_impact = {
								greedy = medium_stress_impact_gain
								wrathful = medium_stress_impact_gain
								vengeful = medium_stress_impact_gain
								sadistic = minor_stress_impact_gain
								stubborn = minor_stress_impact_gain
							}
						}
					}
				}
			}
		}
	}
	option = { # Call the summoners
		name = mtgram.0100.c
		skill = learning
		trigger = { 
			any_learning_councillor = {
				count > 0
			}
		}
		scope:chancellor = {
			duel = {
				skill = learning 
				value = 25
				30 = { # Success
					compare_modifier = {
						value = scope:duel_value
						multiplier = 2
						min = -10
					}

					desc = mtgram.0100.c.duel_success
					
					root = {	
						send_interface_toast = {
							left_icon = root
							title = mtgram.0100.success.toast.t
						}

						ai_colonisation_effect = { WASTELANDS = title:b_cennmoin }
					}
					title:b_cennmoin.title_province = {
						if = {
							limit = {
								has_building = wonder_ettenmoors_01
							}
							remove_building = wonder_ettenmoors_01
							add_building = wonder_ettenmoors_02
						}
					}
					spawn_army = {		  
						uses_supply = no
						inheritable = no
						name = hill_troll
						men_at_arms = {
							type = hill_troll
							stacks = 1
						}
						location = title:b_cennmoin.title_province
					}
				}
				30 = { # Failure
					compare_modifier = {
						value = scope:duel_value
						multiplier = -2
						min = -25
					}
					desc = mtgram.0100.c.duel_failure
					root = {
						send_interface_toast = {
							left_icon = root
							title = mtgram.0100.failure.toast.t	
												
							add_prestige = medium_prestige_loss
							stress_impact = {
								greedy = medium_stress_impact_gain
								wrathful = medium_stress_impact_gain
								vengeful = medium_stress_impact_gain
								sadistic = minor_stress_impact_gain
								stubborn = minor_stress_impact_gain
							}
						}
					}					
				}
			}
		}
	}
	option = { # Not this time...
		name = mtgram.0100.d
		hidden_effect = {
			remove_decision_cooldown = decision_subjugate_the_ettenmoors
			add_piety = 500
		}
	}

	after = {
		scope:troll = {
			silent_disappearance_effect = yes
		}
	}

dol_amroth.0414 = { # Decide what is to be done with Umbar
	content_source = realms_dlc
	title = dol_amroth.0414.t
	desc = dol_amroth.0414.desc
	theme = raid
	window = visit_settlement_window
	override_background = {
		reference = umbar
	}
	left_portrait = {
		character = root
		animation = war_over_win
	}
	right_portrait = {
		character = scope:umbarholder
		animation = shock
	}
	lower_center_portrait = {
		character = scope:opportunistic_haradrim_lord
		trigger = { exists = scope:opportunistic_haradrim_lord }
		animation = personality_greedy 
	}
	lower_left_portrait = {
		character = scope:opportunistic_umbarean
		animation = menacing
	}
	immediate = { # Effects from the war outcome
		title:c_umbar.holder.top_liege = { save_scope_as = umbarholder }
		title:e_umbar = {
			random_in_de_jure_hierarchy = {
				limit = {
					tier = tier_county
					holder = { faith_is_evil = no }
					holder = { is_independend_ruler = yes }
				}
				holder = { save_scope_as = opportunistic_haradrim_lord }
			}
		}
		show_as_tooltip = {
			scope:umbarholder = { destroy_title = title:k_umbar }
			title:c_umbar = {
				change_development_level = -10
				add_county_modifier = {
					modifier = dol_amroth_umbar_burned_modifier
					years = 15 
				}
			}
			title:c_umbar.title_province = {
				destroy_random_building_variable_effect = yes
				destroy_random_building_effect = yes
			}
		}
		create_character = { # Create a new character to hold the title
			location = title:c_umbar.title_province
			culture = culture:corsair
			faith = faith:faith_corsair
			gender_female_chance = 25
			save_scope_as = opportunistic_umbarean
		}
	}
	option = { # Burn the city and leave
		name = dol_amroth.0414.a
		add_prestige = major_prestige_gain
		give_nickname = nick_corsair_slayer
		add_character_flag = had_nickname_event
		custom_tooltip = dol_amroth.0414.a.tt
		ai_chance = { base = 100 }
	}
	option = { # Hand Umbar over to a Haradrim lord - preexisting lord
		trigger = { exists = scope:opportunistic_haradrim_lord }
		name = dol_amroth.0414.b
		scope:opportunistic_haradrim_lord = {
			get_title = title:c_umbar 
			get_title = title:d_umbar
			set_realm_capital = title:c_umbar
			add_gold = 500 # They plunder the city thoroughly
		}
        create_alliance = {
            target = scope:opportunistic_haradrim_lord
            allied_through_owner = root
            allied_through_target = scope:opportunistic_haradrim_lord
        }
        scope:opportunistic_haradrim_lord = {
            add_opinion = {
                modifier = event_negotiated_alliance_opinion
                target = root
            }
		}
	}
	option = { # If no suitable ruler exists, give to an opportunistic upstart
		trigger = { NOT = { exists = scope:opportunistic_haradrim_lord } }
		name = dol_amroth.0414.c

		scope:opportunistic_umbarean = {
			get_title = title:c_umbar 
			get_title = title:d_umbar
			set_realm_capital = title:c_umbar
			add_gold = 500 # They plunder the city thoroughly
		}
        create_alliance = {
            target = scope:opportunistic_umbarean
            allied_through_owner = root
            allied_through_target = scope:opportunistic_umbarean
        }
        scope:opportunistic_umbarean = {
            add_opinion = {
                modifier = event_negotiated_alliance_opinion
                target = root
            }
		}
	}
	option = { # Take Umbar yourself, having to deal with a burning city full of enemies 
		name = dol_amroth.0414.d
		get_title = title:c_umbar
	}
	after = {
		if = { # Clean up 
			limit = {
				scope:opportunistic_umbarean = { is_ruler = no }
			}
			scope:opportunistic_umbarean = { silent_disappearance_effect = yes }
		}
	}
}

dol_amroth.0403 = { # Amrothian Reserves Recruited
	title = dol_amroth.0403.t
	desc = dol_amroth.0403.desc
	theme = martial_chivalry_focus

		character = root
		animation = marshal_shield
	}
	
	immediate = { # Effects from the decision 
		show_as_tooltip = {
			spawn_army = {
				uses_supply = yes
				inheritable = yes
				name = amrothian_reserves
				levies = 50

					type = gondorian_bowmen
					stacks = 2
				}
				men_at_arms = {
					type = gondorian_footmen
					stacks = 3
				}
				location = capital_province
			}
		}
	}

	option = {
		name = dol_amroth.0403.a
	}
}

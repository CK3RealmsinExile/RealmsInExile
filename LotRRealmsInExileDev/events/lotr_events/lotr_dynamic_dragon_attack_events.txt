scripted_effect collect_title_for_attack = {
	if = {
		limit = { has_title = capital_county.kingdom }
		capital_county.kingdom = { save_scope_as = title_to_be_attacked }
	}
}

scripted_trigger title_is_eligible_for_dragon_attack = {
	scope:title_to_be_attacked.title_province = { terrain = halls }
	scope:title_to_be_attacked = {
		any_de_jure_county = {
			percent > 0.51
			holder = { any_liege_or_above = { this = root } }
		}
	}
	scope:title_to_be_attacked = { dragon_attack_minimum_value >= 200 }
	NOT = {
		is_target_in_global_variable_list = {
			name = title_is_dragon_food
			target = scope:title_to_be_attacked
		}
	}
}

scripted_effect list_generation = {
	every_in_list = {
		list = dragon_amusement_park
		limit = { title_province = { has_building_or_higher = $DWARVEN_BUILDING$_01 } }

scripted_effect dragon_destroyed_kingdom = {
	destroy_title = var:capital.kingdom
	if = {
		limit = { has_title = var:capital.duchy }
		destroy_title = var:capital.duchy
	}
	var:capital.kingdom = {
		every_de_jure_county = {
			save_scope_as = county_to_burn
			if = {
				limit = { holder = { is_wastelands = no } }
				root = {
					add_to_list = {
						name = holders
						value = prev.holder
					}
				}
			}
			make_settlement_county_wilderness = { COUNTY = scope:county_to_burn }
			if = {
				limit = { NOT = { title_province = { has_building = dragon_desolation } } }
				title_province = { add_building = dragon_desolation	}
			}
		}
		if = {
			limit = { NOT = { title_province = { has_building = dragon_den } } }
			title_province = { add_building = dragon_den }
		}
	}
}

scripted_effect dragon_defeated_came_at_us = {
	if = {
		limit = { NOT = { exists = scope:title_to_be_attacked } }
		random_held_title = {
			limit = {
				is_target_in_global_variable_list = {
					name = title_is_dragon_food
					target = capital_county.kingdom
				}
			}
			save_scope_as = title_to_be_attacked
		}
	}
	create_artifact_dragon_skull_effect = { OWNER = this }
	dynasty = { add_dynasty_prestige = 1000 }
	add_prestige = 1500
	every_held_title = {
		limit = {
			is_target_in_global_variable_list = {
				name = title_is_dragon_food
				target = scope:title_to_be_attacked
			}
		}
		remove_list_variable = {
			name = title_is_dragon_food
			target = scope:title_to_be_attacked
		}
	}
	random_owned_story = {
		limit = { story_type = lotr_dynamic_dragon_attack_story }
		end_story = yes
	}
}

lotr_dynamic_dragon_attack.0001 = {
	hidden = yes
	trigger = {
		NOT = { exists = global_var:last_great_dragon_passed }
		NOT = { exists = global_var:dragon_cooldown }
	}
	immediate = {
		set_global_variable = {
			name = dragon_cooldown
			years = 5
		}
		random_ruler = {
			limit = {
				primary_title.title_province = { terrain = halls }
				capital_county.title_province = { terrain = halls }
				highest_held_title_tier >= tier_kingdom
			}
			trigger_event = lotr_dynamic_dragon_attack.0002
		}
	}
}

lotr_dynamic_dragon_attack.0002 = {
	hidden = yes
	trigger = {
		is_landed = yes
	}
	cooldown = { years = 5 }
	immediate = {
		collect_title_for_attack = yes
		gather_title_info = yes
		if = {
			limit = { title_is_eligible_for_dragon_attack = yes }
			random = {
				chance = 5
				add_to_global_variable_list = {
					name = title_is_dragon_food
					target = scope:title_to_be_attacked
				}
				trigger_event = lotr_dynamic_dragon_attack.0003
			}
		}
	}
}

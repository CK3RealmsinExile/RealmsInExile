		trigger_event = {

		}
	}

    immediate = {
		title:k_arthedain.holder = {
			save_scope_as = sender
		}
		ROOT = {
			save_scope_as = recipient
		}
	}

    option = {  # Swear fealty
		name = arnor.3002.a
        ai_chance = {
            base = 50
            modifier = {	# 100 if lore friendly
            	add = 50

            }
        }
		
		scope:sender = { #Argeleb gets empire title so he can vassalize a King
			create_title_and_vassal_change = {
				type = granted
				save_scope_as = title_change

			}
			title:e_arnor = {
				change_title_holder = {
					holder = scope:sender
					change = scope:title_change
					take_baronies = yes
				}	
			}
			resolve_title_and_vassal_change = scope:title_change
		}
		
        scope:recipient = {
			create_title_and_vassal_change = {
				type = swear_fealty
				save_scope_as = change
				add_claim_on_loss = yes
			}
			change_liege = {
				liege = scope:sender
				change = scope:change
			}
			resolve_title_and_vassal_change = scope:change
		}
	}

    option = {  # Do not swear fealty
		name = arnor.3002.b
        
        ai_chance = {
            base = 50
            modifier = {	# 0 if lore friendly
            	add = -50
            	has_game_rule = default_ai_behavior
            }
        }
        scope:sender = {
            add_pressed_claim = scope:recipient.primary_title
        }
        scope:recipient = { add_prestige = 150 }
	}
}

namespace = lotr_hegemony

lotr_hegemonmy.0001 = {
	hidden = yes

	immediate = {
		if = {
			limit = { scope:title = title:h_mountains }
			trigger_event = lotr_hegemonmy.0002
		}
		else_if = {
			limit = { scope:title = title:h_reunited_kingdom }
			trigger_event = lotr_hegemonmy.0003
		}
		else_if = {
			limit = { scope:title = title:h_numenor_in_exile }
			trigger_event = lotr_hegemonmy.0004
		}
		else_if = {
			limit = { scope:title = title:h_pharazain }
			trigger_event = lotr_hegemonmy.0005
		}
		else_if = {
			limit = { scope:title = title:h_orc_horde }
			trigger_event = lotr_hegemonmy.0006
		}
		else_if = {
			limit = { scope:title = title:h_noldor }
			trigger_event = lotr_hegemonmy.0007
		}
	}
}

lotr_hegemonmy.0002 = {
	content_source = realms_dlc
	type = character_event
	title = lotr_hegemonmy.0002.title
	desc = lotr_hegemonmy.0002.desc
	theme = realm
	override_background = {
		reference = throne_room
	}

	trigger = {
		is_underground_race = no
	}

	option = {
		name = lotr_hegemonmy.0002.a
		destroy_title = title:h_mountains
	}
}

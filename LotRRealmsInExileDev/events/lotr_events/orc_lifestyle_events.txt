# Hatred of the Free Folk
orc_lifestyle.0020 = {
	type = character_event
	title = orc_lifestyle.0020.t
	desc = orc_lifestyle.0020.desc
	theme = war
	left_portrait = root
	override_background = { reference = armory }
	
	trigger = {
		is_orc = yes
		NOT = {  has_character_flag = had_orc_lifestyle_event_0020 }
	}
	immediate = {
		add_character_flag = { flag = had_orc_lifestyle_event_0020 }
		save_scope_as = hater
	}
	
	option = { # All
		name = orc_lifestyle.0020.a
		custom_tooltip = orc_lifestyle.0020.a.tt
		hatred_of_free_peoples_effect = yes
		ai_chance = { base = 1 }
	}
	option = { # Humans
		name = orc_lifestyle.0020.b
		custom_tooltip = orc_lifestyle.0020.a.tt
		hatred_of_free_peoples_humans_effect = yes
		ai_chance = { base = 1 }
	}
	# Elves
	# Dwarves
	# Hobbits
	# Drúedain
	option = { # Nah!
		name = orc_lifestyle.0020.g
		ai_chance = { base = 0 }
		add_gold = medium_gold_value
		add_piety = minor_piety_loss
	}
}
orc_lifestyle.0021 = {
	hidden = yes
	trigger = { has_variable = enslavement_free_folk_county }
	immediate = { remove_orc_enslavement_free_folk_county_effect = yes }
}

# Orc Fell Ability
orc_lifestyle.0030 = {
	type = character_event
	title = orc_lifestyle.0030.t
	desc = orc_lifestyle.0030.desc
	theme = realm
	left_portrait = root
	override_background = { reference = throne_room }
	
	trigger = {
		is_orc = yes
		NOT = {  has_character_flag = had_orc_lifestyle_event_0030 }
	}
	immediate = {
		add_character_flag = { flag = had_orc_lifestyle_event_0030 }
	}
	
	option = {
		name = orc_lifestyle.0030.a
		add_character_modifier = {
			modifier = orc_fell_ability_horrible_strength_modifier
			years = 5
		}
		ai_chance = { base = 1 }
	}
	option = {
		name = orc_lifestyle.0030.b
		trigger = {
			NOT = { has_trait = arbitrary }
			OR = {
				has_trait = deceitful
				has_trait = impatient
				has_trait = possessed
			}
		}
		add_character_modifier = {
			modifier = orc_fell_ability_snake_like_speed_modifier
			years = 5
		}
		ai_chance = { base = 1 }
	}
	option = {
		name = orc_lifestyle.0030.c
		trigger = { NOT = { has_trait = wheezing } }
		add_character_modifier = {
			modifier = orc_fell_ability_yell_of_triumph_modifier
			years = 5
		}
		custom_tooltip = orc_lifestyle.0030.c.tt
		ai_chance = { base = 1 }
	}
	option = {
		name = orc_lifestyle.0030.d
		trigger = {
			OR = {
				has_trait = scarred
				has_trait = stubborn
				has_trait = vengeful
			}
		}
		add_character_modifier = {
			modifier = orc_fell_ability_hideous_toughness_modifier
			years = 5
		}
		add_trait = scarred
		random_list = {
			15 = { add_trait = cancer }
			85 = {}
		}
		ai_chance = { base = 1 }
	}
	option = {
		name = orc_lifestyle.0030.e
		trigger = {
			NOT = { has_trait = brave }
			has_trait = craven
		}
		add_character_modifier = {
			modifier = orc_fell_ability_hate_of_sunlight_modifier
			years = 5
		}
		custom_tooltip = orc_lifestyle.0030.e.tt
		ai_chance = { base = 1 }
	}
	option = {
		name = orc_lifestyle.0030.f
		trigger = { has_trait = craven }
		add_character_modifier = {
			modifier = orc_fell_ability_craven_modifier
			years = 7
		}
		ai_chance = { base = 1 }
	}
}
# Wight Fell Ability
# Matt note: this theme of event can be used for other evil creatures.
#	Olog-hai fell abilities: Thickened Hide, Denizen of the Dark, Fear of Fire, Iron Grip, Horrible Strength, Yell of Triumph.
#	Dire Warg fell abilities: Fear of Fire, Snake-like Speed, Thickened Hide, Poisoned Bite, Horrible Strength, Warg Howl.
orc_lifestyle.0035 = {
	type = character_event
	title = wight_lifestyle.0035.t
	desc = wight_lifestyle.0035.desc
	theme = realm
	left_portrait = root
	override_background = { reference = throne_room }
	
	trigger = {
		is_undead = yes
		NOT = {  has_character_flag = had_wight_lifestyle_event_0035 }
	}
	immediate = {
		add_character_flag = { flag = had_wight_lifestyle_event_0035 }
	}
	
	option = {
		name = wight_lifestyle.0035.a
		add_prestige = minor_prestige_loss
		add_character_modifier = {
			modifier = wight_fell_ability_denizen_of_the_dark_modifier
			years = 5
		}
		ai_chance = { base = 1 }
	}
	option = {
		name = wight_lifestyle.0035.b
		trigger = {
			OR = {
				has_trait = sorcerer_1
				has_trait = sorcerer_2
				has_trait = sorcerer_3
				has_trait = nazgul
			}
		}
		add_piety = minor_piety_loss
		add_character_modifier = {
			modifier = wight_fell_ability_dreadful_spells_modifier
			years = 10
		}
		ai_chance = { base = 1 }
	}
	option = {
		name = wight_lifestyle.0035.c
		trigger = { has_trait = reckless }
		add_character_modifier = {
			modifier = wight_fell_ability_hate_of_sunlight_modifier
			years = 5
		}
		ai_chance = { base = 1 }
	}
	option = {
		name = wight_lifestyle.0035.d
		trigger = { NOT = { has_trait = brave } }
		add_character_modifier = {
			modifier = wight_fell_ability_fear_of_fire_modifier
			years = 5
		}
		ai_chance = { base = 1 }
	}
	option = {
		name = wight_lifestyle.0035.e
		trigger = { has_trait = nazgul }
		add_prestige = medium_prestige_loss
		add_character_modifier = {
			modifier = wight_fell_ability_foul_thing_modifier
			years = 5
		}
		ai_chance = { base = 1 }
	}
	option = {
		name = wight_lifestyle.0035.f
		trigger = { has_trait = spindly }
		add_character_modifier = {
			modifier = wight_fell_ability_iron_grip_modifier
			years = 5
		}
		ai_chance = { base = 1 }
	}
}

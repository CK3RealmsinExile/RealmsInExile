mumakan.0200 = { # Formed Utter South Hegemony
	type = character_event
	content_source = realms_dlc
	title = mumakan.0200.t
	desc = mumakan.0200.desc
	theme = crown 

	left_portrait = {
		character = root
		animation = personality_zealous
	}

	immediate = {
		show_as_tooltip = { # Effects from the decision 
			create_title_and_vassal_change = {
				type = created
				save_scope_as = title_change
				add_claim_on_loss = yes
			}
			title:h_utter_south = {
				change_title_holder = {
					holder = root
					change = scope:title_change
				}
			}
			resolve_title_and_vassal_change = scope:title_change
			set_primary_title_to = title:h_utter_south

			title:e_mumakan = {
				set_de_jure_liege_title = title:h_utter_south
			}
			if = { # Shayn
				limit = {
					has_title = title:e_shayn
					NOT = { has_title = title:d_tamska_main }
				}
				title:e_shayn = { set_de_jure_liege_title = title:h_utter_south }
			}
			title:e_usakan = {
				set_de_jure_liege_title = title:h_utter_south
			}
			if = { # Mag and Tumag
				limit = {
					has_title = title:e_mag_tumag
					has_title = title:d_tamska_main
				}
				title:e_mag_tumag = { set_de_jure_liege_title = title:h_utter_south }
			}

			root.dynasty = { add_dynasty_prestige = 1000 }
		}
	}

	option = { # I did it
		name = mumakan.0200.a 
	}
}

mumakan.0400 = { # Letter sent by Koronande to beg for mercy 
	type = letter_event
	opening = {
		desc = mumakan.0400.opening
	}
	desc = mumakan.0400.desc
	sender = scope:koronandelord
	
	immediate = {
		title:k_koronande.holder = { save_scope_as = koronandelord }
		random_character_war = {
			limit = { using_cb = koronande_war_cb }
			save_scope_as = koronande_war
		}
	}

	option = { # What? No! No mercy for you.
		name = mumakan.0400.a
		add_dread = medium_dread_gain
		title:k_koronande.holder = {
			trigger_event = {
				id = mumakan.0401 
				days = 1 
			}
		}
	}
	option = { # Very well - you can govern the far coastlands for me.
		name = mumakan.0400.b
		add_prestige = medium_prestige_gain
		show_as_tooltip = {
			scope:koronande_war = {
				end_war = white_peace
			}
			scope:koronandelord = { 
				destroy_title = title:k_koronande 
				get_title = title:c_amburak
				get_title = title c_tanande
				get_title = title:d_tanande
			}
			create_title_and_vassal_change = {
				type = conquest
				save_scope_as = change
				add_claim_on_loss = yes
			}
			custom_tooltip = {
				text = mumakan.0400.b.tt
				title:k_koronande = {
					every_in_de_jure_hierarchy = {
						limit = { exists = holder }
						holder = {
							change_liege = {
								liege = root 
								change = scope:change 
							}
						}
					}
				}
				resolve_title_and_vassal_change = scope:change
				# The faraway isles become independent since Mumakan has no fleet
				title:c_udatorai.holder = {
					create_title_and_vassal_change = {
						type = independency
						save_scope_as = change
						add_claim_on_loss = no
					}
					becomes_independent = { change = scope:change }
					resolve_title_and_vassal_change = scope:change
				}
				title:c_medatorai.holder = {
					create_title_and_vassal_change = {
						type = independency
						save_scope_as = change
						add_claim_on_loss = no
					}
					becomes_independent = { change = scope:change }
					resolve_title_and_vassal_change = scope:change
				}
				title:c_medatorai.holder = {
					get_title = title:c_por_loperi 
				}
			}
		}
		scope:koronandelord = {
			trigger_event = {
				id = mumakan.0402 
				days = 1 
			}
		}
		ai_chance = {
			base = 100 
		}
	}
}

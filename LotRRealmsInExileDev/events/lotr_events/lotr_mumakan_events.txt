namespace = mumakan

### Decision Events ###

mumakan.0200 = { # Formed Utter South Hegemony
	type = character_event
	content_source = realms_dlc
	title = mumakan.0200.t
	desc = mumakan.0200.desc
	theme = crown 

	left_portrait = {
		character = root
		animation = personality_zealous
	}

	immediate = {
		show_as_tooltip = { # Effects from the decision 
			create_title_and_vassal_change = {
				type = created
				save_scope_as = title_change
				add_claim_on_loss = yes
			}
			title:h_utter_south = {
				change_title_holder = {
					holder = root
					change = scope:title_change
				}
			}
			resolve_title_and_vassal_change = scope:title_change
			set_primary_title_to = title:h_utter_south

			title:e_mumakan = {
				set_de_jure_liege_title = title:h_utter_south
			}
			if = { # Shayn
				limit = {
					has_title = title:e_shayn
					NOT = { has_title = title:d_tamska_main }
				}
				title:e_shayn = { set_de_jure_liege_title = title:h_utter_south }
			}
			title:e_usakan = {
				set_de_jure_liege_title = title:h_utter_south
			}
			if = { # Mag and Tumag
				limit = {
					has_title = title:e_mag_tumag
					has_title = title:d_tamska_main
				}
				title:e_mag_tumag = { set_de_jure_liege_title = title:h_utter_south }
			}

			root.dynasty = { add_dynasty_prestige = 1000 }
		}
	}

	option = { # I did it
		name = mumakan.0200.a 
	}
}

mumakan.0201 = { # Talilan Conquered
	type = character_event
	content_source = realms_dlc
	title = mumakan.0201.t
	desc = mumakan.0201.desc
	theme = raid 

	left_portrait = {
		character = root
		animation = aggressive_sword
	}

	immediate = {
		show_as_tooltip = { # Effects from the decision 
			root.dynasty = { add_dynasty_prestige = 750 }
			title:c_taullond.title_province = { 
				remove_building = wonder_hall_of_the_king
				add_special_building = wonder_hall_of_the_king_occupied
			}
			add_to_global_variable_list = {
				name = unavailable_unique_decisions
				target = flag:decision_subjugate_talilan
			}
			add_elven_hope_global_effect = { VALUE = massive_elven_hope_loss HOPE_EFFECT = effect_elven_hope_important_site_lost }
		}
	}

	option = { # That'll shut those elves up
		name = mumakan.0201.a 
	}
}

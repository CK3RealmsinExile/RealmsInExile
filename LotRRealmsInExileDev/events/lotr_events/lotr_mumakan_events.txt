### Decision Events ###

mumakan.0200 = { # Formed Utter South Hegemony
	type = character_event
	content_source = realms_dlc
	title = mumakan.0200.t
	desc = mumakan.0200.desc
	theme = crown 

	left_portrait = {
		character = root
		animation = personality_zealous
	}

	immediate = {
		show_as_tooltip = { # Effects from the decision 
			create_title_and_vassal_change = {
				type = created
				save_scope_as = title_change
				add_claim_on_loss = yes
			}
			title:h_utter_south = {
				change_title_holder = {
					holder = root
					change = scope:title_change
				}
			}
			resolve_title_and_vassal_change = scope:title_change
			set_primary_title_to = title:h_utter_south

			title:e_mumakan = {
				set_de_jure_liege_title = title:h_utter_south
			}
			if = { # Shayn
				limit = {
					has_title = title:e_shayn
					NOT = { has_title = title:d_tamska_main }
				}
				title:e_shayn = { set_de_jure_liege_title = title:h_utter_south }
			}
			title:e_usakan = {
				set_de_jure_liege_title = title:h_utter_south
			}
			if = { # Mag and Tumag
				limit = {
					has_title = title:e_mag_tumag
					has_title = title:d_tamska_main
				}
				title:e_mag_tumag = { set_de_jure_liege_title = title:h_utter_south }
			}

			root.dynasty = { add_dynasty_prestige = 1000 }
		}
	}

	option = { # I did it
		name = mumakan.0200.a 
	}
}

mumakan.0202 = { # Nafarati Kingdom formed
	type = character_event
	content_source = realms_dlc
	title = mumakan.0202.t
	desc = mumakan.0202.desc
	theme = battle 

	left_portrait = {
		character = root
		animation = aggressive_longaxe
	}
	override_background = { reference = wilderness_jungle }

	immediate = {
		show_as_tooltip = { # Effects from the decision 
			root.dynasty = { add_dynasty_prestige = 500 }
			root.capital_province ?= {
				add_great_building = nafarati_war_shrine
			}
			add_legitimacy = major_legitimacy_gain
			title:k_nafarati = { set_de_jure_liege_title = title:e_mumakan }
			every_held_title = {
				limit = { tier = tier_duchy }
				set_de_jure_liege_title = title:k_nafarati
			}
			every_vassal = {
				limit = { highest_held_title_tier = tier_duchy }
				every_held_title = {
					limit = { tier = tier_duchy }
					set_de_jure_liege_title = title:k_nafarati
				}
			}
		}
	}

	option = { # Rise of the Nafarati!
		name = mumakan.0202.a 
	}
}

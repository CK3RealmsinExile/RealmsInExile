warband.2001 = { # Arlefnui
	type = character_event
	title = warband.2001.t
	desc = warband.2001.desc
	theme = war
	left_portrait = {
		character = root
		animation = personality_bold
	}

	immediate = {
		add_unpressed_claim = k_pinnath_gelin
	}

	option = { # Small claim, more warbands.
		name = warband.2001.a
		custom_tooltip = warband.2001.a.tt
		spawn_arlefnui_event_troops_large_effect = yes
		start_war = {
#			name = "war_3000_arlefnui_claim"
			casus_belli = invasion_war
			target = title:d_pinnath_gelin.holder.top_liege
			target_title = d_pinnath_gelin
		}
		ai_chance = {
			base = 0
		}
	}

	option = { # Large claim.
		name = warband.2001.b
		custom_tooltip = warband.2001.b.tt
		spawn_arlefnui_event_troops_effect = yes
		ai_chance = {
			base = 50
		}
	}

#		}

			base = 50
		}
	}
}

warband.2002 = { # Nan-i-Feryn Umbar offer
	type = character_event
	title = warband.2002.t
	desc = warband.2002.desc
	theme = war
	left_portrait = {
		character = root
		animation = worry
	}
	right_portrait = {
		character = title:k_umbar.holder
		animation = war_attacker
	}

	option = { # Yes
		name = warband.2002.a
		spawn_arlefnui_umbar_event_troops_effect = yes
		add_gold = 150
		start_war = {
			casus_belli = duchy_conquest_cb
			target = title:k_pinnath_gelin.holder.liege
			target_title = title:k_pinnath_gelin
			claimant = root
		}
		title:k_umbar.holder = {
			trigger_event = umbar.0002
		}
		ai_chance = {
			base = 70
		}
	}

	option = { # No
		name = warband.2002.b
		title:k_umbar.holder = {
			trigger_event = umbar.0003
		}
		ai_chance = {
			base = 15
		}
	}

	option = { # No take gold
		name = warband.2002.c
		add_gold = 100
		title:k_umbar.holder = {
			trigger_event = umbar.0004
		}
		ai_chance = {
			base = 15
		}
	}
}

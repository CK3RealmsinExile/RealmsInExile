				add_defender = root

			}
		}
		add_prestige = 250
		add_piety = 100
		ai_chance = {
			base = 50

			modifier = { # Double likelihood if not outnumbered
				lotr_armies_are_outnumbered = no
				factor = 2
			}

			ai_value_modifier = { # Honour can swing up to +/- 50, compassion and boldness up to +/- 25
				ai_honor = 0.5
				ai_compassion = 0.25
				ai_boldness = 0.25

orcs.0000 = {
	type = character_event
	title = orcs.0000.t
	desc = orcs.0000.desc
	theme = war
	left_portrait = root
	
	trigger = {
		NOT = {
			has_character_flag = being_taught_commander_trait
			OR = {
				has_trait = banner_orc
				has_trait = whip_master
				has_trait = explosive_engineer
			}
		}
		number_of_commander_traits < 3
		is_orc = yes
	}

	weight_multiplier = {
		base = 1
		compare_modifier = {
			factor = martial
			multiplier = 0.1
			max = 2
		}
		modifier = {
			factor = 0.3
			number_of_commander_traits > 0
		}
		modifier = {
			factor = 0.3
			number_of_commander_traits > 1
		}
	}

	option = {
		name = orcs.0000.a
		trigger = {
			NOT = { has_trait = banner_orc }
			OR = {
				has_trait = diligent
				has_trait = stubborn
				has_trait = zealous
				has_trait = education_martial_1
				has_trait = education_martial_2
				has_trait = education_martial_3
				has_trait = education_martial_4
				has_lifestyle = martial_lifestyle
			}
		}
		add_banner_orc_trait_effect = yes
		ai_chance = {
			base = 100
			ai_value_modifier = {
				ai_zeal = 0.5
				ai_energy = 0.5
				ai_boldness = 1
				ai_vengefulness = -0.2
			}
		}
	}

	option = {
		name = orcs.0000.b
		trigger = {
			NOT = { has_trait = whip_master }
			OR = {
				has_trait = sadistic
				has_trait = impatient
				has_trait = paranoid
				has_lifestyle = martial_lifestyle
			}
		}
		add_whip_master_trait_effect = yes
		ai_chance = {
			base = 100
			ai_value_modifier = {
				ai_zeal = 0.25
				ai_boldness = -0.5
				ai_vengefulness = 1
				ai_compassion = -1
			}
		}
	}
	
	option = {
		name = orcs.0000.c
		trigger = {
			NOT = { has_trait = explosive_engineer }
			OR = {
				has_trait = vengeful
				has_trait = impatient
				has_trait = zealous
				has_lifestyle = martial_lifestyle
			}
		}
		add_explosive_engineer_trait_effect = yes

				ai_rationality = 1

	}

		name = orcs.0000.d
		add_martial_skill = { 1 3 }
		orc_shouty_mc_shoutface_effect = yes
		ai_chance = { base = 1 }
	}

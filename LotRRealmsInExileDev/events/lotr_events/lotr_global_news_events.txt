# Castamiri Restoration War - Gondor Victory e.g., Umbar breaks, maybe add event to explain collapse?
news_event.0013 = {
	content_source = realms_dlc
	type = character_event
	window = fullscreen_event
	title = news_event.0013.t
	desc = news_event.0013.desc
	theme = court
	override_background = { reference = return_of_the_king }
		
	trigger = { # we need to prevent this from firing for all players when a new player joins in MP
		is_ai = no
 	}

	immediate = {
		play_music_cue = lotr_cue_gondor
		title:e_umbar.holder = {
			save_scope_as = emperor_of_umbar
		}
	}

	option = {
		name = news_event.0013.a
		trigger = {
			faith_is_good = yes
		}
	}

	option = {
		name = news_event.0013.b
		trigger = {
			faith_is_neutral = yes
		}
	}

	option = {
		name = news_event.0013.c
		trigger = {
			faith_is_evil = yes
		}
	}

	after = {
		scope:emperor_of_umbar = { # destroyes any kingdom/empire tiered title of emporer of Anadune
			every_held_title = {
				limit = {
					OR = {
						tier = tier_empire
						# tier = tier_kingdom # Commenting this out, maybe have empire titles be destroyed and let revolts from natives/corsairs
						title = title:d_high_steward #King's Men head of faith
					}
				}
				holder = { destroy_title = prev }
			}
			add_character_flag = pressed_castamiri_claim
			add_character_modifier = {
				modifier = castamiri_claim_failed
				years = 50
			}
			dynasty:dynasty_kamal = {
				add_dynasty_modifier = { modifier = dynasty_castamiri_claim_failed years = 30 }
			}

			trigger_event = { # Event to start CW in Anadune | Put this here to make root the emporer of Anadune
				id = umbar.1500
				days = { 5 15 }
			}
		}
		if = { # Add claim to King/Steward of Gondor for Umbar
			limit = {
				exists = title:e_gondor.holder
			}
			title:e_gondor.holder = {
				add_pressed_claim = title:k_umbar
				add_pressed_claim = title:d_umbar
				add_pressed_claim = title:c_umbar
			}
		} else = {
			title:e_gondor_steward.holder = {
				add_pressed_claim = title:k_umbar
				add_pressed_claim = title:d_umbar
				add_pressed_claim = title:c_umbar
			}
		}
	}
}

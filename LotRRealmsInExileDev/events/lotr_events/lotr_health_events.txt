namespace = lotr_health_events

lotr_health_events.3000 = { #apply Visible Disfigurement on_action pulse event
	scope = none
	hidden = yes

	immediate = {
		root = {
			lotr_random_visible_disfigurement_effect = yes
			trigger_event = {
				id = lotr_health_events.3001
			}
		}
	}
}

lotr_health_events.3001 = { #decision event to keep or remove the mask and apply disfigurement trait randomly
	type = character_event
	title = lotr_health_events.3001.t
	desc = {
		desc = lotr_health_events.3001.desc_intro
		first_valid = {
			triggered_desc = {
				trigger = {
					scope:disfigured_root = { has_inactive_trait = visible_disfigurement_1 }
				}
				desc = lotr_health_events.3001.desc_disfigurement_1
			}
			triggered_desc = {
				trigger = {
					scope:disfigured_root = { has_inactive_trait = visible_disfigurement_2 }
				}
				desc = lotr_health_events.3001.desc_disfigurement_2
			}
			triggered_desc = {
				trigger = {
					scope:disfigured_root = { has_inactive_trait = visible_disfigurement_3 }
				}
				desc = lotr_health_events.3001.desc_disfigurement_3
			}
			triggered_desc = {
				trigger = {
					scope:disfigured_root = { has_inactive_trait = visible_disfigurement_4 }
				}
				desc = lotr_health_events.3001.desc_disfigurement_4
			}
			triggered_desc = {
				trigger = {
					scope:disfigured_root = { has_inactive_trait = visible_disfigurement_5 }
				}
				desc = lotr_health_events.3001.desc_disfigurement_5
			}
		}
	}

	theme = recovery

	left_portrait = {
		character = root
		animation = shock
	}

	immediate = {
		root = { save_scope_as = disfigured_root }
		add_character_flag = lotr_force_disfigurement_bandages
	}

	option = { #don't wear the mask
		name = lotr_health_events.3001.a

		ai_chance = {
			base = 0
			modifier = {
				ai_boldness >= medium_positive_ai_value
				add = 10
			}
		}
	}

	option = { #wear the mask
		name = lotr_health_events.3001.b
		add_character_modifier = wearing_disfigurement_mask_modifier

		ai_chance = {
			base = 0
			modifier = {
				ai_boldness < medium_positive_ai_value
				add = 10
			}
		}
	}

	option = { #wear a full facemask meant to cover your entire face
		name = lotr_health_events.3001.c
		add_character_flag = disfigurement_fully_masked
		add_character_modifier = wearing_disfigurement_mask_modifier

		ai_chance = {
			base = 0
			modifier = {
				ai_sociability <= very_high_negative_ai_value
				add = 10
			}
		}
	}

	after = {
		remove_character_flag = lotr_force_disfigurement_bandages
	}
}

lotr_health_events.3010 = { # Custom Character Disfigurement choice
	type = character_event
	title = lotr_health_events.3010.t
	desc = lotr_health_events.3010.desc

	theme = recovery
	override_background = {
		reference = throne_room
	}
	left_portrait = {
		character = scope:cc_dummy_disfigured
		hide_info = yes
		camera = camera_head
	}
	immediate = {
		if = {
			limit = { NOT = { exists = scope:cc_dummy_disfigured } }
			create_character = {
				template = generic_peasant_character
				gender_female_chance = {
					if = {
						limit = { root = { is_female = yes } }
						add = 100
					}
				}
				employer = root
				dynasty = none
				save_scope_as = cc_dummy_disfigured
			}
			hidden_effect = {
				scope:cc_dummy_disfigured = {
					copy_inheritable_appearance_from = root
					copy_traits = root
					remove_trait = disfigured
				}
			}
		}
	}

	option = { # Clear Disfigurement
		name = lotr_health_events.3010.a
		scope:cc_dummy_disfigured = { hidden_effect = { lotr_remove_visible_disfigurement_effect = yes } }
		trigger_event = lotr_health_events.3010
	}
	option = { # Disgigurement 1
		name = lotr_health_events.3010.b
		trigger = { scope:cc_dummy_disfigured = { NOT = { has_inactive_trait = visible_disfigurement_1 } } }
		scope:cc_dummy_disfigured = { lotr_custom_character_disfigurement = { DISFIGUREMENT = visible_disfigurement_1 } }
		trigger_event = lotr_health_events.3010
	}
	option = { # Disgigurement 2
		name = lotr_health_events.3010.c
		trigger = { scope:cc_dummy_disfigured = { NOT = { has_inactive_trait = visible_disfigurement_2 } } }
		scope:cc_dummy_disfigured = { lotr_custom_character_disfigurement = { DISFIGUREMENT = visible_disfigurement_2 } }
		trigger_event = lotr_health_events.3010
	}
	option = { # Disgigurement 3
		name = lotr_health_events.3010.d
		trigger = { scope:cc_dummy_disfigured = { NOT = { has_inactive_trait = visible_disfigurement_3 } } }
		scope:cc_dummy_disfigured = { lotr_custom_character_disfigurement = { DISFIGUREMENT = visible_disfigurement_3 } }
		trigger_event = lotr_health_events.3010
	}
	option = { # Disgigurement 4
		name = lotr_health_events.3010.e
		trigger = { scope:cc_dummy_disfigured = { NOT = { has_inactive_trait = visible_disfigurement_4 } } }
		scope:cc_dummy_disfigured = { lotr_custom_character_disfigurement = { DISFIGUREMENT = visible_disfigurement_4 } }
		trigger_event = lotr_health_events.3010
	}
	option = { # Disgigurement 4
		name = lotr_health_events.3010.f
		trigger = { scope:cc_dummy_disfigured = { NOT = { has_inactive_trait = visible_disfigurement_5 } } }
		scope:cc_dummy_disfigured = { lotr_custom_character_disfigurement = { DISFIGUREMENT = visible_disfigurement_5 } }
		trigger_event = lotr_health_events.3010
	}
	option = { # Accept
		name = {
			trigger = { scope:cc_dummy_disfigured = { lotr_has_visible_disfigurement_trigger = yes } }
			text = lotr_health_events.3010.g
		}
		name = {
			trigger = { scope:cc_dummy_disfigured = { lotr_has_visible_disfigurement_trigger = no } }
			text = lotr_health_events.3010.h
		}
		hidden_effect = {
			if = {
				limit = { scope:cc_dummy_disfigured = { has_inactive_trait = visible_disfigurement_1 } }
				lotr_custom_character_disfigurement = { DISFIGUREMENT = visible_disfigurement_1 }
			}
			else_if = {
				limit = { scope:cc_dummy_disfigured = { has_inactive_trait = visible_disfigurement_2 } }
				lotr_custom_character_disfigurement = { DISFIGUREMENT = visible_disfigurement_2 }
			}
			else_if = {
				limit = { scope:cc_dummy_disfigured = { has_inactive_trait = visible_disfigurement_3 } }
				lotr_custom_character_disfigurement = { DISFIGUREMENT = visible_disfigurement_3 }
			}
			else_if = {
				limit = { scope:cc_dummy_disfigured = { has_inactive_trait = visible_disfigurement_4 } }
				lotr_custom_character_disfigurement = { DISFIGUREMENT = visible_disfigurement_4 }
			}
			else_if = {
				limit = { scope:cc_dummy_disfigured = { has_inactive_trait = visible_disfigurement_5 } }
				lotr_custom_character_disfigurement = { DISFIGUREMENT = visible_disfigurement_5 }
			}
			else = { lotr_remove_visible_disfigurement_effect = yes remove_trait = disfigured }
		}
		remove_character_flag = lotr_force_disfigurement_bandages
		hidden_effect = { scope:cc_dummy_disfigured = { death = { death_reason = death_vanished } } }
		#lotr_custom_character_effect = yes
	}
}

namespace = lotr_appearance_decisions_events

##########################################
# Visible Disfigurement
##########################################
lotr_appearance_decisions_events.4000 = { #decision event to put the mask back on
	type = character_event
	title = lotr_appearance_decisions_events.4000.t
	desc = lotr_appearance_decisions_events.4000.desc
	theme = mental_health

	left_portrait = {
		character = root
		animation = horrified_reflection_01
	}

	immediate = {
		root = { save_scope_as = disfigured_root }
	}

	option = { #put on a mask
		name = lotr_appearance_decisions_events.4000.a

		add_character_modifier = {
			modifier = wearing_disfigurement_mask_modifier
		}
	}

	option = { #cover entire face
		name = lotr_appearance_decisions_events.4000.b

		add_character_modifier = {
			modifier = wearing_disfigurement_mask_modifier
		}
		add_character_flag = disfigurement_fully_masked
	}

	option = { #nevermind
		name = lotr_appearance_decisions_events.4000.c
	}
}

lotr_appearance_decisions_events.4001 = { #decision event to take off the mask
	type = character_event
	title = lotr_appearance_decisions_events.4001.t
	desc = lotr_appearance_decisions_events.4001.desc
	theme = dread

	left_portrait = {
		character = root
		animation = holding_mirror_03
	}

	immediate = {
		root = { save_scope_as = disfigured_root }
	}

	option = { #remove my mask
		name = lotr_appearance_decisions_events.4001.a

		remove_character_modifier = wearing_disfigurement_mask_modifier
		hidden_effect = {
			if = {
				limit = { has_character_flag = disfigurement_fully_masked }
				remove_character_flag = disfigurement_fully_masked
			}
		}
	}

	option = { #nevermind
		name = lotr_appearance_decisions_events.4001.b
	}
}

namespace = lotr_appearance_decisions_events

##########################################
# Visible Disfigurement
##########################################
lotr_appearance_decisions_events.4000 = { #decision event to put the mask back on
	type = character_event
	title = lotr_appearance_decisions_events.4000.t
	desc = lotr_appearance_decisions_events.4000.desc
	theme = mental_health

	left_portrait = {
		character = root
		animation = shock
	}

	immediate = {
		root = { save_scope_as = disfigured_root }
	}

	option = { #put on a mask
		name = lotr_appearance_decisions_events.4000.a

		add_character_modifier = {
			modifier = wearing_disfigurement_mask_modifier
		}
	}

	option = { #cover entire face
		name = lotr_appearance_decisions_events.4000.b

		add_character_modifier = {
			modifier = wearing_disfigurement_mask_modifier
		}
		add_character_flag = disfigurement_fully_masked
	}

	option = { #nevermind
		name = lotr_appearance_decisions_events.4000.c
	}
}

lotr_appearance_decisions_events.4001 = { #decision event to take off the mask
	type = character_event
	title = lotr_appearance_decisions_events.4001.t
	desc = lotr_appearance_decisions_events.4001.desc
	theme = dread

	left_portrait = {
		character = root
		animation = personality_rational
	}

	immediate = {
		root = { save_scope_as = disfigured_root }
	}

	option = { #remove my mask
		name = lotr_appearance_decisions_events.4001.a

		remove_character_modifier = wearing_disfigurement_mask_modifier
		hidden_effect = {
			if = {
				limit = { has_character_flag = disfigurement_fully_masked }
				remove_character_flag = disfigurement_fully_masked
			}
		}
	}

	option = { #nevermind
		name = lotr_appearance_decisions_events.4001.b
	}
}

# One-Eyed Eyepatch
lotr_appearance_decisions_events.5000 = { #decision event to change or remove eyepatch
	type = character_event
	title = lotr_appearance_decisions_events.5000.t
	desc = lotr_appearance_decisions_events.5000.desc

	theme = recovery

	right_portrait = {
		character = root
		animation = personality_rational
	}

	option = { #Simple Cloth Eye Patch
		name = lotr_appearance_decisions_events.5000.a
		trigger = {
			NOT = { has_character_flag = beric_eyepatch }
		}
		lotr_clear_eye_patch_flags_effect = yes
		add_character_flag = beric_eyepatch
		trigger_event = lotr_appearance_decisions_events.5000
	}

	option = { #Basic Eye Patch
		name = lotr_appearance_decisions_events.5000.b
		trigger = {
			NOT = { has_character_flag = standard_eyepatch }
		}
		lotr_clear_eye_patch_flags_effect = yes
		add_character_flag = standard_eyepatch
		trigger_event = lotr_appearance_decisions_events.5000
	}

	option = { #Triangle Eye Patch
		name = lotr_appearance_decisions_events.5000.c
		trigger = {
			NOT = { has_character_flag = triangle_eyepatch }
		}
		lotr_clear_eye_patch_flags_effect = yes
		add_character_flag = triangle_eyepatch
		trigger_event = lotr_appearance_decisions_events.5000
	}

	option = { #Round Eye Patch
		name = lotr_appearance_decisions_events.5000.d
		trigger = {
			NOT = { has_character_flag = round_eyepatch }
		}
		lotr_clear_eye_patch_flags_effect = yes
		add_character_flag = round_eyepatch
		trigger_event = lotr_appearance_decisions_events.5000
	}

	option = { #Decorated Eye Patch
		name = lotr_appearance_decisions_events.5000.e
		trigger = {
			NOT = { has_character_flag = decorated_eyepatch }
		}
		lotr_clear_eye_patch_flags_effect = yes
		add_character_flag = decorated_eyepatch
		trigger_event = lotr_appearance_decisions_events.5000
	}

	option = { #Eye Wrap Clean
		name = lotr_appearance_decisions_events.5000.f
		trigger = {
			NOT = { has_character_flag = eye_bandage_clean }
		}
		lotr_clear_eye_patch_flags_effect = yes
		add_character_flag = eye_bandage_clean
		trigger_event = lotr_appearance_decisions_events.5000
	}

	option = { #None
		name = lotr_appearance_decisions_events.5000.g
		trigger = {
			NOT = { has_character_flag = no_eyepatch }
		}
		lotr_remove_eye_patch_effect = yes
		trigger_event = lotr_appearance_decisions_events.5000
	}

	option = { #This is fine
		name = lotr_appearance_decisions_events.5000.h
		trigger = { has_character_flag = one_eyed_healed }
	}
}

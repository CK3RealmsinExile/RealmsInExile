# Dunlending Lifestyle
dunland.1005 = {
	content_source = realms_dlc
	type = character_event
	title = dunland.1005.t
	desc = dunland.1005.desc
	theme = realm
	left_portrait = root
	override_background = { reference = throne_room }
	
	trigger = {
		lotr_gwathuirim_heritage_trigger = yes
		NOT = {
			culture = { has_cultural_pillar = heritage_danan_lin }
			has_character_flag = had_dunlending_lifestyle_event_0010
			has_trait = impatient
		}
	}
	immediate = {
		add_character_flag = { flag = had_dunlending_lifestyle_event_0010 }
	}
	
	option = {
		name = dunland.1005.a
		trigger = {
			OR = {
				has_trait = lifestyle_hunter
				has_trait = brave
				has_trait = patient
				has_trait = paranoid
			}
		}
		add_character_modifier = {
			modifier = wild_forest_dweller_modifier
			years = 15
		}
		ai_chance = { base = 1 }
	}
	option = {
		name = dunland.1005.b
		trigger = {
			OR = {
				has_trait = cautious_leader
				has_trait = vengeful
			}
		}
		add_character_modifier = {
			modifier = wild_first_among_equals_modifier
			years = 15
		}
		ai_chance = { base = 1 }
	}
	option = {
		name = dunland.1005.c

				has_trait = education_learning
			}
		}
		add_character_modifier = {
			modifier = wild_eye_of_isengard_modifier

		}
		ai_chance = { base = 1 }
	}
	option = {
		name = dunland.1005.d
		trigger = {
			OR = {
				has_trait = athletic
				has_trait = vengeful
				has_trait = paranoid
			}
		}
		add_character_modifier = {
			modifier = wild_mountain_savage_modifier
			years = 15
		}
		ai_chance = { base = 1 }
	}
	option = {

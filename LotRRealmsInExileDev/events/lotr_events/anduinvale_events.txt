anduin_vale.0100 = { # #Beorning starting event
	content_source = realms_dlc
	type = character_event
	title = anduin_vale.0100.t
	desc = anduin_vale.0100.desc
	theme = stewardship_duty_focus
	override_background = {
		reference = carrock
	}
	left_portrait = {
		character = root
		animation = personality_bold
	}
	right_portrait = {
		character = character:lineofbeorn1
		animation = admiration
	}
	
	option = { # 
		name = anduin_vale.0100.a
		custom_tooltip = anduin_vale.0100.a.tt
		trigger_event = {
			id = anduin_vale.0101
			years = 14
		}
	}
}

namespace = lotr_defensive_alliance

lotr_defensive_alliance.0001 = {
	hidden = yes
	trigger = { is_confederation_member = yes }
	immediate = {
		confederation ?= { remove_confederation_member = scope:character }
	}
}


### Debug event for the Balan-lai. Prevents early kick of Balan-Lai members
lotr_defensive_alliance.0002 = {
	hidden = yes
	trigger = {
		is_confederation_member = yes
		NOT = { scope:character = { has_trait = balan_lai_member } }
	}
	immediate = {
		trigger_event = { 
			id = lotr_defensive_alliance.0001
			days = 1
		}
	}
}

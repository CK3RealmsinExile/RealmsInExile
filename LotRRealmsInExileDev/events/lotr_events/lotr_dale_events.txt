				if = {
					limit = {
						NOT = { has_title = title:k_durins_folk }

				}
			}
			
			dynasty = { add_dynasty_prestige = 1000 }
			add_prestige = massive_prestige_value

	}
	option = { #We did it!
		name = dale.2003.a

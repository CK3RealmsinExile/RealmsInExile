namespace = angmar_wars

# Start Situation
angmar_wars.0001 = {
	content_source = realms_dlc
	type = character_event
	window = fullscreen_event
	title = angmar_wars.0001.t
	desc = angmar_wars.0001.desc
	theme = realm
	trigger = { # we need to prevent this from firing for all players when a new player joins in MP
		is_ai = no
 	}
	override_background = { reference = angmar_wars_fullscreen }
	override_sound = { reference = "event:/DLC/FP2/SFX/UI/fp2_struggle_ui_intro_animate" }

	cooldown = { years = 500 }

	widgets = {
		widget = {
 			gui = "event_window_widget_situation_info_angmar_wars"
 			container = "dynamic_content_widget"
 			controller = situation_info
 			setup_scope = { situation:angmar_wars_situation = { save_scope_as = situation } }
		}
	}
	option = { # Good
		name = angmar_wars.0001.a
		clicksound = "event:/DLC/FP2/SFX/UI/fp2_struggle_start_select"
		trigger = {
			is_arnor_aligned = yes
		}
	}
	option = { # Evil
		name = angmar_wars.0001.b
		clicksound = "event:/DLC/FP2/SFX/UI/fp2_struggle_start_select"
		trigger = {
			is_angmar_aligned = yes
		}
	}
	option = { # Neutral
		name = angmar_wars.0001.c
		clicksound = "event:/DLC/FP2/SFX/UI/fp2_struggle_start_select"
		trigger = {
			is_angmar_aligned = no
			is_arnor_aligned = no
		}
	}
}

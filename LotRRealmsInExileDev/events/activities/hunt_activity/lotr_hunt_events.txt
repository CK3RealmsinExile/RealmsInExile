# The Beauty of Nature
lotr_hunt.0001 = {
	type = activity_event
	title = lotr_hunt.0001.t
	desc = lotr_hunt.0001.desc
	theme = hunt_activity
	left_portrait = {
		character = root
		animation = survey
		camera = camera_event_center_away
	}
	right_portrait = {
		character = scope:hunting_elf
		animation = personality_callous
		camera = camera_event_center_away
	}

	cooldown = { years = 1 }

	trigger = {
		# Correct intent for raising hope
		is_elf = yes
	}
	
	immediate = {
		scope:activity = {
			random_attending_character = { 
				limit = { NOT = { this = root } }
				save_scope_as = hunting_elf 
			}
		}
	}
	
	option = { # Man (Elf) this forest is so beautiful. And it endures, like we should.
		name = lotr_hunt.0001.a 
		add_elven_hope_effect = { VALUE = minor_elven_hope_gain HOPE_EFFECT = effect_elven_hope_inspiring_speech }
	}
	
	option = { # No! Focus on the prey, it's getting away!
		name = lotr_hunt.0001.b
		trigger = { scope:activity = { has_activity_type = activity_hunt } }
		hunt_activity_success_change_effect = { CHANGE = increase_medium }
	}
	
	option = { # Let's focus on the inspection.
		name = lotr_hunt.0001.c
		trigger = { scope:activity = { has_activity_type = activity_survey } }
		scope:activity = {
			add_activity_log_entry = {
				key = inspection_mid_border_key
				show_in_conclusion = yes
				character = root
				root = {
					inspection_survey_intent_development = { VALUE = 10 }
					inspection_success_chance_change_effect = { CHANGE = increase_medium }
					#custom_tooltip = bp3_survey.0020.acceptance.great_success
				}
			}
		}
	}
}

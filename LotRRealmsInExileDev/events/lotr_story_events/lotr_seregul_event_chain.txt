#### Scripted Effects ####

scripted_effect change_oligarchic_holdings_rulers = {
		if = {
			limit = { any_sub_realm_county = { this = title:c_shalala } }
			title:c_shalala = { add_to_list = transfer_list }	
		}
		if = {
			limit = { any_sub_realm_county = { this = title:c_kwambiri } }
			title:c_kwambiri = { add_to_list = transfer_list }	
		}

		create_title_and_vassal_change = {
			type = $TYPE_TITLE_CHANGE$
			save_scope_as = change
			add_claim_on_loss = no
		}

		every_in_list = {
			list = transfer_list
			change_title_holder = {
				holder = $NEW_HOLDER$
				change = scope:change
				take_baronies = yes
			}
			if = {
				limit = { title_province ?= { NOT = { has_holding_type = $HOLDING_TYPE$ } } }
				title_province ?= { set_holding_type = $HOLDING_TYPE$ }
			}
			if = {
				limit = { always = $CONTROL_LOWERED_BOOL$}
				random_list = {
					33 = {
						change_county_control = -30
					}
					33 = {
						change_county_control = -60
					}
					33 = {
						change_county_control = -90
					}
				}
			}
			resolve_title_and_vassal_change = scope:change
		}
}

seregul.0252 = { # Time's up, the heartlands of Pél revolt!
	content_source = realms_dlc
	type = character_event
	theme = war
	title = seregul.0252.t
	desc = seregul.0252.desc
	left_portrait = {
		character = root
		animation = worry
	}
	right_portrait = {
		character = scope:rebel_leader
		animation = aggressive_sword
	}
	trigger = {
		NOT = {
			is_target_in_global_variable_list = {
				name = unavailable_unique_decisions
				target = flag:decision_stabilize_pel
			}
		}
	}
	immediate = {
		title:k_pel = {
			ordered_dejure_vassal_title_holder = {
				limit = { 
					this != root
					is_at_war = no
					is_imprisoned = no
					culture != root.culture
					highest_held_title_tier > tier_barony
					highest_held_title_tier < tier_kingdom
				}
				order_by = military_power
				position = 1
				save_scope_as = rebel_leader
			}
			every_dejure_vassal_title_holder = {
				limit = { 
					this != scope:rebel_leader
					is_imprisoned = no
					culture != root.culture
					highest_held_title_tier > tier_barony
					highest_held_title_tier < tier_kingdom
					NOT = { is_at_war_with = root }
				}
				add_to_list ?= seregul_attackers
			}
			every_dejure_vassal_title_holder = {
				limit = { 
					this != root
					is_at_war = no
					is_imprisoned = no
					culture = culture:kharadunaim
					highest_held_title_tier > tier_barony
					highest_held_title_tier < tier_kingdom
					NOT = { is_at_war_with = root }
				}
				add_to_list ?= seregul_defenders
			}
		}
		scope:rebel_leader = {
			start_war = {
				casus_belli = nation_fracturing_faction_war
				target = root
				target_title = title:k_pel
			}
		}
		scope:rebel_leader = {
			random_character_war = {
				limit = {
					using_cb = nation_fracturing_faction_war
				}
				save_scope_as = join_war
			}
		}
		hidden_effect = { confederation = { remove_confederation_member = root } }
	}
   
	option = { # Time to rebuild Pél the hard way
		name = seregul.0252.a
		every_in_list = {
			list = seregul_attackers
			scope:join_war = {
				add_attacker ?= prev
			}
		}
		every_in_list = {
			list = seregul_defenders
			scope:join_war = {
				add_defender ?= prev
			}
		}
	}
	after = {
		trigger_event = {
			id = seregul.0253
			days = 5
		}
	}
}

skorgrim_dourhand.0100 = {	# Khathza-zul conversion event
	content_source = realms_dlc
	type = character_event
	title = skorgrim_dourhand.0100.t
	desc = skorgrim_dourhand.0100.desc
	theme = faith
	override_background = {
		reference = kitchen_dwarven
	}

	left_portrait = {
		character = scope:skorgrim_dourhand
		animation = rage
	}
	lower_left_portrait = scope:naugladur
	lower_center_portrait = scope:elu_thingol
	lower_right_portrait = scope:khathza_the_watcher
	

	immediate = {
		play_music_cue = lotr_cue_dwarves
		character:lineofdourhand17 = { save_scope_as = skorgrim_dourhand }
		character:lineofuri3 = { save_scope_as = naugladur }
		character:lineofelwe1 = { save_scope_as = elu_thingol }
		character:lineofkhathzaibriz1 = { save_scope_as = khathza_the_watcher }	
		character:nazgul1 = { save_scope_as = witch_king } #Added
	}

	option = {
		name =  skorgrim_dourhand.0100.a
		add_character_flag = {
 			flag = delay_player_faith_conversion_notification_event
 			weeks = 1
 		}
 		set_character_faith_with_conversion = faith:faith_firebeards
		ai_chance = {
			base = 100
		}
		scope:witch_king = {
			trigger_event = { 
				id = skorgrim_dourhand.0101
				days =  7 
			}
		}
	}

	option = {
		name =  skorgrim_dourhand.0100.b
		# Do not convert and stop story content
		custom_tooltip = opt_out_of_story_content
		add_piety = 100
		ai_chance = {
			base = 0
		}
	}
}

skorgrim_dourhand.0101 = { #Witch-King heard about skorgrim and sees a chance to expand his influence
	content_source = realms_dlc
	type = character_event
	title = skorgrim_dourhand.0101.t
	desc = skorgrim_dourhand.0101.desc
	theme = martial_strategy_focus

	override_background = {
		reference = throne_room_evil
	}
	
	left_portrait = {
		character = scope:witch_king
		animation = spymaster
	}

	lower_center_portrait = scope:skorgrim_dourhand
	
	immediate = {
		play_music_cue = lotr_cue_nazgul
		character:lineofdourhand17 = { save_scope_as = skorgrim_dourhand }
		character:nazgul1 = { save_scope_as = witch_king }
		random_artifact = {
			limit = { has_variable = ring_of_firebeards }
			save_scope_as = ring_of_firebeards
		}
		random_artifact = {
			limit = { has_variable = ring_of_broadbeams }
			save_scope_as = ring_of_broadbeams
		}
	}
	artifact = {
		target = scope:ring_of_firebeards
		position = lower_right_portrait
		trigger = { exists = scope:ring_of_firebeards }
	}
	artifact = {
		target = scope:ring_of_broadbeams
		position = lower_left_portrait
		trigger = { exists = scope:ring_of_broadbeams }
	}
	
	option = { #Send the emissary
		name =  skorgrim_dourhand.0101.a
		ai_chance = {
			base = 100 #He is evil and wants power
		}
		scope:skorgrim_dourhand = {
			trigger_event = { 
				id = skorgrim_dourhand.0102 #An offer 
				days = { 5 8 }
			}
		}
	}

	option = { # Not worthy
		name =  skorgrim_dourhand.0101.b
		# Do not send the offer and stop story content
		custom_tooltip = opt_out_of_story_content
		ai_chance = {
			base = 0 #He is evil and wants power
		}
	}
}

zorzir_event_chain.0100 = { # Decide to stay near the border or retreat deeper into the tunnels.
	content_source = realms_dlc
	type = character_event
	title = zorzir_event_chain.0100.t
	desc = zorzir_event_chain.0100.desc
	theme = dwarven
	override_background = {
		reference = armory_dwarven
	}
	immediate = {
		character:lineofkreyir11 = { save_scope_as = zorzir }
		character:lineofthanul83 = { save_scope_as = thanul }
		character:lineofvarakh101 = { save_scope_as = varakh }
		character:thirdlineofthrar89 = { save_scope_as = godrathor }
		hidden_effect = {
			scope:godrathor = {
				death = {
					death_reason = death_battle
				}
			}
		}
	}
	left_portrait = {
		character = scope:zorzir
		animation = page_flipping
	}
	right_portrait = {
		character = scope:varakh
		animation = disapproval
	}
	lower_center_portrait = {
		character = scope:thanul
		animation = beg
	}

	option = { #Stay at the border
		name = zorzir_event_chain.0100.a
		add_to_global_variable_list = {
			name = unavailable_unique_decisions
			target = flag:decision_zorzir_find_captains_enabled
		}
		add_character_modifier = {
			modifier = zorzir_blacklock_survivors
			years = 10
		}
		spawn_army = {
			name = event_troop_default_name
			levies = 340
			location = root.location
			origin = root.location
			men_at_arms = {
				type = blacklock_skirmishers
				stacks = 1
			}
			men_at_arms = {
				type = blacklock_heavy_infantry
				stacks = 1
			}
		}
	}
	option = { #Retreat into the tunnels
		name = zorzir_event_chain.0100.b
		add_to_global_variable_list = {
			name = unavailable_unique_decisions
			target = flag:decision_zorzir_find_captains_enabled
		}
		#delete old holding, gain new holding
			
		title:c_felakheled = {
			save_scope_as = old_county
		}
		title:c_khurfuknak = {
			save_scope_as = new_county
		}
		
		create_title_and_vassal_change = {
			type = returned
			save_scope_as = change
			add_claim_on_loss = no
		}

		scope:new_county = {
			set_county_culture = root.culture
			set_county_faith = root.faith
			generate_coa = yes
			change_development_level = 3
		}
		scope:new_county= {
			change_title_holder = {
				holder = root
				change = scope:change
			}
		}
		resolve_title_and_vassal_change = scope:change

		scope:new_county = { generate_coa = yes }
		make_settlement_county_wilderness = { COUNTY = scope:old_county }		
		
		province:7902 = {
			set_holding_type = dwarven_holding
			add_building = dwarven_depths_02
			add_building = dwarven_terrace_farm_01
			add_building = dwarven_fungal_farm_01
			add_building = dwarven_quarry_02
		}
		
		title:c_khurfuknak = {
			add_county_modifier = {
				modifier = zorzir_survey_outpost
				years = 10
			}
		}		
	}
}

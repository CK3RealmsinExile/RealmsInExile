bellakar_story.0106 = { # Bellakar receives replies
	type = letter_event
	opening = bellakar_story.0106.opening
	desc = {
		first_valid = {
			triggered_desc = {
				trigger = {
					scope:urud_udil = {
						has_character_flag = urud_udil_vassal_refuse
					}
				}
				desc = bellakar_story.0106.desc.nay
			}
			triggered_desc = {
				trigger = {
				}
				desc = bellakar_story.0106.desc.yay
			}						
		}
	}	
	sender = scope:urud_udil

	immediate = {
		title:e_bellakar.holder = {save_scope_as = bellakar_emperor}
	}

	option = { # Good!
		name = bellakar_story.0106.a
		trigger = {
			scope:urud_udil = {
				NOT = { has_character_flag = urud_udil_vassal_refuse }
			}
		}		
		scope:urud_udil = {
			create_title_and_vassal_change = {
				type = swear_fealty
				save_scope_as = change
				add_claim_on_loss = no
			}
			change_liege = {
				liege = scope:bellakar_emperor
				change = scope:change
			}
			resolve_title_and_vassal_change = scope:change		
			
			if = {
				limit = {
					has_character_flag = urud_udil_rights
				}
				vassal_contract_set_obligation_level = {
					type = title_revocation_rights
					level = 1
				}
				vassal_contract_set_obligation_level = {
					type = religious_rights
					level = 1
				}				
			}
			
			else_if = {
				limit = {
					has_character_flag = urud_udil_vassal_feudalise
				}
				change_government = feudal_government
				scope:bellakar_emperor = {
					pay_short_term_gold = {
						gold = 300
						target = scope:urud_udil
					}
				}
			}
			
			else = {
				add_prestige = 150
			}			
		}
		ai_chance = {
			base = 100
		}		
	}
	option = { # Wow what a bad offer
		name = bellakar_story.0106.b
		trigger = {
			scope:urud_udil = {
				NOR = { 
					has_character_flag = urud_udil_vassal_refuse
					has_character_flag = urud_udil_vassal_free
				}
			}
		}
		add_prestige = 100
		ai_chance = {
			base = 0
		}				
	}
	option = { # Offer Rejected
		name = bellakar_story.0106.c
		trigger = {
			scope:urud_udil = {
				has_character_flag = urud_udil_vassal_refuse
			}
		}
		show_as_tooltip = {
			add_opinion = {

				opinion = -50			
			}			
		}
		add_prestige = -100
		ai_chance = {
			base = 100
		}		
	}
}

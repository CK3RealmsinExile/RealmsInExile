#Arnalt the Mundburgian, by Brice Underhill
lotr_fund_inspiration.6514 = {
	content_source = realms_dlc
	type = character_event
	title = lotr_fund_inspiration.6514.t
	desc = {
		desc = lotr_fund_inspiration.6514.introduction
		desc = lotr_fund_inspiration.6514.desc
	}
	theme = martial
	override_background = {
		reference = minas_tirith
	}

	right_portrait = { 
		character = scope:inspiration_owner
		animation = personality_bold
	}

	trigger = {
		exists = scope:inspiration
		ROOT = {
			NOR = { 
				AND = {
					exists = title:e_gondor_steward
					has_title = title:e_gondor_steward
				}
				AND = {
					exists = title:e_gondor
					has_title = title:e_gondor
				}
				AND = {
					exists = title:e_castamir
					has_title = title:e_castamir
				}
			}
		}
		scope:inspiration_owner = {
			location = { geographical_region = middleearth_west_gondor_easternfiefdoms }
			faith_is_evil = no
		}
		trigger_if = {
			limit = { exists = title:e_gondor_steward.holder }
			can_be_knight_trigger = {ARMY_OWNER = title:e_gondor_steward.holder}
		}
		trigger_else_if = {
			limit = { exists = title:e_gondor_steward.holder }
			can_be_knight_trigger = {ARMY_OWNER = title:e_gondor.holder}
		}
		trigger_else_if = {
			limit = { exists = title:e_gondor_steward.holder }
			can_be_knight_trigger = {ARMY_OWNER = title:e_castamir.holder}
		}
		trigger_else = {
			always = no
		}
	}
	
	cooldown = { years = 10 }

	weight_multiplier = {
		base = 1
	}

	option = { #make them your marshal, they become good at understanding soldiers
		name = lotr_fund_inspiration.6514.a
		
		trigger = {
			scope:inspiration_owner = {
				can_be_marshal_trigger = { COURT_OWNER = root }
			}
		}
		
		root = { fire_councillor = cp:councillor_marshal }
		root = {
			assign_councillor_type = {
				type = councillor_marshal
				target = scope:inspiration_owner
			}
		}
		scope:inspiration_owner = {
			add_trait = mundburgian
		}
		hidden_effect = {
			end_inspiration_sponsorship = scope:inspiration
		}
		ai_chance = {
			base = 10
			ai_value_modifier = {
				ai_compassion = 0.5
				ai_rationality = 0.5
			}
		}
	}

	option = { #threaten to arrest the adventurer for not returning
		name = lotr_fund_inspiration.6514.b
		duel = {
			skill = martial
			target = scope:inspiration_owner
			10 = {
				custom_tooltip = lotr_fund_inspiration.6514.b.toast_outcome.positive
				compare_modifier = {
					value = scope:duel_value
					multiplier = 0.5
				}
				send_interface_toast = {
					title = lotr_fund_inspiration.6514.b.toast_outcome.positive
					custom_tooltip = lotr_fund_inspiration.6514.b.toast_outcome.positive.continue
					left_icon = root
					scope:inspiration_owner = {
						change_artifact_quality_effect = { AMOUNT = 2 }
					}
				}
			}
			20 = {
				trigger = {
					scope:inspiration_owner = { cannot_be_killed = no }
				}
				custom_tooltip = lotr_fund_inspiration.6514.b.toast_outcome.refusal
				send_interface_toast = {
					title = lotr_fund_inspiration.6514.b.toast_outcome.refusal
					custom_tooltip = lotr_fund_inspiration.6514.b.toast_outcome.positive.flee
					left_icon = root
					scope:inspiration_owner = {
						end_inspiration_sponsorship = scope:inspiration
						add_prestige = medium_prestige_loss
						hidden_effect = {
							death = { death_reason = death_vanished }
						}
					}
				}
			}
		}
		ai_chance = {
			base = 10
			ai_value_modifier = {
				ai_vengefulness = 0.5
				ai_boldness = 0.5
			}
		}
	}

	option = { # let em go, you don't need em
		name = lotr_fund_inspiration.6514.c
		end_inspiration_sponsorship = scope:inspiration
		trigger = {
			scope:inspiration_owner = { cannot_be_killed = no }
		}
		scope:inspiration_owner = {
			hidden_effect = {
				death = { death_reason = death_vanished }
			}
		}
		ai_chance = {
			base = 10
		}
	}
}

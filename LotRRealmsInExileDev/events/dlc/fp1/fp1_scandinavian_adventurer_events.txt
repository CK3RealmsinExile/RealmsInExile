scripted_trigger significant_north_germanic_presence_trigger = {
	holder = {
		OR = {
			trigger_if = {	#Orcs
				limit = {
					has_game_rule = northern_orcs_invade								
				}
				AND = {
					culture = { has_cultural_pillar = heritage_orc } #OrcAdventure
					NOR = {
						culture = culture:rukhashai
						culture = { any_parent_culture_or_above = { this = culture:rukhashai } }
					}
				}
			}						
			trigger_if = {	#Goblins
				limit = {
					has_game_rule = northern_orcs_invade								
				}
				AND = {
					culture = { has_cultural_pillar = heritage_goblin } #OrcAdventure
					NOR = {
						culture = culture:echomaker_goblin
						culture = { any_parent_culture_or_above = { this = culture:echomaker_goblin } }
					}
				}
			}

			trigger_if = {	#southern Orcs
				limit = {
					has_game_rule = southern_orcs_invade								
				}
				AND = {
					culture = { has_cultural_pillar = heritage_orc } #OrcAdventure
					OR = {
						culture = culture:rukhashai
						culture = { any_parent_culture_or_above = { this = culture:rukhashai } }
					}
				}
			}						
			trigger_if = {	#southern Goblins
				limit = {
					has_game_rule = southern_orcs_invade								
				}
				AND = {
					culture = { has_cultural_pillar = heritage_goblin } #OrcAdventure
					OR = {
						culture = culture:echomaker_goblin
						culture = { any_parent_culture_or_above = { this = culture:echomaker_goblin } }
					}
				}
			}
			
			trigger_if = {	#Eotheod
				limit = {has_game_rule = northern_orcs_no_invade}
				culture = { has_cultural_pillar = heritage_hadorim }
			}		
			
			trigger_if = {	#Daen Lintis
					limit = {has_game_rule = northern_orcs_no_invade}
					culture = { has_cultural_pillar = heritage_daen_lintis } #LotR
			}	
			trigger_if = {	#Corsairs
					limit = {has_game_rule = southern_orcs_no_invade}
					culture = { has_cultural_tradition = tradition_corsairs_of_umbar }
			}
			
			trigger_if = {culture = { has_cultural_pillar = heritage_undead }}#LotR- Nazgul count
			trigger_if = {culture = { has_cultural_pillar = heritage_fallen_ainur }} #LotR- Evil Maiar count
			
			has_character_modifier = embraced_local_traditions_modifier
			AND = {
				exists = dynasty
				dynasty = {
					has_dynasty_modifier = settled_norse_dynasty_modifier
				}
			}
		}
		OR = {
			AND = {
				highest_held_title_tier >= tier_duchy
				any_held_title = {
					count >= 3
					tier = tier_county
					title_province = { geographical_region = $SCANDI_ADV_TARGET_REGION$ }
				}
			}
			AND = {
				exists = dynasty
				dynasty = {
					has_dynasty_modifier = settled_norse_dynasty_modifier
				}
			}
		}
	}
}

				culture = { has_cultural_pillar = heritage_hadorim }
			}		
			
			trigger_if = {	#Daen Lintis
					limit = {has_game_rule = northern_orcs_no_invade}
					culture = { has_cultural_pillar = heritage_daen_lintis } #LotR
			}	
			trigger_if = {	#Corsairs
					limit = {has_game_rule = southern_orcs_no_invade}
					culture = { has_cultural_tradition = tradition_corsairs_of_umbar }
			}
			trigger_if = {culture = { has_cultural_pillar = heritage_undead }}#LotR- Don't attack Nazgul
			trigger_if = {culture = { has_cultural_pillar = heritage_fallen_ainur }} #LotR- Don't attack Evil Maiar
			
			has_character_modifier = embraced_local_traditions_modifier
			AND = {
				exists = dynasty
				dynasty = {
					has_dynasty_modifier = settled_norse_dynasty_modifier
				}
			}
		
			#Plus any places already being attacked.
			top_liege = {
				any_character_war = { using_cb = fp1_scandi_adventurer_conquest }
			}
			#Or which have already earnt a ceasefire with this particular adventurer. Or any of their liege
			OR = {
				any_truce_holder = { this = scope:scandi_adventurer }
				any_liege_or_above = { 
					any_truce_holder = { this = scope:scandi_adventurer }
				}
			}
			#And finally anywhere that has earnt a little peace.
			OR = {
				has_character_modifier = fp1_reprieve_from_sa_modifier
				any_liege_or_above = { has_character_modifier = fp1_reprieve_from_sa_modifier }
			}
		}
		OR = {
			is_ai = yes
			#Filter out players, unless they're large enough to take the hit.
			AND = {
				is_ai = no
				sub_realm_size >= 8
			}
		}
	}
}

scripted_effect scandinavian_adventurer_start_war_effect = {
	#Give the adventurer a dynamic title to tide them over.
	create_dynamic_title = {
		tier = duchy
		name = {
			first_valid = {
				triggered_desc = {
					trigger = {
						OR = {
							faith = { has_doctrine = doctrine_gender_male_dominated }
							faith = { has_doctrine = doctrine_gender_male_orientated }
						}
						
					}
					desc = VIKING_ARMY_MEN_NAME
				}
				triggered_desc = {
					trigger = {
						OR = {
							faith = { has_doctrine = doctrine_gender_female_dominated }
							faith = { has_doctrine = doctrine_gender_female_orientated }
						}
					}
					desc = VIKING_ARMY_WOMEN_NAME
				}
				triggered_desc = {
					trigger = {
						faith = { has_doctrine = doctrine_gender_equal }
					}
					desc = VIKING_ARMY_NEUTRAL_NAME
				}
				#Fallback, just in case.
				desc = VIKING_ARMY_NEUTRAL_NAME
			}
		}
		adj = VIKING_ARMY_adj
	}
	create_title_and_vassal_change = {
		type = created
		save_scope_as = change
		add_claim_on_loss = no
	}
	scope:new_title = {
		set_capital_county = scope:adventurer_target
		set_landless_title = yes
		set_destroy_on_gain_same_tier = yes
		set_no_automatic_claims = yes
		set_can_be_named_after_dynasty = no
		change_title_holder = {
			holder = scope:scandi_adventurer
			change = scope:change
		}
	}
	resolve_title_and_vassal_change = scope:change
	scope:new_title = {
		generate_coa = yes
		set_variable = {
			name = temporary_title
			value = yes
		}
	}
	#Declare the war.
	scope:scandi_adventurer = {
		start_war = {
			casus_belli = fp1_scandi_adventurer_conquest
			target = scope:adventurer_target.holder.top_liege
			target_title = scope:adventurer_target.duchy
		}
	}
	# Setup event troops.
	#OrcAdventure spawn area for orcs
	scope:scandi_adventurer = {	
	# Set a random spawn location FOR NORTHERN ORCS (Attacking Mordor)
	if = {	
		limit = {
			scope:scandi_adventurer = { 
				is_orc = yes 
				NOR = {				
					culture = culture:rukhashai
					culture = { any_parent_culture_or_above = { this = culture:rukhashai } }
					culture = culture:echomaker_goblin
					culture = { any_parent_culture_or_above = { this = culture:echomaker_goblin } }

				}
			}
			scope:adventurer_target.title_province = {
				geographical_region = middleearth_mordor
			}
			 #any_county_in_region = {
			#	 region = middleearth_mordor
			#	 this = adventurer_target
				# holder = { is_in_list = western_scandi_targets_list }
				# valid_adventurer_target_trigger = yes
			 #}
		}
				
		## Try to pick a good spot in Mordor
		if = {
			limit = {
				any_county_in_region = {
					region = middleearth_mordor #LotR
					# region = world_europe_north
					valid_scandinavian_county_for_embark_trigger = yes
				}
			}
			random_county_in_region = {
				region = middleearth_mordor #LotR
				# region = world_europe_north
				limit = { valid_scandinavian_county_for_embark_trigger = yes }
				# Weight up according to development, so armies tend to spawn in places where more randos would congregate.
				weight = {
					modifier = {
						add = {
							value = development_level
							multiply = 10
						}
					}
				}
				save_temporary_scope_as = selected_county
				scope:scandi_adventurer = {
					set_variable = {
						name = random_location
						value = scope:selected_county.title_province
						days = 365
					}
				}
			}
		}
		## Otherwise, we used our backup of Barad-Dur.
		else = {
			set_variable = {
				name = random_location
				value = title:b_barad_dur.title_province #LotR
				# value = title:b_lister.title_province
				days = 365
			}
		}
	}	
	
	
	# Set a random spawn location FOR NORTHERN ORCS (Not attacking Mordor)
	else_if = {	
		limit = {
			scope:scandi_adventurer = { 
				is_orc = yes 
				NOR = {				
					culture = culture:rukhashai
					culture = { any_parent_culture_or_above = { this = culture:rukhashai } }
					culture = culture:echomaker_goblin
					culture = { any_parent_culture_or_above = { this = culture:echomaker_goblin } }

				}
			}
			# every_county_in_region = {
				# region = middleearth_mordor
				# NOT = {
					# holder = { is_in_list = western_scandi_targets_list }
				# }
			# }
		}
				
		## Try to pick a good spot in the Misty Mountains
		if = {
			limit = {
				any_county_in_region = {
					region = middleearth_mountains_misty
					NOT = {
						holder = { is_in_list = western_scandi_targets_list }
					}	
					valid_adventurer_target_trigger = yes
				}
				any_county_in_region = {
					region = middleearth_mountains_misty #LotR
					# region = world_europe_north
					valid_scandinavian_county_for_embark_trigger = yes
				}
			}
			random_county_in_region = {
				region = middleearth_mountains_misty #LotR
				# region = world_europe_north
				limit = { valid_scandinavian_county_for_embark_trigger = yes }
				# Weight up according to development, so armies tend to spawn in places where more randos would congregate.
				weight = {
					modifier = {
						add = {
							value = development_level
							multiply = 10
						}
					}
				}
				save_temporary_scope_as = selected_county
				scope:scandi_adventurer = {
					set_variable = {
						name = random_location
						value = scope:selected_county.title_province
						days = 365
					}
				}
			}
		}
		## Otherwise, we used our backup of Front Porch, by the High Pass
		else = {
			set_variable = {
				name = random_location
				value = title:b_front_porch.title_province #LotR
				# value = title:b_lister.title_province
				days = 365
			}
		}
	}
	
	else_if = {		#Is Southern Orc
		limit = {
			scope:scandi_adventurer = { 
				is_orc = yes 
				OR = {				
					culture = culture:rukhashai
					culture = { any_parent_culture_or_above = { this = culture:rukhashai } }
					culture = culture:echomaker_goblin
					culture = { any_parent_culture_or_above = { this = culture:echomaker_goblin } }

				}
			}
		}
		# Set a random spawn location FOR ORCS in the South
		## Try to pick a good spot in the Yellow Mountains
		if = {
			limit = {
				any_county_in_region = {
					region = middleearth_mountains_yellow #LotR
					# region = world_europe_north
					valid_scandinavian_county_for_embark_trigger = yes
				}
			}
			random_county_in_region = {
				region = middleearth_mountains_yellow #LotR
				# region = world_europe_north
				limit = { valid_scandinavian_county_for_embark_trigger = yes }
				# Weight up according to development, so armies tend to spawn in places where more randos would congregate.
				weight = {
					modifier = {
						add = {
							value = development_level
							multiply = 10
						}
					}
				}
				save_temporary_scope_as = selected_county
				scope:scandi_adventurer = {
					set_variable = {
						name = random_location
						value = scope:selected_county.title_province
						days = 365
					}
				}
			}
		}
		## Otherwise, we use Tul Askar, gateway into the Yellow Mountains
		else = {
			set_variable = {
				name = random_location
				value = title:b_tul_askar.title_province #LotR
				# value = title:b_lister.title_province
				days = 365
			}

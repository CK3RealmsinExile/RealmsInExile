@orc_authority_cooldown_years = 20
@tribal_authority_cooldown_years = 10

orcish_government_authority = {
	default = orc_authority_1
	cumulative = yes
	flag = realm_law
	
	orc_authority_0 = {
		modifier = {
			barons_and_minor_landholders_opinion = 20
			glory_hound_opinion = 10
			parochial_opinion = 10
			courtly_opinion = 5
		}
		flag = uses_orc_authority

		can_keep = {
			realm_law_use_orcish_government_authority = yes
		}

		on_pass = {
			# Remove modifiers.
			remove_law_related_modifiers_effect = yes
		}
	}
	
	orc_authority_1 = {
		modifier = {
			barons_and_minor_landholders_opinion = -30
			glory_hound_opinion = -15
			parochial_opinion = -15
			courtly_opinion = -5
			minority_opinion = -10
		}
		flag = uses_orc_authority
		flag = title_revocation_allowed
		flag = vassal_retraction_allowed
		flag = can_change_partition_succession_laws
		flag = diarchs_want_to_subsidise_without_this_flag

		can_keep = {
			realm_law_use_orcish_government_authority = yes
		}

		can_pass = {
			trigger_if = {
				limit = { has_realm_law = orc_authority_0 }
				custom_description = {
					subject = root
					text = "has_orc_authority_cooldown"
					NAND = {
						has_variable = orc_authority_cooldown
						NOT = {
							culture = { has_innovation = innovation_all_things }
						}
					}
				}
			}
			can_pass_law_ca1_trigger = yes
		}
		
		pass_cost = {
			prestige = {
				if = {
					limit = {
						NOT = { has_realm_law = orc_authority_2 }
					}
					add = increase_orc_authority_prestige_cost
				}
			}
		}
		
		on_pass = {
			# Increment cooldown breaker tally, if appropriate.
			calculate_authority_cooldown_break_effect = yes
			# Set regular cooldown.
			set_variable = {
				name = corv_authority_cooldown
				years = @orc_authority_cooldown_years
			}
			# Remove modifiers.
			remove_law_related_modifiers_effect = yes
		}

		ai_will_do = {
			if = {
				limit = { has_realm_law = orc_authority_0 }
				value = 1
			}
		}
	}
	
	orc_authority_2 = {
		modifier = {
			barons_and_minor_landholders_opinion = -10
			glory_hound_opinion = -10
			parochial_opinion = -10
			courtly_opinion = -5
			zealot_opinion = -10
			minority_opinion = -10
			vassal_tax_contribution_mult = 0.1
			vassal_levy_contribution_mult = 0.1
		}
		flag = uses_orc_authority
		flag = vassal_internal_wars_banned
		flag = vassal_refusal_is_treason
		flag = can_change_succession_laws
		flag = titles_cannot_leave_realm_on_succession # Hardcoded flag
		
		can_keep = {
			realm_law_use_orcish_government_authority = yes
			trigger_if = {
				limit = {
					government_allows = administrative
					is_independent_ruler = no
				}
				liege = { has_realm_law = orc_authority_2 }
			}
		}

		can_pass = {
			trigger_if = {
				limit = {
					NOT = { has_realm_law = orc_authority_3 }
					NAND = {
						government_allows = administrative
						is_independent_ruler = no
					}
				}
				has_realm_law = orc_authority_1
				custom_description = {
					subject = root
					text = "has_orc_authority_cooldown"
					NAND = {
						has_variable = orc_authority_cooldown
						NOT = {
							culture = { has_innovation = innovation_all_things }
						}
					}
				}
			}
			can_pass_law_ca2_trigger = yes
		}

		pass_cost = {
			prestige = {
				if = {
					limit = { NOT = { has_realm_law = orc_authority_3 } }
					add = increase_orc_authority_prestige_cost
				}
			}
		}
		
		on_pass = {
			# Increment cooldown breaker tally, if appropriate.
			calculate_authority_cooldown_break_effect = yes
			# Set regular cooldown.
			set_variable = {
				name = orc_authority_cooldown
				years = @orc_authority_cooldown_years
			}
			# Remove modifiers.
			remove_law_related_modifiers_effect = yes
		}

		ai_will_do = {
			if = {
				limit = { has_realm_law = orc_authority_1 }
				value = 1
			}
		}
	}
}

tgp_debate_success_score_value = {
	value = {
		add = learning
		add = diplomacy
		add = martial
		every_relation = {
			type = disciple
			add = {
				add = learning
				add = diplomacy
				add = martial
				multiply = 0.5
			}
		}
		add = {
			if = {
				limit = {
					has_trait = confucian_education
					has_trait_xp = {
						trait = confucian_education
						value > 70
					}
				}
				add = 20
			}
			else_if = {
				limit = {
					has_trait = confucian_education
					has_trait_xp = {
						trait = confucian_education
						value > 50
					}
				}
				add = 15
			}
			else_if = {
				limit = {
					has_trait = confucian_education
					has_trait_xp = {
						trait = confucian_education
						value > 20
					}
				}
				add = 10
			}
			else_if = {
				limit = { has_trait = confucian_education }
				add = 5
			}
		}
		add = {
			if = {
				limit = {
					this = scope:host
					scope:activity = {
						has_activity_option = {
							category = debate_option_books
							option = debate_books_normal
						}
					}
				}
				add = 15
			}
		}
		add = {
			if = {
				limit = {
					this = scope:host
					scope:activity = {
						has_activity_option = {
							category = debate_option_books
							option = debate_books_good
						}
					}
				}
				add = 30
			}
		}
	}
}

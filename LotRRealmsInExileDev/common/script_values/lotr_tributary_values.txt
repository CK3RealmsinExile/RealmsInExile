tributary_ranger_subsidy_value = {
    value = {
        if = {
            limit = { suzerain.prestige_level >= 4 }
            add = 4
        }
        else_if = {
            limit = { suzerain.prestige_level = 3 }
            add = 3
        }
        else_if = {
            limit = { suzerain.prestige_level = 2 }
            add = 2
        }
        else = { add = 1 }
        if = {
            limit = { suzerain.highest_held_title_tier = tier_hegemony }
            multiply = 5
        }
        else_if = {
            limit = { suzerain.highest_held_title_tier = tier_empire }
            multiply = 2
        }
        else_if = {
            limit = { suzerain.highest_held_title_tier = tier_kingdom }
            multiply = 1
        }
        else_if = {
            limit = { suzerain.highest_held_title_tier = tier_duchy }
            multiply = 0.1
        }
        else = {
            multiply = 0
        }
    }
}

tributary_ranger_suzerain_subsidy_value = {
    value = {
        holder = {
            if = {
                limit = { county.holder.prestige_level >= 4 }
                add = 4
            }
            else_if = {
                limit = { county.holder.prestige_level = 3 }
                add = 3
            }
            else_if = {
                limit = { county.holder.prestige_level = 2 }
                add = 2
            }
            else = { add = 1 }
            if = {
                limit = { county.holder.highest_held_title_tier = tier_hegemony }
                multiply = 5
            }
            else_if = {
                limit = { county.holder.highest_held_title_tier = tier_empire }
                multiply = 2
            }
            else_if = {
                limit = { county.holder.highest_held_title_tier = tier_kingdom }
                multiply = 1
            }
            else_if = {
                limit = { county.holder.highest_held_title_tier = tier_duchy }
                multiply = 0.5
            }
            else = {
                multiply = 0.1
            }
        }
    }
    multiply = suzerain_ranger_count
}

suzerain_ranger_count = {
    value = 0
    every_tributary = {
        limit = {
            subject_contract_has_flag = has_ranger_camp_contract
        }
        add = 1
    }
}

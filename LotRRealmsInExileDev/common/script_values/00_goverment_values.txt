				check_tax_collector_aptitude = {
					CHARACTER = scope:tax_collector
					VALUE = 2
				}
			}
			add = 3

		else_if = {
			limit = {
				check_tax_collector_aptitude = {
					CHARACTER = scope:tax_collector
					VALUE = 1
				}
			}

			add = 0
		}
	}
}

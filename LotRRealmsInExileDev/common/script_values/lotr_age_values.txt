
majority_age = {
	if = {
		limit = { is_elf = yes }
		value = 50
	}
	else_if = {
		limit = { is_dwarf = yes }
		value = 40
	}
	else_if = {
		limit = { is_hobbit = yes }
		value = 33
	}
	else = {
		value = 16
	}
}

nubile_age = {
	value = 16
	if = {
		limit = { is_elf = yes }
		add = 34
	}
	else_if = {
		limit = { is_dwarf = yes }
		add = 54
	}
	else_if = {
		limit = { is_hobbit = yes }
		add = 17
	}
	if = { # Level 1 numenorean
		limit = {
			numenorean_blood_tier >= 1
		}
		add = 4
	}
	if = { # Level 4 numenorean
		limit = {
			numenorean_blood_tier >= 4
		}
		add = 10
	}
	if = { # Level 7 numenorean
		limit = {
			numenorean_blood_tier >= 7
		}
		add = 10
	}
	if = { # Level 10 numenorean
		limit = {
			numenorean_blood_tier = 10
		}
		add = 60
	}
}

years_to_nubility = {
	value = nubile_age
	subtract = age
	min = 0
}

numenorean_fertility_years = {
	value = numenorean_blood_tier
	multiply = 10
}

additional_years_of_fertility = {
	value = 0
	
	if = {
		limit = {
			OR = {
				cannot_be_killed = yes
				is_elf = yes
			}
		}
		add = 13000
	}
	if = {
		limit = { is_orc = yes }
		add = 160
	}
	else_if = {
		limit = { 
			culture = { has_cultural_pillar = heritage_longbeard }
		}
		add = 150
	}
	else_if = {
		limit = { is_dwarf = yes }
		add = 130
	}
	else_if = {
		limit = { is_hobbit = yes }
		add = 20
	}
	else_if = {
		limit = { is_druedain = yes }
		subtract = 20
	}
	
	add = numenorean_fertility_years
	
	if = {
		limit = { has_trait = fecund }
		add = 5
	}
	if = {
		limit = { has_trait = dwarf_father }
		add = 25
	}
	if = {
		limit = { has_character_modifier = reached_meneltarma }
		add = 20
	}
}

fertility_age_equivalent = {
	value = age
	subtract = additional_years_of_fertility
	min = 0
}

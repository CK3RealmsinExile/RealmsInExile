lotr_secrecy_hud_display = {
    value = 0

    if = {
        limit = {
            scope:active_level_flag ?= flag:secrecy_level_1
        }
        add = lotr_current_secrecy
    } 
    if = {
        limit = {
            scope:active_level_flag ?= flag:secrecy_level_2
            lotr_current_secrecy > 20
        }
        add = {
            value = lotr_current_secrecy
            subtract = 20
        }
    } 
    if = {
        limit = {
            scope:active_level_flag ?= flag:secrecy_level_3
            lotr_current_secrecy > 40
        }
        add = {
            value = lotr_current_secrecy
            subtract = 40
        }
    } 
    if = {
        limit = {
            scope:active_level_flag ?= flag:secrecy_level_4
            lotr_current_secrecy > 60
        }
        add = {
            value = lotr_current_secrecy
            subtract = 60
        }
    } 
    if = {
        limit = {
            scope:active_level_flag ?= flag:secrecy_level_5
            lotr_current_secrecy > 80
        }
        add = {
            value = lotr_current_secrecy
            subtract = 80
        }
    }
}

lotr_total_secrecy = {
    value = 100
}

lotr_current_secrecy = {
    value = 50
}

lotr_secrecy_monthly_change = {
    value = 0.5
}

lotr_character_corruption = {
    value = 0

    if = {
        limit = { has_variable = lotr_corruption }
        add = var:lotr_corruption
    }
}

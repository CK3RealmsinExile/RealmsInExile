feudal_levies_exempt = {
	value = 0
}

feudal_levies_low = {
	value = 0.1

	if = { 
		limit = {
			scope:liege = {
				OR = {
					has_trait = istari
					has_trait = maiar
					this = character:lineisildur42
				}	
			}
		}
		add = 0.1
	}

	if = {
		limit = {
			AND = {
				scope:vassal = { is_elf = no }
				scope:liege = { is_elf = yes }
			}
		}
		multiply = 0
	}
}

lotr_form_tedjin_confederacy_total_convinced = {
	value = 0
	if = {
		limit = {
			any_county_in_region = {
				region = custom_reform_tedjin_confederacy
				exists = holder
				holder = { is_independent_ruler = yes }
				holder.culture = culture:tedjin
				NOT = { holder = root }
				count = all  
				holder = { add_to_temporary_list = tedjin_rulers }
			}
		}
	}
	every_in_list = {
		list = tedjin_rulers
		limit = {
			OR = {
				# Is you, just in case
				this = root
				# your friend
				has_relation_friend = root
				AND = {
					has_diarchy_type = regency
					diarch = {
						has_relation_friend = root
					}
				}
				# your lover
				has_relation_lover = root
				AND = {
					has_diarchy_type = regency
					diarch = {
						has_relation_lover = root
					}
				}
				# Your ally
				any_ally = {
					this = root
				}
				AND = {
					has_diarchy_type = regency
					diarch = {
						any_ally = {
							this = root
						}
					}
				}
				# Your hooked
				root = {
					has_hook = prev
				}
				AND = {
					has_diarchy_type = regency
					diarch = {
						root = {
							has_hook = prev
						}
					}
				}
				#Your tributary
				root = {
					OR = {
						is_target_in_variable_list = {
							name = non_permanent_tributaries
							target = prev
						}
						is_target_in_variable_list = {
							name = permanent_tributaries
							target = prev
						}
					}
				}
			}
		}
		add = 1
	}
}

# Individual Power Value - Scope: character
admin_factions_character_power_value = {
	value = {
		add = {
			value = military_power #current_military_strength
			multiply = 0.01 #0.05
			desc = FACTION_POWER_MILITARY_STRENGTH
		}
	}
}

# Total Faction Power Value - Scope: situation_group_participant
participant_group_faction_power_value = {
	value = 0
	scope:situation_participant_group = {
		every_situation_group_participant = {
			if = {
				limit = { has_variable = faction_power }
				add = var:faction_power
			}
		}
	}
}

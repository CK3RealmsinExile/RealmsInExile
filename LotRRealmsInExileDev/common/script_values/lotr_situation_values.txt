### Loyalist Group ###
lotr_kin_strife_join_loyalist_group_chance = {
	value = 0
	if = { limit = { save_temporary_scope_as = curr_character } } #Saves character as scope

	add = lotr_kin_strife_join_loyalist_group_chance_relationships
	add = lotr_kin_strife_join_loyalist_group_chance_house
}

lotr_kin_strife_join_loyalist_group_chance_relationships = {
	value = 0
	### Positive Relationships ###
	if = { #Guardian
		limit = { situation:kin_strife.situation_participant_group:loyalist_leader = { any_situation_group_participant = { count > 0 has_relation_guardian = scope:curr_character } } }
		add = 1
	}
	if = { #Friend
		limit = { situation:kin_strife.situation_participant_group:loyalist_leader = { any_situation_group_participant = { count > 0 has_relation_friend = scope:curr_character } } }
		add = 3
	}
	if = { #Crush
		limit = { situation:kin_strife.situation_participant_group:loyalist_leader = { any_situation_group_participant = { count > 0 has_relation_crush = scope:curr_character } } }
		add = 3
	}
	if = { #Best Friend
		limit = { situation:kin_strife.situation_participant_group:loyalist_leader = { any_situation_group_participant = { count > 0 has_relation_best_friend = scope:curr_character } } }
		add = 6
	}
	if = { #Lover
		limit = { situation:kin_strife.situation_participant_group:loyalist_leader = { any_situation_group_participant = { count > 0 has_relation_lover = scope:curr_character } } }
		add = 6
	}
	if = { #Soulmate
		limit = { situation:kin_strife.situation_participant_group:loyalist_leader = { any_situation_group_participant = { count > 0 has_relation_soulmate = scope:curr_character } } }
		add = 12
	}
	### Negative Relationships ###
	if = { #Bully
		limit = { situation:kin_strife.situation_participant_group:loyalist_leader = { any_situation_group_participant = { count > 0 has_relation_bully = scope:curr_character } } }
		add = -1
	}
	if = { #Victim
		limit = { situation:kin_strife.situation_participant_group:loyalist_leader = { any_situation_group_participant = { count > 0 has_relation_victim = scope:curr_character } } }
		add = -2
	}
	if = { #Grudge
		limit = { situation:kin_strife.situation_participant_group:loyalist_leader = { any_situation_group_participant = { count > 0 has_relation_grudge = scope:curr_character } } }
		add = -2
	}
	if = { #Rival
		limit = { situation:kin_strife.situation_participant_group:loyalist_leader = { any_situation_group_participant = { count > 0 has_relation_rival = scope:curr_character } } }
		add = -6
	}
	if = { #Nemesis
		limit = { situation:kin_strife.situation_participant_group:loyalist_leader = { any_situation_group_participant = { count > 0 has_relation_nemesis = scope:curr_character } } }
		add = -9
	}
}

lotr_kin_strife_join_loyalist_group_chance_house = {
	value = 0
	if = {
		limit = { dynasty = { has_dynasty_modifier = kin_strife_support_loyalist } }
		add = 20
	}
	if = {
		limit = {
			situation:kin_strife.situation_participant_group:loyalist_group = {
				any_situation_group_participant = {
					count > 0
					house = scope:curr_character.house
				}
			}
		}
		situation:kin_strife.situation_participant_group:loyalist_group = {
			every_situation_group_participant = {
				limit = { house = scope:curr_character.house }
				add = 3
			}
		}
	}
}

}

	value = 0

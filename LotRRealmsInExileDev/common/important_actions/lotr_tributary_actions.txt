action_can_establish_protectorate = {
	priority = 800
	combine_into_one = yes
	check_create_action = {
		every_neighboring_top_liege_realm_owner = {
			limit = {
				root = {
					is_character_interaction_potentially_accepted = {
						recipient = prev
						interaction = offer_protection_interaction
					}
				}
			}
			try_create_important_action = {
				important_action_type = action_can_establish_protectorate
				actor = root
				recipient = this
			}
		}
		
	}

	effect = {
		open_interaction_window = {
			interaction = offer_protection_interaction
			actor = scope:actor
			recipient = scope:recipient
		}
	}
}

action_can_swear_fealty = {
	priority = 900
	check_create_action = {
		if = {
			limit = {
				has_realm_law = camp_purpose_rangers
				NOT = { subject_contract_has_flag = has_ranger_camp_contract }
				domicile.domicile_location.county.holder.top_liege ?= {
					same_alignment = { TARGET = root }
					OR = {
						primary_title = title:d_northern_dunedain
						AND = {
							primary_title.tier > tier_duchy
							realm_size >= minor_realm_size
						}
					}
				}
			}
			try_create_important_action = {
				important_action_type = action_can_swear_fealty
				actor = root
				recipient = domicile.domicile_location.county.holder.top_liege
			}
		}
	}

	effect = {
		open_interaction_window = {
			interaction = offer_fealty_interaction
			actor = scope:actor
			recipient = scope:recipient
		}
	}
}

action_upgrade_settlement_county = {
	priority = 900
	combine_into_one = yes
	check_create_action = {
		every_held_county = {
			limit = {
				is_landless_type_title = no
				is_noble_family_title = no
				title_province = { has_building = settlement_03 }
				development_level >= 3
			}
			try_create_important_action = {
				important_action_type = action_upgrade_settlement_county
				landed_title = this
				actor = root
			}
		}
	}

	effect = {
		scope:landed_title.title_province = {
			open_view_data = {
				view = holding_view
			}
		}
	}
}

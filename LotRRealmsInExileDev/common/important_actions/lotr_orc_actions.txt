action_lotr_no_orc_second_in_command = {
	type = alert

	check_create_action = {
		if = {
			limit = {
				NOT = { has_diarchy_type = lotr_orc_second_in_command }
				is_orc = yes
				is_landed = yes
			}
			try_create_important_action = {
				important_action_type = action_lotr_no_orc_second_in_command
				actor = root
				recipient = root
			}
		}
	}

	effect = {
		scope:actor = {
			open_view_data = {
				view = my_realm
				data = flag:bookmark_succession
			}
		}
	}
}

action_lotr_increase_orcish_law = {
	type = alert
	icon = action_upgrade_dominance

	check_create_action = {
		if = {
			limit = {
				#has_mpo_dlc_trigger = yes
				government_has_flag = government_is_orcish
				OR = {
					NOT = { has_variable = orc_authority_cooldown }
					culture = { has_innovation = innovation_all_things }
				}
				OR = {
					AND = {
						has_realm_law = orc_authority_1
						prestige >= increase_orc_authority_prestige_cost
					}
					AND = {
						has_realm_law = orc_authority_2
						prestige >= increase_orc_authority_prestige_cost
					}
					AND = {
						has_realm_law = orc_authority_3
						prestige >= increase_orc_authority_prestige_cost
					}
					AND = {
						has_realm_law = orc_authority_4
						prestige >= {
							value = increase_orc_authority_prestige_cost
							multiply = 2
						}
					}
				}
			}
			try_create_important_action = {
				important_action_type = action_lotr_increase_orcish_law
				actor = root
			}
		}
	}

	effect = {
		scope:actor = {
			open_view_data = {
				view = my_realm
				data = flag:bookmark_laws
			}
		}
	}
}

lotr_haradrim_legacy_track = {
	is_shown = {
		dynasty = {
			OR = {
				dynast.culture = {
					OR = {
						# Anyone with some haradric blood
						this = culture:apysaic
						any_parent_culture_or_above = {
							this = culture:apysaic
						}
					}
				}
				has_dynasty_perk = lotr_haradrim_legacy_1
			}
		}
	}
}

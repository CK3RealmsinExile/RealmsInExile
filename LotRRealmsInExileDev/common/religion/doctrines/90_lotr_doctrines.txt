### PEOPLE OF DARKNESS DOCTRINES ###

# Trig Note - These need to be first so they override all other hosility modifiers
is_adulation_of_the_dark_faith = {
	group = "special"
	
	is_available_on_create = {
		has_doctrine = special_doctrine_is_adulation_of_the_dark_faith
	}

	special_doctrine_is_adulation_of_the_dark_faith = {
		parameters = {
			# Since doctrines cannot modify other faith's views of us, only our view of others, all Adulation of the Dark faiths (aka "official" Sauronic or Melkorite faiths) are given this special doctrine which makes them friendlier to Men of Darkness Faiths with the Dominion of the Dark Influence tenet.
			hostility_override_special_doctrine_is_survival_of_the_strong_faith = 1
			hostility_override_special_doctrine_is_adulation_of_the_dark_faith = 0

			hostility_override_doctrine_echoes_of_the_ainur = 3
			hostility_override_doctrine_a_people_apart = 2
			hostility_override_doctrine_dominion_of_the_dark = 1
			# For loc:
			is_adulation_of_the_dark_faith_righteous_desc = 1
			is_adulation_of_the_dark_faith_astray_desc = 1
			is_adulation_of_the_dark_faith_hostile_desc = 1
			is_adulation_of_the_dark_faith_evil_desc = 1
		}
	}
}

doctrine_influence = {
	group = "main_group"

	is_available_on_create = {
        religion = {
            is_in_family = rf_darkness
        }
    }

	doctrine_echoes_of_the_ainur = {
		piety_cost = {
			value = faith_doctrine_cost_massive
			if = {
				limit = { has_doctrine = doctrine_echoes_of_the_ainur }
				multiply = faith_unchanged_doctrine_cost_mult
			}
			else_if = {
				limit = { has_doctrine = doctrine_dominion_of_the_dark }
				multiply = faith_changed_doctrine_cost_mult_two_step
			}
		}
		parameters = {
			
			hostility_override_special_doctrine_is_survival_of_the_strong_faith = 3 # Evil
			hostility_override_special_doctrine_is_adulation_of_the_dark_faith = 3 # Evil
			hostility_override_special_doctrine_is_freepeople_faith = 1 # Astray
			hostility_override_doctrine_major_branch_dwarven = 1 # Astray

			hostility_override_doctrine_echoes_of_the_ainur = 1 # Astray
			hostility_override_doctrine_a_people_apart = 2 # Hostile
			hostility_override_doctrine_dominion_of_the_dark = 3 # Evil

			#Localisation:
			doctrine_echoes_of_the_ainur_astray_desc = 0
			doctrine_echoes_of_the_ainur_hostile_desc = 0
			doctrine_echoes_of_the_ainur_evil_desc = 0
		}
	}

	doctrine_a_people_apart = {
		piety_cost = {
			value = faith_doctrine_cost_massive
			if = {
				limit = { has_doctrine = doctrine_a_people_apart }

			}
		}
		parameters = {
			# Brice comment : redundant values as MoD religion is already hostile to all other religions
			# hostility_override_special_doctrine_is_freepeople_faith = 2 # Hostile
			hostility_override_doctrine_echoes_of_the_ainur = 2 # Hostile
			# hostility_override_doctrine_a_people_apart = 2 # Hostile
			hostility_override_doctrine_dominion_of_the_dark = 2 # Hostile
			# hostility_override_special_doctrine_is_survival_of_the_strong_faith = 2 # Hostile
			# hostility_override_special_doctrine_is_adulation_of_the_dark_faith = 2 # Hostile
			#Localisation:
			doctrine_a_people_apart_astray_desc = 0 # Other "a people apart" OF SAME RELIGION are astray
			doctrine_a_people_apart_hostile_desc = 0
		}
	}

	doctrine_dominion_of_the_dark = {
		piety_cost = {
			value = faith_doctrine_cost_massive
			if = {

			hostility_override_special_doctrine_is_freepeople_faith = 3 # Evil
			hostility_override_doctrine_major_branch_dwarven = 3 # Evil

			hostility_override_doctrine_echoes_of_the_ainur = 3 # Evil

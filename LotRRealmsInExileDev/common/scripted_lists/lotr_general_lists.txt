	base = living_character
	conditions = { 
		is_from_ruler_designer = no
		OR = {
			not_valar = yes
			not_nazgul = yes
			not_istari = yes
			not_istari_evil = yes
			not_maiar = yes
			not_maiar_fallen = yes
		}
	}
}

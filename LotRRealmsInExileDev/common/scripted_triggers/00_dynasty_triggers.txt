

is_house_head = { # For unevaluated triggers
	trigger_if = {  
		limit = {
			exists = this
		}
		save_temporary_scope_as = house_check
		exists = house
		house = {
			house_head = scope:house_check
		}
	}
}

is_dynast = {
	save_temporary_scope_as = dynast_check
	custom_description = {
		text = is_dynast
		subject = scope:dynast_check

		exists = dynasty
		dynasty = {
			dynast = scope:dynast_check
		}
	}
}

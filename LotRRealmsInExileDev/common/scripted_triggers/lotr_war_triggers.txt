dwarf_and_goblin_underground_county_cb_trigger = {
	trigger_if = {
		limit = {
			scope:target = { title_province = { terrain = halls } }
		}
		scope:attacker = {
			OR = {
				is_goblin = yes
				is_dwarf = yes
			}	
		}
		scope:defender = {
			OR = {
				is_goblin = yes
				is_dwarf = yes
			}	
		}
	}
}

}

underground_cb_trigger = {
	OR = {
		AND = {
			$CHARACTER$ = { is_ai = no }
			any_sub_realm_county = {
				tier = tier_county
				any_neighboring_county = {
					this = $TARGET$
				}
			}
		}
		any_neighboring_county = {
			holder = { target_is_same_character_or_above = $CHARACTER$ }
		}
		$TARGET$.tier = tier_duchy
	}
}

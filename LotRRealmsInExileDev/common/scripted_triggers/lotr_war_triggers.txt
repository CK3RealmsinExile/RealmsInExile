dwarf_and_goblin_underground_county_cb_trigger = {
	trigger_if = {
		limit = {
			scope:target = { title_province = { terrain = halls } }
		}
		scope:attacker = {
			OR = {
				is_goblin = yes
				is_dwarf = yes
			}	
		}
		scope:defender = {
			OR = {
				is_goblin = yes
				is_dwarf = yes
			}	
		}
	}
}

should_use_nomadic_music_trigger = {
	OR = {
		# culture = {
		# 	has_cultural_pillar = heritage_mongolic
		# }
		# culture = {
		# 	has_cultural_pillar = heritage_turkic
		# }
		# faith.religion = faith:tengri_pagan.religion
		lotr_steppe_cultures_trigger = yes
		faith.religion = religion:talataerim_religion
		AND = {
			exists = root.capital_county
			government_has_flag = government_is_nomadic
			situation:the_great_steppe = {
				any_situation_sub_region = {
					situation_sub_region_has_county = root.capital_county
				}
			}
		}
	}
}

should_use_iberian_music_trigger = {
	OR = {
		culture = {
			OR = {
				has_cultural_pillar = heritage_northern_adunaic
				has_cultural_pillar = heritage_bellakari
				has_cultural_pillar = heritage_southern_adunaic
			}
			# has_cultural_pillar = heritage_iberian
		}
		# culture = {
		# 	any_parent_culture_or_above = { has_cultural_pillar = heritage_iberian }
		# }
		AND = {
			exists = capital_province
			# capital_province = { geographical_region = world_europe_west_iberia }
			capital_province = { 
				OR = {
					geographical_region = middleearth_harad_near_umbar
					geographical_region = middleearth_harad_far_bellakar
					geographical_region = middleearth_uttersouth_thanihazad
				}
			}
		}
	}
}

should_use_iranian_music_trigger = {
	OR = {
		lotr_talathrim_cultures_trigger = yes
		AND = {
			exists = capital_province
			capital_province = { 
				OR = {
					geographical_region = middleearth_rhun_seaofrhun
					geographical_region = middleearth_rhun_khand
					geographical_region = middleearth_harad_bayoformal
					geographical_region = middleearth_harad_near
				}
			}
		}
		culture = {
			# has_cultural_pillar = heritage_iranian
			has_cultural_pillar = heritage_northern_apysaic
		}
	}
}

### General Triggers ###

is_in_angmar_wars = { # Character scope
	any_character_situation = {
		situation_type = angmar_wars_situation
	}
}

title_is_in_angmar_wars = { # Title scope
	holder ?= {
		is_in_angmar_wars = yes
	}
}

is_arnorian_ruler = { # Character scope
	# ROOT scope. Saving this to avoid errors in certain cases. 
	# The trigger_if does not check any triggers, so it is ONLY used to save the character scope. This is vanilla confention. Blame them ¯\_(ツ)_/¯
	trigger_if = { limit = { save_temporary_scope_as = character_scope } }
	any_character_situation = {
		situation_type = angmar_wars_situation
		any_situation_sub_region = {
			any_situation_sub_region_participant_group = {
				participant_group_type = arnorian_rulers
				participant_group_has_character = scope:character_scope
			}
		}
	}
}

is_arnor_aligned = {
	OR = {
		is_arnorian_ruler = yes
		is_elven_ruler = yes
	}
}

is_angmar_aligned = {
	OR = {
		is_angmar_ruler = yes
	}
}

angmar_wars_is_opposite_side = { # From character scope, takes CHARACTER = comparison
	trigger_if = {
		limit = {
			is_angmar_aligned = yes
		}
		$CHARACTER$ = {
			is_arnor_aligned = yes
		}
	}
	trigger_else_if = {
		limit = {
			is_arnor_aligned = yes
		}
		$CHARACTER$ = {
			is_angmar_aligned = yes
		}
	}
	trigger_else = {
		always = no # For neutrals and such
	}
}

# Phase Triggers 

is_stalemate = {
	any_character_situation = {
		any_situation_sub_region = {
			sub_region_current_phase = angmar_wars_stalemate
		}
	}
}

# Random Triggers

angmar_wars_allowed_cbs_trigger = { # From CB triggers, takes scope:attacker and scope:defender
	NOT = { 
		scope:attacker = {
			angmar_wars_is_opposite_side = { CHARACTER = scope:defender }
		}
	}
}

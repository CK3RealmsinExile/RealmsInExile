lotr_is_shown_basic_activity_trigger = {
	is_nazgul = no 
	is_in_valinor = no
	is_wastelands = no
	NOT = { has_trait = sauron_ringless }
}

lotr_is_shown_inspection_activity_trigger = {
	NOT = { has_trait = sauron_ringless }
	is_in_valinor = no #LotR
	is_lotr_adult = yes
}

lotr_is_shown_pilgrimage_activity_trigger = {
	lotr_is_shown_basic_activity_trigger = yes
	faith = { #LotR
		NOT = { has_doctrine_parameter = forbidden_from_pilgrimage }
		any_holy_site = { exists = this }	
	}
}

lotr_is_shown_adult_education_activity_trigger = {
	is_lotr_adult = yes #LotR
	NOT = { #LotR
		is_orc = yes # Orcs don't go to college!
		cannot_be_killed = yes
		is_in_valinor = yes
	}
}

lotr_can_sail = {
	NOR = {
		is_maiar_fallen = yes
		is_orc = yes # Orcs hate water
		is_hobbit = yes # Most hobbits aren't fond of boating
		is_druedain = yes # They don't build ships
		trigger_if = { # Can only trigger if you're a dwarf holding Kheledul (directly)
			limit = { is_dwarf = yes }
			NOT = { has_title = title:c_kheledul }
		}
		faith = { has_doctrine_parameter = no_sailing } # Obv
	}
}

lotr_has_visible_disfigurement_trigger = {
	OR = {
		has_inactive_trait = visible_disfigurement_1
		has_inactive_trait = visible_disfigurement_2
		has_inactive_trait = visible_disfigurement_3
		has_inactive_trait = visible_disfigurement_4
		has_inactive_trait = visible_disfigurement_5
	}
}

lotr_visible_disfigurement_trigger = {
	has_trait = disfigured
	NOT = {
		lotr_has_visible_disfigurement_trigger = yes
	}
}

lotr_visible_disfigurement_bandages_trigger = {
	has_trait = disfigured
	OR = {
		lotr_has_visible_disfigurement_trigger = no
		has_character_flag = lotr_force_disfigurement_bandages
	}
}

lotr_visible_disfigurement_fully_masked_trigger = {
	OR = {
		has_character_flag = disfigurement_fully_masked
		has_inactive_trait = disfigurement_fully_masked
	}
}

lotr_visible_disfigurement_partially_masked_trigger = {
	OR = {
		AND = {
			has_character_modifier = wearing_disfigurement_mask_modifier
			NOT = { has_character_flag = disfigurement_fully_masked }
		}
		has_inactive_trait = disfigurement_partially_masked
	}
}

lotr_has_any_eye_patch_trigger = {
	has_trait = one_eyed
	OR = {
		has_character_flag = beric_eyepatch
		has_character_flag = decorated_eyepatch
		has_character_flag = standard_eyepatch
		has_character_flag = round_eyepatch
		has_character_flag = triangle_eyepatch
		has_character_flag = eye_bandage_clean

		has_inactive_trait = beric_eyepatch
		has_inactive_trait = decorated_eyepatch
		has_inactive_trait = standard_eyepatch
		has_inactive_trait = round_eyepatch
		has_inactive_trait = triangle_eyepatch
		has_inactive_trait = eye_bandage_clean
	}
}

not_has_county_on_island_region = {
	OR = {
		NOR = {
			any_held_title = {
				target_is_de_jure_liege_or_above = title:d_andunaloni
			}
			any_vassal_or_below = {
				any_held_title = {
					target_is_de_jure_liege_or_above = title:d_andunaloni
				}
			}
		}
		AND = {
			any_held_title = {
				target_is_de_jure_liege_or_above = title:d_andunaloni
				any_county_province = {
					has_holding_type = settlement_holding
				}
			}
			NOT = {
				any_held_title = {
					target_is_de_jure_liege_or_above = title:d_andunaloni
					any_county_province = {
						OR = {
							has_holding_type = castle_holding
							has_holding_type = elven_holding
							has_holding_type = church_holding
							has_holding_type = city_holding
						}
					}
				}
			}
		}
	}
}

}

above_settlement_limit = {
	trigger_if = {
		limit = { NOT = { dynasty ?= { has_dynasty_modifier = easy_arnor_settle } } }
		trigger_if = {
			limit = { highest_held_title_tier = tier_county }
			any_held_title = {
				tier = tier_county
				title_province ?= { has_holding_type = settlement_holding }
				count > 1
			}
		}
		trigger_else_if = {
			limit = { highest_held_title_tier = tier_duchy }
			any_held_title = {
				tier = tier_county
				title_province ?= { has_holding_type = settlement_holding }
				count > 2
			}
		}
		trigger_else_if = {
			limit = { highest_held_title_tier = tier_kingdom }
			any_held_title = {
				tier = tier_county
				title_province ?= { has_holding_type = settlement_holding }
				count > 3
			}
		}
		trigger_else = {
			any_held_title = {
				tier = tier_county
				title_province ?= { has_holding_type = settlement_holding }
				count > 4
			}
		}
	}
	trigger_else = {
		trigger_if = {
			limit = { highest_held_title_tier = tier_county }
			any_held_title = {
				tier = tier_county
				title_province ?= { has_holding_type = settlement_holding }
				count > 3
			}
		}
		trigger_else_if = {
			limit = { highest_held_title_tier = tier_duchy }
			any_held_title = {
				tier = tier_county
				title_province ?= { has_holding_type = settlement_holding }
				count > 4
			}
		}
		trigger_else_if = {
			limit = { highest_held_title_tier = tier_kingdom }
			any_held_title = {
				tier = tier_county
				title_province ?= { has_holding_type = settlement_holding }
				count > 5
			}
		}
		trigger_else = {
			any_held_title = {
				tier = tier_county
				title_province ?= { has_holding_type = settlement_holding }
				count > 6
			}
		}
	}
}

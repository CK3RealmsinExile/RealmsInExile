### Martial ###
lotr_orc_marshal_exists = {
    trigger_if = {
        limit = { is_orc = yes }
        scope:councillor = {
            OR = {
                is_performing_council_task = task_orc_1_martial_1
                is_performing_council_task = task_orc_1_martial_2
                is_performing_council_task = task_orc_1_martial_3

                is_performing_council_task = task_orc_2_martial_1
                is_performing_council_task = task_orc_2_martial_2
                is_performing_council_task = task_orc_2_martial_3
                
                is_performing_council_task = task_orc_3_martial_1
                is_performing_council_task = task_orc_3_martial_2
                is_performing_council_task = task_orc_3_martial_3
                
                is_performing_council_task = task_orc_4_martial_1
                is_performing_council_task = task_orc_4_martial_2
                is_performing_council_task = task_orc_4_martial_3
            }
        }
    }
    trigger_else = {
        always = no
    }
}

lotr_orc_marshal_base_trigger_exists = {
    OR = {
        AND = {
            exists = cp:councillor_marshal
            scope:councillor = cp:councillor_marshal
        }
        lotr_orc_marshal_exists = yes
    }
}

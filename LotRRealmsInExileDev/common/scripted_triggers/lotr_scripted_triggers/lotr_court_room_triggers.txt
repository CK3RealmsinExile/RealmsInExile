has_dunedain_court_illustration_trigger = {
	trigger_if = {
		limit = { exists = capital_province }
		AND = {
			capital_province.culture = { has_cultural_pillar = heritage_dunedain }
			OR = {
				capital_province = { geographical_region = middleearth_west_gondor }
				capital_province = { geographical_region = middleearth_west_arnor }
				capital_province = province:20 #Orthanc
			}
		}
	}
	trigger_else_if = {
		limit = { NOT = { exists = capital_province } }
		trigger_if = { 
			limit = { exists = host.capital_province }
			AND = {
				host.capital_province.culture = { has_cultural_pillar = heritage_dunedain }
				OR = {
					host.capital_province = { geographical_region = middleearth_west_gondor }
					host.capital_province = { geographical_region = middleearth_west_arnor }
				}
			}
		}
		trigger_else_if = {
			limit = { exists = liege.capital_province }
			AND = {
				liege.capital_province.culture = { has_cultural_pillar = heritage_dunedain }
				OR = {
					liege.capital_province = { geographical_region = middleearth_west_gondor }
					liege.capital_province = { geographical_region = middleearth_west_arnor }
				}
			}
		}
		# trigger_else_if = { #Beardless note #Commented out because it's not based just on location
		#	limit = { exists = location.province_owner.capital_province }
		#	OR = {
		#		location.province_owner.capital_province = { geographical_region = middleearth_west_gondor }
		#		location.province_owner.capital_province = { geographical_region = middleearth_west_arnor }
		#	}
		# }
		trigger_else = { always = yes }
	}
	trigger_else = { always = yes }
}

has_dwarven_court_illustration_trigger = {
	OR = {
		capital_county.culture = {
			OR = {
				has_cultural_pillar = heritage_longbeard
				has_cultural_pillar = heritage_firebeard
				has_cultural_pillar = heritage_broadbeam
				has_cultural_pillar = heritage_ironfist
				has_cultural_pillar = heritage_stiffbeard
				has_cultural_pillar = heritage_blacklock
				has_cultural_pillar = heritage_stonefoot
			}
		}
		trigger_if = {
			limit = { capital_county.culture = { is_orc_culture = yes } }
			capital_county.title_province = { terrain = halls }
		}
	}
}

coa_dunland_kingdom_trigger = {
	OR = {
		scope:culture = { has_coa_gfx = dunlending_group_coa_gfx }
		AND = {
			exists = primary_title
			primary_title = {
				any_this_title_or_de_jure_above = {
					this = title:k_dunland
				}
			}
		}
	}
}

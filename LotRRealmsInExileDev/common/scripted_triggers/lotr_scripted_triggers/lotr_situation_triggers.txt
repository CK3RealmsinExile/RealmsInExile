has_highest_hope = { #T5 hope
	# ROOT scope. Saving this to avoid errors in certain cases. 
	# The trigger_if does not check any triggers, so it is ONLY used to save the character scope. This is vanilla confention. Blame them ¯\_(ツ)_/¯
	trigger_if = { limit = { save_temporary_scope_as = character_scope } }
	any_character_situation = {
		any_situation_sub_region = {
			sub_region_current_phase = situation_elven_thriving
			situation_sub_region_has_county = scope:character_scope.capital_county
		}
	}
}

### Kin-Strife Situation ###
############################

lotr_can_be_loyalist_leader = {
	is_wastelands = no
	has_variable = kin_strife_participant
	var:kin_strife_participant = flag:loyalist_leader
	has_trait = kin_strife_loyalist
}

lotr_can_be_purist_leader = {
	is_wastelands = no
	has_variable = kin_strife_participant
	var:kin_strife_participant = flag:purist_leader
	has_trait = kin_strife_purist
}

lotr_can_be_loyalist_group = {
	is_wastelands = no
	lotr_can_be_loyalist_leader = no
	has_variable = kin_strife_participant
	var:kin_strife_participant = flag:loyalist_group
	has_trait = kin_strife_loyalist
}

lotr_can_be_purist_group = {
	is_wastelands = no
	lotr_can_be_purist_leader = no
	has_variable = kin_strife_participant
	var:kin_strife_participant = flag:purist_group
	has_trait = kin_strife_purist
}

lotr_can_be_neutral_group = {
	is_wastelands = no
	lotr_can_be_loyalist_leader = no

	has_variable = kin_strife_participant

}

	NOT = { is_spouse_of = root }

lotr_can_adopt_wanua_trigger = {
	OR = { 
		culture = {
			OR = { # required to make tooltip work
				has_cultural_tradition = tradition_seafaring
				has_cultural_tradition = tradition_maritime_mercantilism
				has_cultural_tradition = tradition_fishermen
				has_cultural_tradition = tradition_maritime_way_of_life
			}
		}
		faith = {
			OR = { # required to make tooltip work
				has_doctrine = tenet_astrology
				has_doctrine = tenet_sacred_waters
			}
		}
	}
}

build_abduct_targets_trigger = {
	any_relation = { # Rivals are good targets for abduct schemes
		type = rival
		count = all
		trigger_if = {
			limit = {
				root = {
					ai_honor <= medium_positive_ai_value
					can_start_scheme = {
						target_character = prev
						type = abduct
					}
				}
			}
			add_to_temporary_list = abduct_targets
		}
	}
	any_relation = { # Same with nemeses
		type = nemesis
		count = all
		trigger_if = {
			limit = {
				root = {
					ai_honor <= high_positive_ai_value
					can_start_scheme = {
						target_character = prev
						type = abduct
					}
				}
			}
			add_to_temporary_list = abduct_targets
		}
	}
	any_neighboring_and_across_water_realm_same_rank_owner = { # Kidnap a neighbouring rulers primary heir to ransom them off 
		trigger_if = {
			limit = {
				exists = primary_heir
				primary_heir.dynasty = this.dynasty
				root = { ai_greed >= low_positive_ai_value }
				primary_heir = { add_to_temporary_list = abduct_targets }
			}
		}
	}
	any_neighboring_and_across_water_top_liege_realm_owner = { # Kidnap a neighbouring rulers primary heir to ransom them off 
		trigger_if = {
			limit = {
				exists = primary_heir
				primary_heir.dynasty = this.dynasty
				root = { ai_greed >= low_positive_ai_value }
				primary_heir = { add_to_temporary_list = abduct_targets }
			}
		}
	}
	trigger_if = { # Kidnap the Top Lieges priamry heir
		limit = { 
			exists = root.top_liege 
			exists = root.top_liege.primary_heir
			top_liege.primary_heir.dynasty = top_liege.dynasty
			OR = {
				ai_greed >= medium_positive_ai_value
				opinion = {
					target = top_liege
					value <= high_negative_opinion
				}
			}
			ai_rationality >= low_positive_ai_value
			top_liege.primary_heir = { add_to_temporary_list = abduct_targets }
		}
	}
	trigger_if = { # If your liege is NOT your Top liege, consider kidnapping him as-well
		limit = {
			exists = liege
			NOT = { liege = top_liege }
			OR = {
				ai_greed >= medium_positive_ai_value
				opinion = {
					target = liege
					value <= high_negative_opinion
				}
			}
			ai_rationality >= medium_positive_ai_value
			liege.primary_heir = {
				add_to_temporary_list = abduct_targets
			}
		}
	}
	trigger_if = { # Abducting troublesome claimants is something you might want to do...
		limit = {
			is_landed = yes
			any_targeting_faction = {
			  	faction_is_type = claimant_faction
				special_character = {
					add_to_temporary_list = abduct_targets
				}
			}
		}
	}
	trigger_if = { # Paranoid characters want to abduct_targets claimants to their titles
		limit = {
			has_trait = paranoid
			ai_compassion <= low_negative_ai_value
			ai_rationality <= 0
			is_landed = yes
			any_held_title = {
				any_claimant = { add_to_temporary_list = abduct_targets }
			}
		}
	}
}

lotr_build_orc_murder_targets_trigger = {
	any_relation = { # Rivals are good targets for murder schemes
		type = rival
		count = all
		trigger_if = {
			limit = {
				root = {
					can_start_scheme = {
						target_character = prev
						type = murder
					}
				}
			}
			add_to_temporary_list = murder_targets
		}
	}
	any_relation = { # Same with nemeses
		type = nemesis
		count = all
		trigger_if = {
			limit = {
				root = {
					can_start_scheme = {
						target_character = prev
						type = murder
					}
				}
			}
			add_to_temporary_list = murder_targets
		}
	}
	trigger_if = { # If you are the Second-in-Command, consider killing your liege
		limit = { 
			exists = root.liege
			root.liege = { has_diarchy_type = lotr_orc_second_in_command }
			root = { is_diarch = yes }
			root.liege.diarch = root
			ai_greed >= low_positive_ai_value
			liege = { add_to_temporary_list = murder_targets }
		}
	}
	trigger_if = { # Killing troublesome faction leaders is something you might want to do...
		limit = {
			is_landed = yes
			any_targeting_faction = {
				exists = yes
				OR = {
					faction_is_type = independence_faction
					faction_is_type = nation_fracturing_faction
					faction_is_type = liberty_faction
					faction_is_type = claimant_faction
				}
				special_character = { add_to_temporary_list = murder_targets }
			}
		}
	}
	trigger_if = { # Paranoid characters want to kill claimants to their titles.
		limit = {
			has_trait = paranoid
			ai_rationality <= 0
			is_landed = yes
			any_held_title = {
				any_claimant = { add_to_temporary_list = murder_targets }
			}
		}
	}
	trigger_if = { # Orcs want to POTENTIALLY kill strong vassals to secure their ruke.
		limit = {
			is_landed = yes
			any_powerful_vassal = {
				OR = {
					martial > root.martial
					intrigue > root.intrigue
					legitimacy_level > root.legitimacy_level
					current_military_strength > root.current_military_strength
				}
				save_temporary_scope_as = murder_target
			}
			OR = {
				has_trait = paranoid
				ai_rationality <= 0
				opinion = {
					target = scope:murder_target
					value <= high_negative_opinion
				}
			}
			scope:murder_target = { add_to_temporary_list = murder_targets }
		}
	}
}

### New Academy Triggers ###

## Harlond
# Show Triggers
show_harlond_academy_trigger = {
	title:b_harlond_gondor.title_province = { has_special_building_slot = no }
}

# Valid Triggers
valid_harlond_academy_trigger = {
    title:b_harlond_gondor.title_province = { has_special_building_slot = no }
    title:b_harlond_gondor = { county = { development_level >= 30 } }
    title:b_harlond_gondor.holder = { faith_is_good = yes }
}

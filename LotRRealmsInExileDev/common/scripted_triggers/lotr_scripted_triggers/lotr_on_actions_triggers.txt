lotr_on_title_lost_add_to_western_adventurer_list = {
    OR = {
        AND = {	#Orcs
            has_game_rule = northern_orcs_invade								
            AND = {
                culture = { has_cultural_pillar = heritage_orc } #OrcAdventure
                NOR = {
                    culture = culture:rukhashai
                    culture = { any_parent_culture_or_above = { this = culture:rukhashai } }
                }
            }							
        }						
        AND = {	#Goblins
            has_game_rule = northern_orcs_invade								
            AND = {
                culture = { has_cultural_pillar = heritage_goblin } #OrcAdventure
                NOR = {
                    culture = culture:echomaker_goblin
                    culture = { any_parent_culture_or_above = { this = culture:echomaker_goblin } }
                }
            }							
        }
        AND = {	#Eotheod						
            has_game_rule = northern_orcs_no_invade
            culture = { has_cultural_pillar = heritage_eotheod }				
        }		
        
        AND = {	#Daen Lintis
            has_game_rule = northern_orcs_no_invade
            culture = { has_cultural_pillar = heritage_daen_lintis } #LotR								
        }
        
        AND = {	#Daen Coentis
            has_game_rule = northern_orcs_no_invade
            culture = { has_cultural_pillar = heritage_daen_coentis } #LotR								
        }
    }
}

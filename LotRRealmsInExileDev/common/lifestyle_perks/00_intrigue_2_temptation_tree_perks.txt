﻿like_weed_in_a_garden_perk = {
	lifestyle = intrigue_lifestyle
	tree = temptation
	position = { 0 0 }
	icon = node_intrigue

	name = { #LOTR
		first_valid = {
			triggered_desc = {
				trigger = {
					exists = this
					is_elf = yes
				}
				desc = like_weed_in_a_garden_perk_elf_name #LOTR - Radiance of the Eldar
			}
			desc = like_weed_in_a_garden_perk_name
		}
	}
	
	auto_selection_weight = {
		value = 11
		if = {
			limit = {
				has_education_intrigue_trigger = yes
			}
			add = 1989
		}
		if = {
			limit = {
				has_focus = intrigue_temptation_focus
			}
			multiply = 5
		}
		if = {
			limit = {
				can_start_new_lifestyle_tree_trigger = no
				NOT = { has_perk = enticing_opportunity_perk }
			}
			multiply = 0
		}
	}
	
	can_be_picked = {
		always = yes
	}
	
	character_modifier = {
		fertility = 0.3
	}
	culture_character_modifier = { #LOTR
		parameter = is_elven_heritage
		fertility = -0.3  #Counteract vanilla buffs
		enemy_hostile_scheme_phase_duration_add = 15
		attraction_opinion = 25
	}
}

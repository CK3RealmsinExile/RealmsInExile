	lifestyle = learning_lifestyle
	tree = medicine
	position = { 2 0 }
	icon = node_learning

		if = {
			limit = {
				has_education_learning_trigger = yes
			}

			}

		}
	}
	
	effect = {

			text = anatomical_studies_perk_effect
		}
	}
}

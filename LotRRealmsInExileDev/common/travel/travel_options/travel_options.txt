experienced_captains_option = {	#LotR TBD: Add mariner trait(s) in addition to forder trait
	is_shown = {
   		is_ruler = yes
	}

	is_valid = {
	}
	
	cost = {
		gold = {
			add = {
				value = medium_travel_option_cost
				desc = experienced_captains_option
			}
		}
	}

	owner_modifier = {
		coastal_sea_travel_danger = medium_sea_danger_reduction
		sea_travel_danger = medium_sea_danger_reduction #Still a bit more dangerous since sea travel is more dangerous than coastal sea
	}
	
	on_applied_effect = {
		hidden_effect = {
			if = { # No reason to character bloat for every AI
				limit = {
					OR = {
						is_ai = no
						highest_held_title_tier >= tier_kingdom
					}
				}
				# Try to find a pool character with the Forder trait
				# LOTR - Mariner trait instead
				if = {
					limit = {
						exists = root.capital_province
						any_pool_character = {
							province = root.capital_province
							is_available_ai_adult = yes
							has_no_particular_noble_roots_trigger = yes
							age >= young_age
							age <= old_age
							NOT = { opposite_alignment = { TARGET = root } }
							is_married = no
							#has_trait = forder
							has_trait = lifestyle_mariner #LOTR
						}
					}
					random_pool_character = {
						province = root.capital_province
						limit = {
							is_available_ai_adult = yes
							has_no_particular_noble_roots_trigger = yes
							age >= young_age
							age <= old_age
							NOT = { opposite_alignment = { TARGET = root } }
							is_married = no
							#has_trait = forder
							has_trait = lifestyle_mariner #LOTR
						}
						save_scope_as = new_guide
					}
				}
				# If we don't have one, we pull a random pool character and add the trait
				else_if = {
					limit = {
						exists = root.capital_province
						any_pool_character = {
							province = root.capital_province
							is_available_ai_adult = yes
							has_no_particular_noble_roots_trigger = yes
							age >= young_age
							age <= old_age
							NOT = { opposite_alignment = { TARGET = root } }
							is_married = no
						}
					}
					random_pool_character = {
						province = root.capital_province
						limit = {
							is_available_ai_adult = yes
							has_no_particular_noble_roots_trigger = yes
							age >= young_age
							age <= old_age
							NOT = { opposite_alignment = { TARGET = root } }
							is_married = no
						}
						#if = {
						#	limit = {
						#		NOT = {
						#			has_trait = forder
						#		}
						#	}
						#	add_trait = forder
						#}
						if = { #LOTR - Mariner instead
							limit = {
								NOT = {
									has_trait = lifestyle_mariner
								}
							}
							add_trait = lifestyle_mariner
							add_trait_xp = {
								trait = lifestyle_mariner
								value = {
									integer_range = {
										min = small_lifestyle_random_xp_low
										max = small_lifestyle_random_xp_high
									}
								}
							}
						}
						save_scope_as = new_guide
					}
				}
				# Fallback scenario, we just create a character
				else = {
					create_character = {
						location = root.location
						template = sea_captain
						save_scope_as = new_guide
					}
				}
				if = {
					limit = {
						exists = scope:new_guide
					}
					scope:new_guide = {
						add_gold = root.medium_travel_option_cost
						move_to_pool = yes
						add_character_flag = travel_option_added_character	#to send it back to pool when the travel is done
					}
					send_interface_toast = {
						type = event_toast_effect_good
						title = captain_recruited.t
						left_icon = scope:new_guide
						root.current_travel_plan ?= {
							add_companion = scope:new_guide
						}
					}
				}
			}
		}
		custom_tooltip = captain_recruited.tt
	}
	
	ai_will_do = {
		value = 0
		if = {
			limit = {
				any_future_path_location = {
					count >= 3
					is_sea_province = yes
				}
			}
			add = {
				value = 100
			}
		}
	}
}

	}

	is_valid = {
		custom_tooltip = {
			text = available_mercenary.tt

				mercenary_company_leader = {
					is_travelling = no
					is_at_war = no
					is_in_army = no

					NOT = {

						}
					}
				}
			}
		}
	}

		}

		travel_safety = 20

			}

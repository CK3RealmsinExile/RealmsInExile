### Advice: Fabricate claim on <title>
#
fabricate_claim_advice = {
	decisions = { unify_region_decision }

	# Is this at all relevant advice?
	is_valid_advice = {
		exists = cp:councillor_court_chaplain
		cp:councillor_court_chaplain ?= {
			save_temporary_scope_as = chaplain
		}
	}

	# Is advice valid for this title?
	is_valid_for_title = {
		NOT = { completely_controls = scope:title }
		NOT = { has_claim_on = scope:title }
		scope:title.duchy = {
			save_temporary_scope_as = title_duchy
		}
	}

	# Are we already doing this?
	is_doing = {
		cp:councillor_court_chaplain ?= {
			trigger_if = {
				limit = {
					exists = councillor_task_target
				}
				councillor_task_target = {
					geographical_region = custom_ireland
				}
			}
			trigger_else = {
				always = no
			}
		}
	}

	# How relevant is this?
	relevance = {
		value = 10
		add = hint_title_importance_value
		if = {
			limit = {
				cp:councillor_court_chaplain.learning >= decent_skill_rating
			}
			add = 10
		}
		else = {
			add = -2
		}
		if = {
			limit = {
				is_character_interaction_potentially_accepted = {
					recipient = scope:title.holder
					interaction = offer_vassalization_interaction
				}
			}
			add = -10
		}
	}

	# Advice summary and description
	summary = hint_summary_fabricate_claim
	description = hint_fabricate_claim
}

#LOTR:	Removed same culture requirement to reduce the amount of hardcoded mayors
#		spawned for Sauron and Saruman.

pool_city_succession = {
	valid_character = {
		is_capable_adult = yes
		OR = {
			trigger_if = { #LotR
				limit = {
					scope:base = {
						NOR = {
							is_maiar = yes
							is_nazgul = yes
							is_sauron = yes
						}
					}
				}
				has_same_culture_as = scope:base
				scope:base.primary_title.title_province.culture ?= culture
			}
		}
		trigger_if = {
			limit = {
				scope:base = { has_realm_law = male_only_law }
			}
			is_male = yes
		}
		trigger_else_if = {
			limit = {
				scope:base = { has_realm_law = female_only_law }
			}
			is_female = yes
		}
		trigger_else = {}
		OR = {
			is_lowborn = yes
			AND = {
				exists = scope:base.dynasty
				dynasty ?= scope:base.dynasty
				is_courtier_of = scope:base
			}
			dynasty ?= {
				any_dynasty_member = {
					count <= 2
				}
			}
		}
		is_claimant = no
		is_imprisoned = no
		is_betrothed = no
	}
	
	character_score = {
		base = 1
		
		modifier = {
			add = 3000
			is_courtier_of = scope:base
		}
		modifier = {
			add = { value = age max = 50 }
		}
		modifier = {
			add = 20
			has_education_stewardship_trigger = yes
		}
		modifier = {
			factor = 5
			is_female = yes
			scope:base.faith = {
				has_doctrine = doctrine_gender_female_dominated
			}
		}
		modifier = {
			factor = 5
			is_male = yes
			scope:base.faith = {
				has_doctrine = doctrine_gender_male_dominated
			}
		}
	}
	
	
	config = {
		age = { 20 80 }
	}
	
	selection_count = 25
	
	pool = faith
	court = yes
	gender = gender_law
}

lotr_orc_government_set_second_in_command = {
	is_shown = { is_orc = yes }
	is_valid = { 
		OR = {
			any_vassal = { is_orc = yes }
			any_courtier_or_guest = { is_orc = yes }
			any_dynasty_member = {
				is_orc = yes
				in_diplomatic_range = root
			}
		}
	}
	effect = {
		clear_variable_list = possible_candidates
		ordered_vassal = {
			limit = { is_orc = yes }
			order_by = learning #TEMP
			max = 10
			min = 0
			check_range_bounds = no
			root = { add_to_variable_list = { name = possible_candidates target = prev } } 
		}

		ordered_courtier_or_guest = {
			limit = { is_orc = yes }
			order_by = learning #TEMP
			max = 10
			min = 0
			check_range_bounds = no
			root = { add_to_variable_list = { name = possible_candidates target = prev } } 
		}
		dynasty = {
			ordered_dynasty_member = {
				limit = { is_orc = yes in_diplomatic_range = root NOT = { this = root } }
				order_by = learning #TEMP
				max = 10
				min = 0
				check_range_bounds = no
				root = { add_to_variable_list = { name = possible_candidates target = prev } } 
			}
		}
	}
}

lotr_orc_government_clear_inheritance_list = {
	effect = { clear_variable_list = possible_candidates }
}

lotr_orc_government_text_conditions = {
	is_valid = { has_diarchy_type = lotr_orc_second_in_command }
	is_shown = { NOT = { has_diarchy_type = lotr_orc_second_in_command } }
}

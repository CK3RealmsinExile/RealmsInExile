recieving_tributary_gold = {
	is_valid = {
		exists = root
		is_tributary = no
		OR = {
			has_permanent_tributaries = yes
			has_non_permanent_tributaries = yes
		}
	}
}

set_title_as_selected = {
	is_shown = {
		trigger_if = {
			limit = { root.tier = tier_empire }
			any_in_global_list = {
				variable = selected_titles_wastelands
				this = root
			}
		}
		trigger_else = {
			trigger_if = {
				limit = { root.tier = tier_kingdom }
				OR = {
					any_in_global_list = {
						variable = selected_titles_wastelands
						this = root.empire
					}
					any_in_global_list = {
						variable = selected_titles_wastelands
						this = root
					}
				}
			}
		}
	}

	effect = {
		if = {
			limit = { 
				NOT = {
					any_in_global_list = {
						variable = selected_titles_wastelands
						this = root
					}
				}
			}
			add_to_global_variable_list = {
				name = selected_titles_wastelands
				target = root
			}

			if = {
				limit = { root.tier = tier_empire }
				every_in_de_jure_hierarchy = {
					limit = { tier = tier_kingdom }
					add_to_global_variable_list = {
						name = selected_titles_wastelands
						target = this
					}
				}
			}
			if = {
				limit = {
					root.tier = tier_kingdom
					empire = {
						NOT = {
							any_in_de_jure_hierarchy = {
								tier = tier_kingdom
								NOT = {
									is_target_in_global_variable_list = {
										name = selected_titles_wastelands
										target = this
									}
								}
							}
						}
					}
				}
				add_to_global_variable_list = {
					name = selected_titles_wastelands
					target = root.empire
				}
			}
		}
		else = {
			remove_list_global_variable = {
				name = selected_titles_wastelands
				target = root
			}

			if = {
				limit = { root.tier = tier_empire }
				every_in_de_jure_hierarchy = {
					limit = { tier = tier_kingdom }
					remove_list_global_variable = {
						name = selected_titles_wastelands
						target = this
					}
				}
			}
			if = {
				limit = {
					tier = tier_kingdom
					any_in_global_list = {
						variable = selected_titles_wastelands
						this = root.empire
					}
				}
				remove_list_global_variable = {
					name = selected_titles_wastelands
					target = root.empire
				}
			}
		}
	}
}

titles_selected_visible = {
	is_shown = {
		any_in_global_list = {
			variable = selected_titles_wastelands
			count > 0
		}
	}
}

set_empire_to_wastelands = {
	is_valid = {
		any_in_global_list = {
			variable = selected_titles_wastelands
			count > 0
		}
	}
	effect = {
		every_in_global_list = {
			variable = selected_titles_wastelands
			set_every_county_wilderness = { SELECTED_TITLE = this }

			save_scope_as = selected_empire
			if = {
				limit = {
					any_in_global_list = {
						variable = all_empires_in_game
						this = scope:selected_empire
					}
				}
				every_de_jure_county = {
					duchy.kingdom = {
						remove_list_global_variable = {
							name = all_kingdoms_in_game
							target = this
						}
					}
				}
	
				remove_list_global_variable = {
					name = all_empires_in_game
					target = scope:selected_empire
				}
			}
			else = {
				remove_list_global_variable = {
					name = all_kingdoms_in_game
					target = scope:selected_empire
				}
			}
		}

		every_in_global_list = {
			variable = selected_titles_wastelands
			save_temporary_scope_as = temp_scope
			remove_list_global_variable = {
				name = selected_titles_wastelands
				target = scope:temp_scope
			}
		}
	}
}

only_de_jure_areas = {

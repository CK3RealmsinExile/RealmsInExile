gui_dwarven_holdings = {
	scope = province
	is_shown = {
		county.title_province = { terrain = halls }
	}
	is_valid = {
		county.title_province = { terrain = halls }
	}
}

recieving_tributary_gold = {
	is_valid = {
		exists = root
		OR = {
			has_permanent_tributaries = yes
			has_non_permanent_tributaries = yes
		}
	}
}

paying_tributary_gold = {
	is_valid = {
		exists = root
		is_tributary = yes
	}
}

set_empire_to_wastelands = {
	effect = {
		set_every_county_wilderness = { EMPIRE = this }

		if = {
			limit = {
				any_in_global_list = {
					variable = all_empires_in_game
					this = root
				}
			}
			every_de_jure_county = {
				duchy.kingdom = {
					remove_list_global_variable = {
						name = all_kingdoms_in_game
						target = this
					}
				}
			}

			remove_list_global_variable = {
				name = all_empires_in_game
				target = root
			}
		}
		else = {
			remove_list_global_variable = {
				name = all_kingdoms_in_game
				target = root
			}
		}
	}
}

only_de_jure_areas = {
	effect = {
		every_in_global_list = {
			variable = non_canon_areas
			set_every_county_wilderness = { EMPIRE = this }
		}
	}
}

set_title_as_selected = {
	is_shown = {
		trigger_if = {
			limit = { root.tier = tier_empire }
			any_in_global_list = {
				variable = selected_titles_wastelands
				this = root
			}
		}
		trigger_else_if = {
			limit = { root.tier = tier_kingdom }
			OR = {
				any_in_global_list = {
					variable = selected_titles_wastelands
					this = root.empire
				}
				any_in_global_list = {
					variable = selected_titles_wastelands
					this = root
				}
			}
		}
		trigger_else = {
			trigger_if = {
				limit = { root.tier = tier_duchy }
				OR = {
					any_in_global_list = {
						variable = selected_titles_wastelands
						this = root.empire
					}
					any_in_global_list = {
						variable = selected_titles_wastelands
						this = root.kingdom
					}
					any_in_global_list = {
						variable = selected_titles_wastelands
						this = root
					}
				}
			}
		}
	}

	effect = {
		if = {
			limit = { 
				NOT = {
					any_in_global_list = {
						variable = selected_titles_wastelands
						this = root
					}
				}
			}
			add_to_global_variable_list = {
				name = selected_titles_wastelands
				target = root
			}

			if = {
				limit = { root.tier = tier_empire }
				every_in_de_jure_hierarchy = {
					limit = { 
						OR = {
							tier = tier_kingdom 
							tier = tier_duchy
						}
					}
					add_to_global_variable_list = {
						name = selected_titles_wastelands
						target = this
					}
				}
			}
			if = {
				limit = {
					root.tier = tier_kingdom
					empire = {
						NOT = {
							any_in_de_jure_hierarchy = {
								tier = tier_kingdom
								NOT = {
									is_target_in_global_variable_list = {
										name = selected_titles_wastelands
										target = this
									}
								}
							}
						}
					}
				}
				add_to_global_variable_list = {
					name = selected_titles_wastelands
					target = root.empire
				}
			}
			if = {
				limit = {
					root.tier = tier_duchy
					kingdom = {
						NOT = {
							any_in_de_jure_hierarchy = {
								tier = tier_duchy
								NOT = {
									is_target_in_global_variable_list = {
										name = selected_titles_wastelands
										target = this
									}
								}
							}
						}
					}
				}
				add_to_global_variable_list = {
					name = selected_titles_wastelands
					target = root.kingdom
				}
			}
			if = {
				limit = { root.tier = tier_kingdom }
				every_in_de_jure_hierarchy = {
					limit = { tier = tier_duchy }
					add_to_global_variable_list = {
						name = selected_titles_wastelands
						target = this
					}
				}
			}
		}
		else = {
			remove_list_global_variable = {
				name = selected_titles_wastelands
				target = root
			}

			if = {
				limit = { root.tier = tier_empire }
				every_in_de_jure_hierarchy = {
					limit = { 
						OR = {
							tier = tier_kingdom 
							tier = tier_duchy
						}
					}
					remove_list_global_variable = {
						name = selected_titles_wastelands
						target = this
					}
				}
			}
			if = {
				limit = {
					tier = tier_kingdom
					any_in_global_list = {
						variable = selected_titles_wastelands
						this = root.empire
					}
				}
				remove_list_global_variable = {
					name = selected_titles_wastelands
					target = root.empire
				}
			}
			if = {
				limit = { root.tier = tier_kingdom }
				every_in_de_jure_hierarchy = {
					limit = { tier = tier_duchy }
					remove_list_global_variable = {
						name = selected_titles_wastelands
						target = this
					}
				}
			}
			if = {
				limit = { root.tier = tier_duchy }
				remove_list_global_variable = {
					name = selected_titles_wastelands
					target = root
				}
				if = {
					limit = {
						any_in_global_list = {
							variable = selected_titles_wastelands
							this = root.kingdom
						}
					}
					remove_list_global_variable = {
						name = selected_titles_wastelands
						target = root.kingdom
					}
				}
				if = {
					limit = {
						any_in_global_list = {
							variable = selected_titles_wastelands
							this = root.empire
						}
					}
					remove_list_global_variable = {
						name = selected_titles_wastelands
						target = root.empire
					}
				}
			}
		}
	}
}

set_empire_to_wastelands = {
	is_valid = {
		any_in_global_list = {
			variable = selected_titles_wastelands
			count > 0
		}
	}
	effect = {
		every_in_global_list = {
			variable = selected_titles_wastelands
			set_every_county_wilderness = { SELECTED_TITLE = this }

			save_scope_as = selected_empire
			if = {
				limit = { scope:selected_empire.tier = tier_empire }
				every_de_jure_county = {
					duchy.kingdom = {
						remove_list_variable = { 
							name = de_jure_kingdoms 
							target = this 
						}
					}
				}
	
				remove_list_global_variable = {
					name = all_empires_in_game
					target = scope:selected_empire
				}
			}
			if = {
				limit = { scope:selected_empire.tier = tier_kingdom }
				empire = {
					remove_list_variable = {
						name = de_jure_kingdoms 
						target = scope:selected_empire
					}
				}
			}
			if = {
				limit = { scope:selected_empire.tier = tier_duchy }
				kingdom = {
					remove_list_variable = {
						name = de_jure_duchies
						target = scope:selected_empire
					}
				}
			}
		}

		every_in_global_list = {
			variable = selected_titles_wastelands
			save_temporary_scope_as = temp_scope
			remove_list_global_variable = {
				name = selected_titles_wastelands
				target = scope:temp_scope
			}
		}
	}
}

	}
	is_valid = {
		trigger_if = {
			limit = {
				OR = {
					is_orc = yes

				}
			}
			custom_tooltip = {
				text = NEED_TO_BE_UNDERGROUND_RACE
				OR = {
					AND = {
						OR = {
							scope:culture = { is_orc_culture = yes }
							scope:culture = { is_dwarf_culture = yes }
						}
						any_held_title = {
							tier = tier_county
							title_province ?= { terrain = halls }
						}
					}
					NOT = {
						any_held_title = {
							tier = tier_county
							title_province ?= { terrain = halls }
						}
					}
				}
			}
		}
		trigger_else = { always = yes }
	}
}

lotr_defensive_alliance_create_title = {
	saved_scopes = {
		player_scope
	}
	is_shown = {
		scope:player_scope = {
			has_defensive_alliance = yes
		}
	}
	effect = {
		save_scope_as = title_scope
		switch = {
			trigger = scope:title_scope.tier
			tier_duchy = {
				save_scope_value_as = { name = title_cost value = scope:player_scope.create_duchy_title_value }
				scope:player_scope = {
					add_prestige = 300
					add_legitimacy = 25
				}
			}
			tier_kingdom = {
				save_scope_value_as = { name = title_cost value = scope:player_scope.create_kingdom_title_value }
				scope:player_scope = {
					add_prestige = 400
					add_legitimacy = 50
				}
			}
			tier_empire = {
				save_scope_value_as = { name = title_cost value = scope:player_scope.create_empire_title_value }
				scope:player_scope = {
					add_prestige = 500
					add_legitimacy = 75
				}
			}
			tier_hegemony = {
				save_scope_value_as = { name = title_cost value = scope:player_scope.create_hegemony_title_value }
				scope:player_scope = {
					add_prestige = 600
					add_legitimacy = 100
				}
			}
		}

		scope:player_scope = {
			get_title = scope:title_scope
			remove_short_term_gold = scope:title_cost
		}
	}

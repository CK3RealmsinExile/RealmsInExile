lotr_fellowship_hud_visible = {
	is_shown = {
		always = yes
	}
}

lotr_fellowship_secrecy_level_active = {
	saved_scopes = {
		active_level_flag
	}
	is_shown = {
		switch = {
			trigger = scope:active_level_flag
			flag:secrecy_level_1 = { root.lotr_current_secrecy <= 20  }
			flag:secrecy_level_2 = { root.lotr_current_secrecy > 20 root.lotr_current_secrecy <= 40 }
			flag:secrecy_level_3 = { root.lotr_current_secrecy > 40 root.lotr_current_secrecy <= 60 }
			flag:secrecy_level_4 = { root.lotr_current_secrecy > 60 root.lotr_current_secrecy <= 80 }
			flag:secrecy_level_5 = { root.lotr_current_secrecy > 80 root.lotr_current_secrecy <= 100 }
		}
	}
}

lotr_fellowship_debug_increase_corruption = {
	effect = {
		if = {
			limit = { var:lotr_corruption < 100 }
			increase_variable = {
				NAME = lotr_corruption
				AMOUNT = 5
			}
		}
	}
}

lotr_fellowship_debug_decrease_corruption = {
	effect = {
		if = {
			limit = { var:lotr_corruption > 0 }
			decrease_variable = {
				NAME = lotr_corruption
				AMOUNT = 5
			}
		}
	}
}

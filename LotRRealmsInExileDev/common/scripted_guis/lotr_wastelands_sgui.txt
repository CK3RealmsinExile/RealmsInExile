gui_not_is_character_wastelands = {
	scope = character

	is_shown = {
		trigger_if = {
			limit = { exists = root.culture }
			NOT = { culture ?= culture:wastelands }
		}
	}
}

gui_not_is_province_wastelands = {
	scope = province
	
	is_shown = {
		trigger_if = {
			limit = {
				exists = county
				NOR = {
					county.tier = tier_duchy
					county.tier = tier_kingdom
					county.tier = tier_empire
				}
			}
			NOT = { county.title_province ?= { culture = culture:wastelands } }
		}
	}
}

gui_is_character_wastelands = {

	is_shown = {

		}
	}
}

gui_can_colonise = {
	scope = character
	
	# Triggered at yearly_on_action.txt -> and then wastelands.006 at lotr_wastelands_events.txt
	ai_is_valid = {	always = no }
	
	is_valid = {
		exists = root

		gold >= colonise_cost_val

		trigger_if = {
			limit = { 
				AND = {
					scope:wastelands = { terrain = halls } 
					NOR = {
						is_dwarf = yes
						is_orc = yes
						is_nazgul = yes
						is_maiar_fallen = yes
					}
				}
			}
			always = no
		}

		# Desolation of Smaug blocker
		trigger_if = {
			limit = { title:c_throne_of_erebor.title_province = { has_building = smaug } }
			custom_tooltip = {
				text = block_settlement_smaug_tooltip
				NOT = { scope:wastelands.county.title_province = { has_building = smaug_desolation } }
			}
		}

		# Elves need at least hope tier 3 to colonize
		trigger_if = {
		    limit = {
			    is_elf = yes
			}
			custom_tooltip = {
			    text = elven_colonize_hope_level
				has_medium_hope_or_higher = yes
			}
		}
		
		
		custom_tooltip = {
			text = colonise_border_county_trigger_tooltip
			# Only surrounding lands
			any_sub_realm_county = {
				any_title_to_title_neighboring_and_across_water_county = {
					this = scope:wastelands.county
				}
			}
		}
		any_sub_realm_county = {
			any_title_to_title_neighboring_and_across_water_county = {
				NOT = { has_county_modifier = block_settlement_ability }
			}
		}
	}

	## Check lotr_wastelands_effects.txt
	effect = {
		if = { # Islands
			limit = {
				exists = scope:wastelands.county
				OR = {
					title:d_andunaloni = {
						any_in_de_jure_hierarchy = {
							this = scope:wastelands.county
						}
					}
					title:d_tolfalas = {
						any_in_de_jure_hierarchy = {
							this = scope:wastelands.county
						}
					}
				}
			}
			custom_tooltip = colonise_island_desc
			hidden_effect = { ai_colonisation_effect = { WASTELANDS = scope:wastelands ROOT_SCOPE = root } }
		}
		else_if = { # Underground
			limit = { scope:wastelands = { terrain = halls } }
			custom_tooltip = colonise_underground_desc
			hidden_effect = { ai_colonisation_effect = { WASTELANDS = scope:wastelands ROOT_SCOPE = root } }
		}
		else = { # Mainland
			custom_tooltip = colonise_land_desc
			hidden_effect = { ai_colonisation_effect = { WASTELANDS = scope:wastelands ROOT_SCOPE = root } }
		}
	}
}

gui_settlement_count = {
	scope = character

	is_valid = {
		trigger_if = {
			limit = { exists = root }
			trigger_if = {
				limit = { is_alive = yes }
				has_variable = num_settled_wilderness 
				var:num_settled_wilderness > settlement_title_tier_limit
			}
		}
	}

	is_shown = {
		trigger_if = {
			limit = { exists = root }
			trigger_if = {
				limit = { is_alive = yes }
				has_variable = num_settled_wilderness 
				var:num_settled_wilderness > 0 
			}
		}
	}
}

lotr_WAC_rotation_button = {
	saved_scopes = {
		targeted_title
	}

	is_valid = {
		NOT = {
			any_owned_story = {
				story_type = story_cycle_automated_colonisation
				has_variable = title_to_colonise
				var:title_to_colonise = scope:targeted_title
			}
		}
	}

	is_shown = { #Used to get different loc keys for " WASTELANDS_AUTOMATION_TOGGLE_TT_DESC "
		any_owned_story = {
			story_type = story_cycle_automated_colonisation
			has_variable = title_to_colonise
			var:title_to_colonise = scope:targeted_title
			has_variable = custom_settings
		}
	}
}
lotr_WAC_effect_button = {
	effect = {
		if = { #If already colonising, stop it
			limit = { 
				any_owned_story = {
					story_type = story_cycle_automated_colonisation
					has_variable = title_to_colonise
					var:title_to_colonise = scope:targeted_title
				}
			}
			random_owned_story = {
				limit = {
					story_type = story_cycle_automated_colonisation
					has_variable = title_to_colonise
					var:title_to_colonise = scope:targeted_title
				}
				end_story = yes
			}
		}
		else = { #If not colonising, start automation
			set_variable = {
				name = automated_colonisation_for_title
				value = scope:targeted_title
			}
			every_owned_story = {
				limit = {
					var:title_to_colonise = { save_temporary_scope_as = saved_title_scope }
					scope:targeted_title = {
						any_in_de_jure_hierarchy = { this = scope:saved_title_scope }
					}
				}
				end_story = yes
			}
			create_story = story_cycle_automated_colonisation
		}
	}
}

lotr_WAC_higher_title_already_selected_tooltip = {
	saved_scopes = {
		targeted_title
	}

	is_valid = {
		trigger_if = {
			limit = { owns_story_of_type = story_cycle_automated_colonisation }
			any_owned_story = {
				story_type = story_cycle_automated_colonisation
				has_variable = title_to_colonise
				trigger_if = {
					limit = { var:title_to_colonise != scope:targeted_title }
					NOT = {
						var:title_to_colonise = {
							any_in_de_jure_hierarchy = { this = scope:targeted_title }
						}
					}
				}
			}
		}
		trigger_else = {
			always = yes
		}
	}
}

lotr_WAC_building_stockade_settings = {
	effect = {
		random_player = {
			limit = { this = root }
			save_scope_as = player_value
		}
		
		if = {
			limit = {
				NOT = {
					is_target_in_global_variable_list = {
						name = lotr_WAC_building_stockade_settings
						target = scope:player_value
					}
				}
			}
			add_to_global_variable_list = {
				name = lotr_WAC_building_stockade_settings
				target = scope:player_value
			}
		}
		else = {
			remove_list_global_variable = {
				name = lotr_WAC_building_stockade_settings
				target = scope:player_value
			}
		}
	}
}

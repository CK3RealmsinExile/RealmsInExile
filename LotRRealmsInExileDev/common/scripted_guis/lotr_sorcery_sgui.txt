can_do_sorcery = {
	is_shown = {
		OR = {
			is_sorcerer = yes
			has_trait = nazgul
			is_evil_istari = yes
			has_trait = maiar
		}
	}
	is_valid = {
		NOT = {
			has_character_modifier = practiced_sorcery_recently
		}
	}
}

lotr_death_spell_intent_selected = {
	effect = {
		if = {
			limit = { has_variable = spell_selected }
			remove_variable = spell_selected
		}
		set_variable = {
			name = spell_selected
			value = flag:death_spell
		}
	}
}

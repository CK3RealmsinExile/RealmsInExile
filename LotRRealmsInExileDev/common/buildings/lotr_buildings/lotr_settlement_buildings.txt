
### Feudal/Elven ###
# Stockades

stockades_01 = {
	construction_time = 125

	can_construct = {	
		has_holding_type = settlement_holding
	}
	
	can_construct_potential = {
		scope:holder = { 
			can_construct_tribal_settlement_building = no
		}
	}

	can_construct_showing_failures_only = {
		custom_description = {
			text = "must_not_have_blockers"
			NOT = { has_building_with_flag = obstacle }
		}
	}
	
	is_enabled = {
		has_building_or_higher = settlement_01
	}
	
	cost_gold = 50
	
	levy = 75
	max_garrison = 100
	province_modifier = {
		defender_holding_advantage = 2
		fort_level = 1
	}
	
	type_icon = "icon_building_palisades.dds"
	flag = lotr_settlement_building
	
	ai_value = {
		base = 1
		ai_tier_1_building_modifier = yes
		modifier = {
			factor = 0
			trigger_if = {
				limit = { scope:holder = { has_variable = WAC_title_holder } }
				NOT = { scope:holder.var:WAC_title_holder = flag:stockades_01_priority }
			}
		}
		modifier = {
			factor = 1000
			scope:holder = { has_character_flag = stockades_01_priority }
		}
		modifier = {
			factor = 0
			scope:holder = { any_owned_story = { story_type = story_conqueror } }
		}
	}
}

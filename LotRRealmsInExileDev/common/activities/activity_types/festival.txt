activity_festival = {
	is_shown = {
		has_tgp_dlc_trigger = yes
		highest_held_title_tier > tier_barony
		# government_is_japanese_trigger = yes #LotR
		trigger_if = {
			limit = {
				is_ai = yes
			}
			is_at_war = no
			short_term_gold >= festival_activity_base_cost
		}
		OR = { #LotR
			culture ?= { has_cultural_pillar = heritage_womaw }
			culture ?= { has_cultural_pillar = heritage_magri }
			culture ?= { has_cultural_pillar = heritage_periannath }
		}
		lotr_is_shown_basic_activity_trigger = yes #LotR
	}
	can_start_showing_failures_only = {
		NOT = { is_activity_type_on_cooldown = activity_festival }
		is_available_adult = yes
		custom_tooltip = {
			text = activity_festival_landed_or_ceremonial_liege_tt
			trigger_if = {
				limit = { is_landed = no }
				tgp_is_ceremonial_liege_trigger = yes
			}
		}
	}

	is_valid = {
		scope:host = {
			is_alive = yes
			is_capable_adult = yes
			is_imprisoned = no
			is_landed_or_landless_administrative = yes
			NOT = { is_incapable = yes }
		}
		# If no one shows up
		trigger_if = {
			limit = { is_current_phase_active = yes }
			has_attending_activity_guests = yes
		}
	}

	on_invalidated = {
		invalidation_imprisoned_effect = yes
		invalidation_incapable_effect = yes
		invalidation_unlanded_effect = yes
		if = { # No one shows up
			limit = {
				has_attending_activity_guests = no
			}
			scope:activity = {
				activity_type = { save_scope_as = activity_type }
				activity_location = { save_scope_as = location }
			}
			scope:host = { trigger_event = activity_system.0100 }
		}
	}

	on_host_death = {
		invalidation_host_death_effect = yes
	}

	is_single_location = yes

	province_filter = domicile_domain
	ai_province_filter = domicile_domain

	is_location_valid = {
		has_holding = yes
	}

	max_province_icons = 5

	cost = {
		gold = {
			add = {
				add = {
					value = festival_activity_base_cost #Base level
					desc = festival_base_cost
				}
				add = {
					value = festival_activity_base_cost
					multiply = activity_cost_scale_by_era
					subtract = festival_activity_base_cost
					desc = activity_cost_scale_by_era_desc
				}
			}
			if = {
				limit = {
					confederation ?= { has_cohesion_level_parameter = any_member_activity_cost_reduced }
				}
				multiply = {
					value = 0.9
					desc = CB_COHESION_PARAMETER_DISCOUNT
				}
			}
			if = {
				limit = {
					dynasty ?= { has_dynasty_perk = tgp_japan_legacy_3 }
				}
				multiply = {
					value = {
						value = 1
						subtract = tgp_dynasty_legacy_perk_festival_discount_value
					}
					desc = TGP_LEGACY_PERK_CHEAPER_FESTIVALS
				}
			}
			if = {
				limit = {
					var:tgp_ceremonial_liege_events_0020_discount ?= flag:festival
				}
				multiply = {
					value = 0.85
					desc = tgp_ceremonial_liege_events_0020_discount_desc
				}
			}
		}
	}

	ui_predicted_cost = {
		# All costs are balanced on County/Early Era starting point
		gold = {
			add = { # Base cost, scaled by era
				value = festival_activity_base_cost
				multiply = activity_cost_scale_by_era
			}
			add = {  
				value = 0 # Bad Option Cost
				add = 10 # Normal Option Cost
				add = 35 # Good Option Cost
				# Scale the sum of the options by era and tier
				add = {
					value = 45
					multiply = activity_cost_scale_by_tier
				}
				add = {
					value = 45
					multiply = activity_cost_scale_by_era
				}
				# Divide by 3 (num of levels for this option) so we get the average option cost
				divide = 3
			}
			if = {
				limit = {
					confederation ?= { has_cohesion_level_parameter = any_member_activity_cost_reduced }
				}
				multiply = {
					value = 0.9
					desc = CB_COHESION_PARAMETER_DISCOUNT
				}
			}
			if = {
				limit = {
					dynasty ?= { has_dynasty_perk = tgp_japan_legacy_3 }
				}
				multiply = {
					value = {
						value = 1
						subtract = tgp_dynasty_legacy_perk_festival_discount_value
					}
					desc = TGP_LEGACY_PERK_CHEAPER_FESTIVALS
				}
			}
			if = {
				limit = {
					var:tgp_ceremonial_liege_events_0020_discount ?= flag:festival
				}
				multiply = {
					value = 0.85
					desc = tgp_ceremonial_liege_events_0020_discount_desc
				}
			}
			# Make it a multiple of 5 (rounded up)
			divide = 5
			ceiling = yes
			multiply = 5
		}
	}

	phases = {
		festival_phase = {
			is_predefined = yes

			on_phase_active = {
				if = {
					limit = { this = scope:host }
					scope:activity = { progress_activity_phase_after = { months = 1 } }
				}
				trigger_event = festival.014
			}

			on_weekly_pulse = {
				# Fires for every attendee
				trigger_event = {
					on_action = festival_random_pulse
				}
			}

			on_end = {
				if = {
					limit = { this = scope:host }
					scope:activity = {
						add_activity_log_entry = {
							key = festival_conc_reward
							score = 100 
							show_in_conclusion = yes
							character = root 
							if = {
								limit = {
									scope:activity = {
										has_activity_option = {
											category = festival_option_activity
											option = festival_activity_extravagant
										}
									}
								}
								root = {
									add_prestige = major_prestige_gain 
									if = {
										limit = {
											government_has_flag = government_has_influence
										}
										change_influence = major_influence_gain
									}
								}
							}
							else_if = {
								limit = {
									scope:activity = {
										has_activity_option = {
											category = festival_option_activity
											option = festival_activity_modest
										}
									}
								}
								root = {
									add_prestige = minor_prestige_gain 
									if = {
										limit = {
											government_has_flag = government_has_influence
										}
										change_influence = minor_influence_gain
									}
								}							
							}
							else = {
								root = {
									add_prestige = medium_prestige_gain 
									if = {
										limit = {
											government_has_flag = government_has_influence
										}
										change_influence = medium_influence_gain
									}
								}
							}
							# Piety for Host from Ritual Celebrations and Hedonism tenets
							disburse_festival_piety_reward_effect = yes
							every_attending_character = {
								limit = { this != root }
								custom = every_guest_scope_tt
								add_opinion = {
									target = root
									modifier = hosted_successful_festival_opinion
								}
							}
						}
					}
					# tgp_ceremonial_liege_events.0020
					if = {
						limit = {
							var:tgp_ceremonial_liege_events_0020_discount ?= flag:festival
						}
						tgp_one_up_activity_effect = yes
						remove_variable = tgp_ceremonial_liege_events_0020_discount
					}
				}
				if = {
					limit = {
						is_ai = no
					}
					if = {
						limit = {
							has_activity_intent = add_to_bloc_intent
							intent_target ?= {
								NOR = {
									has_any_bad_relationship_with_character_trigger = { CHARACTER = root }
									has_any_good_relationship_with_character_trigger = { CHARACTER = intent_target.house.house_confederation.leading_house.house_head }
								}
								opinion = {
									target = root
									value >= 50
								}
								opinion = {
									target = intent_target.house.house_confederation.leading_house.house_head
									value < 50
								}
							}
						}
						if = { # LEAVE OLD BLOC IF NECESSARY
							limit = {
								intent_target = {
									exists = house.house_confederation
								}
							}
							intent_target.house = {
								set_variable = {
									name = bloc_leaving_reason
									value = flag:reason_events
									days = 4
								}
								tgp_leave_house_bloc_effect = {
									OPINION = flag:yes
									TRUCE = flag:no
								}
							}
							intent_target = {
								add_character_flag = {
									flag = invited_to_bloc_recently
									years = 2
								}
							}
						}
						scope:activity = {
							add_activity_log_entry = {
								key = festival_intent_target_joined_bloc
								character = root.intent_target
								score = 100 
								show_in_conclusion = yes
								root.intent_target.house = {
									tgp_join_house_bloc_effect = {
										INVITER = root.house
										OPINION = flag:yes
									}
								}
							}
						}
					}
					else_if = {
						limit = {
							has_activity_intent = remove_from_bloc_intent
							intent_target ?= {
								OR = {
									opinion = {
										target = house.house_confederation.leading_house.house_head
										value <= -30
									}
									house.house_confederation.cohesion < 25
								}				
							}
						}
						scope:activity = {
							add_activity_log_entry = {
								key = festival_intent_target_left_bloc
								character = root.intent_target
								score = 100 
								show_in_conclusion = yes
								root.intent_target.house = {
									tgp_leave_house_bloc_effect = {
										OPINION = flag:yes
										TRUCE = flag:no
									}
								}
							}
						}
					}
				}
				if = {
					limit = {
						is_ai = no
					}
					trigger_event = festival.2000 # A closing event with some potential benefits just for the player
				}
			}
		}
	}

	phases = { # Done # activity = {complete_activity = yes}
		expedition_phase = {
			is_predefined = yes

			is_shown = {}
			on_phase_active = {
				if = {
					limit = { scope:host = { is_independent_ruler = no } }
					scope:host = { set_variable = host_is_vassal }
				}
				# Generates a character if one isn't found
				if = {
					limit = { 
						NOT = { 
							exists = scope:opportunistic_dwarf 
							scope:host = { has_character_flag = satellite_chosen }
						} 
					}
					create_character = {
						template = random_dwarf_template
						employer = scope:host
						save_scope_as = opportunistic_dwarf
					}
					scope:opportunistic_dwarf = { add_to_activity = scope:activity }
					scope:host = {
						set_variable = {
							name = opportunistic_dwarf
							value = scope:opportunistic_dwarf
						}
					}
				}
		
				# Give title to dwarf char
				if = {
					limit = { scope:activity = { has_activity_option = { category = special_type option = expedition_type_standard } } }
					scope:activity.activity_location = { activity_title_change_effect = { RULER = scope:host } }
					scope:host = { trigger_event = expedition.9000 }
				}
				else = {
					if = {
						limit = { scope:activity = { has_activity_option = { category = expedition_option_gold_investment option = expedition_gold_investment_low } } }
						scope:host = { add_character_flag = low_gold_investment_expedition }
					}
					else_if = {
						limit = { scope:activity = { has_activity_option = { category = expedition_option_gold_investment option = expedition_gold_investment_medium } } } 
						scope:host = { add_character_flag = medium_gold_investment_expedition }
					}
					else_if = {
						limit = { scope:activity = { has_activity_option = { category = expedition_option_gold_investment option = expedition_gold_investment_high } } }
						scope:host = { add_character_flag = high_gold_investment_expedition }
					}
					
					scope:host = { trigger_event = expedition.0010 } #FOR TESTING PURPOSES ~~ Aerien
					#scope:activity.activity_location = { activity_title_change_effect = { RULER = scope:host.var:opportunistic_dwarf } }
					
				}

				invalidate_activity = scope:activity
			}

			on_enter_phase = {
			}

			on_end = { # Fires for every attendee
				target_weight_modifier_effect = { VALUE = -5 }
				scope:activity.activity_location = { save_scope_as = activity_location }
				scope:activity = { skip_activity_phase = yes }
			}
		}

dynasty_blessing = {
	icon = public
	
	is_valid = {
		is_current_phase_active = yes
		exists = scope:host.dynasty
		trigger_if = {
			limit = {
				scope:activity = { has_activity_type = activity_wedding }
			}
			any_attending_character = {
				NOR = {
					this = scope:host
					this = scope:activity.special_guest:spouse_1
					this = scope:activity.special_guest:spouse_2
				}
				is_physically_able_ai_adult = yes
				location = scope:province
				dynasty ?= { dynasty_prestige_level > scope:host.dynasty.dynasty_prestige_level }
			}
		}
		trigger_if = {
			limit = {
				scope:activity = { has_activity_type = activity_coronation }
			}
			any_attending_character = {
				NOR = {
					this = scope:host
				}
				is_physically_able_ai_adult = yes
				location = scope:province
				dynasty ?= { dynasty_prestige_level > scope:host.dynasty.dynasty_prestige_level }
			}
		}
	}

	weight = {
		value = 1
		if = {
			limit = {
				scope:activity = {
					has_activity_type = activity_wedding
					has_activity_option = {
						category = wedding_option_decoration
						option = wedding_decoration_good
					}
				}
			}
			add = 0.5
		}

		if = {
			limit = {
				scope:activity = {
					has_activity_type = activity_coronation
					has_activity_option = {
						category = ceremony_size
						option = coronation_large_ceremony
					}
				}
			}
			add = 0.5
		}
	}

	effect = {
		scope:host = {
			save_scope_as = root_scope
		}
		
		if = {
			limit = {
				scope:activity = {
					has_activity_type = activity_wedding
				}
			}
			random_attending_character = {
				limit = {
					NOR = {
						this = scope:host
						this = scope:activity.special_guest:spouse_1
						this = scope:activity.special_guest:spouse_2
					}
					is_physically_able_ai_adult = yes
					location = scope:province
					dynasty ?= { dynasty_prestige_level > scope:host.dynasty.dynasty_prestige_level }
				}
				save_scope_as = first
			}		
		}

		if = {
			limit = {
				scope:activity = {
					has_activity_type = activity_coronation
				}
			}
			random_attending_character = {
				limit = {
					NOR = {
						this = scope:host
					}
					is_physically_able_ai_adult = yes
					location = scope:province
					dynasty ?= { dynasty_prestige_level > scope:host.dynasty.dynasty_prestige_level }
				}
				save_scope_as = first
			}		
		}

		add_activity_log_entry = {
			key = dynasty_blessing
			tags = { pulse_action }
			character = scope:first
			
			scope:host.dynasty = {
				add_dynasty_prestige = medium_dynasty_prestige_gain
			}
			scope:first = {
				add_opinion = {
					modifier = impressed_opinion
					target = scope:host
					opinion = 10
				}
			}
		}
	}
}

cultural_acceptance = {
	icon = stewardship
	
	is_valid = {
		is_current_phase_active = yes
		exists = scope:activity.special_guest:spouse_1.culture
		exists = scope:activity.special_guest:spouse_2.culture
		scope:activity.special_guest:spouse_1.culture != scope:activity.special_guest:spouse_2.culture
		trigger_if = {
			limit = {
				scope:activity = { has_activity_type = activity_wedding }
			}
			NOT = { scope:activity.special_guest:spouse_1.culture = scope:activity.special_guest:spouse_2.culture }
		}
		trigger_if = {
			limit = {
				scope:activity = { has_activity_type = activity_coronation }
			}
			scope:activity.var:officiator ?= { is_alive = yes }
			NOT = { scope:host.culture = scope:activity.var:officiator.culture }
		}
	}

	weight = {
		value = 1
	}

	effect = {
		if = {
			limit = {
				scope:activity = { has_activity_type = activity_wedding }
			}
			scope:activity.special_guest:spouse_1 = { save_scope_as = first }
			scope:activity.special_guest:spouse_2 = { save_scope_as = second }
		}

		if = {
			limit = {
				scope:activity = { 
					has_activity_type = activity_coronation 
					has_variable = officiator
				}
			}
			scope:host = { save_scope_as = first }
			scope:activity.var:officiator = { save_scope_as = second }
		}

		add_activity_log_entry = {
			key = cultural_acceptance
			tags = { pulse_action }
			character = scope:first
			target = scope:second
			
			scope:first.culture = {
				change_cultural_acceptance = {
					target = scope:second.culture
					value = miniscule_positive_culture_acceptance
					desc = cultural_acceptance_gain_hold_court_event_outcome
				}
			}
			scope:second.culture = {
				change_cultural_acceptance = {
					target = scope:first.culture
					value = miniscule_positive_culture_acceptance
					desc = cultural_acceptance_gain_hold_court_event_outcome
				}
			}
		}
	}
}

### Tour Activity Intents

lechery_intent = {
	is_shown = {
		is_alive = yes
		is_adult = yes
		is_imprisoned = no
		NOT = { 
			OR = {
				is_elf = yes 
				is_in_valinor = yes
			}
		}
		might_cheat_on_every_partner_trigger = yes
	}
	
	auto_complete = yes

	is_valid = {
		NOT = { has_trait = celibate }
	}

	on_invalidated = {
		trigger_event = activity_system.0090
	}

	ai_will_do = {
		value = 1
		if = {
			limit = {
				OR = {
					has_trait = lustful
					has_trait = rakish
				}
			}
			add = 250
		}
		if = {
			limit = {
				has_trait = zealous
				faith = {
					trait_is_virtue = lustful
				}
			}
			add = 50
		}
		if = {
			limit = {
				OR = {
					has_trait = chaste
					AND = {
						has_trait = zealous
						NOT = {
							faith = {
								trait_is_virtue = lustful
							}
						}
					}
				}
			}
			multiply = 0
		}
	}

	scripted_animation = {
		animation = flirtation
	}
}

#Shared with Pilgrimages
altruism_intent = {
	auto_complete = yes

	is_shown = {
		has_dlc_feature = advanced_activities
		is_alive = yes
		is_imprisoned = no
	}

	on_invalidated = {
		trigger_event = activity_system.0090
	}

	ai_will_do = {
		value = 0
		add = {
			value = ai_greed
			multiply = -1
		}
		add = {
			value = ai_compassion
			multiply = 0.5
		}
		if = {
			limit = {
				OR = {
					has_trait = generous
					has_trait = profligate
				}
			}
			add = 50
		}
		if = {
			limit = {
				has_trait = zealous
				faith = {
					trait_is_virtue = generous
				}
			}
			add = 50
		}
		if = {
			limit = {
				OR = {
					has_trait = greedy
					has_trait = callous
					has_trait = sadistic
				}
			}
			multiply = 0
		}
		if = {
			limit = { # Don't do it if you're poor
				short_term_gold <= medium_gold_value
			}
			multiply = 0
		}
	}

	scripted_animation = {
		animation = personality_forgiving
	}
}

justice_intent = {

	auto_complete = yes

	on_invalidated = {
		trigger_event = activity_system.0090
	}
	ai_will_do = {
		value = ai_honor
		if = {
			limit = {
				OR = {
					has_trait = just
					has_trait = arbitrary # an arbitrary character would still think they're morally just
				}
			}
			add = 50
		}
		if = {
			limit = {
				OR = {
					has_trait = compassionate
					has_trait = shy
					has_trait = callous
				}
			}
			subtract = 50
		}
	}
	scripted_animation = {
		animation = personality_honorable
	}
}

### Learn Sorcery activity ###
activity_learn_sorcery = {
	solo = yes
	color = { 0.2 0.8 0.4 1 }
	on_spawn = {
		invite_character_to_activity = activity_owner
		accept_invitation_for_character = activity_owner
		set_variable = {
			name = sorcery_quest
			value = 0
		}
	}
	on_activate = {
		activity_owner = {
			trigger_event = {
				id = evil.0100
#				days = 1
			}
		}
	}
	on_complete = {
		activity_owner = {
			remove_variable = sorcery_quest
		}	
	}
}

### Helper Function
lotr_set_parent_culture_variables = {
	random_parent_culture = {
		save_scope_as = parent_culture_1
	}
	random_parent_culture = {
		limit = { this != scope:parent_culture_1 }
		save_scope_as = parent_culture_2
	}

	if = {
		limit = { is_hybrid_culture = yes }
		set_variable = lotr_is_hybridized_culture
	}

	if = {
		limit = { exists = scope:parent_culture_1 }
		set_variable = { name = parent_1 value = scope:parent_culture_1 }
	}
	if = {
		limit = { exists = scope:parent_culture_2 }
		set_variable = { name = parent_2 value = scope:parent_culture_2 }
	}
	if = {
		limit = { NOT = { has_variable = lotr_culture_formed } }
		set_variable = { name = lotr_culture_formed value = current_year }
	}
}

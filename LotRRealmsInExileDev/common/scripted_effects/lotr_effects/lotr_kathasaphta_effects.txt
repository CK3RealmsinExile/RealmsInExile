
kathasaptha_manuscript_event_chain_cleanup_effect = {		
	if = {
		limit = { has_variable = kathasaptha_manuscript_type }
		remove_variable = kathasaptha_manuscript_type
	}
	if = {
		limit = { has_variable = kathasaptha_manuscript_author }
		remove_variable = kathasaptha_manuscript_author
	}
	if = {
		limit = { has_character_modifier = kathasaptha_annotating_manuscript }
		remove_character_modifier = kathasaptha_annotating_manuscript
	}
}

kathasaptha_manuscript_general_effect = {
	add_piety = medium_piety_gain
	if = {
		limit = {
			var:kathasaptha_manuscript_author ?= flag:self
		}
		add_prestige = medium_prestige_gain
	}
}

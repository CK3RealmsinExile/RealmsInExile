lotr_vassalize_all_rulers_as_root = { ### Vassalizes all rulers
	create_title_and_vassal_change = {
		type = swear_fealty
		save_scope_as = change
		add_claim_on_loss = no
	}
				
	every_county = {
		limit = {
			holder = { is_wastelands = no }
			is_wastelands = no
			holder != root
			holder = { is_in_valinor = no }
			holder.top_liege != root
			holder.highest_held_title_tier < root.highest_held_title_tier
		}
		holder = {
			change_liege = {
				liege = root
				change = scope:change
			}
		}
	}
	resolve_title_and_vassal_change = scope:change
}

lotr_tributarize_everyone_as_root = { ### Tributarizes all rulers
	every_independent_ruler = {
		limit = {
			is_tributary = no
			is_in_valinor = no
			is_wastelands = no
			this != character:lineofsauron
			is_landed = yes
		}
		save_scope_as = curr_ruler
		start_tributary = {
			contract_group = tributary_subjugated
			suzerain = character:lineofsauron
		}
		if = {
			limit = { 
				is_wastelands = yes
				is_tributary = yes
			}
			end_tributary = yes
		}
	}
}

lotr_take_all_titles_as_root = { ### Takes all titles
	create_title_and_vassal_change = {
		type = conquest
		save_scope_as = change
		add_claim_on_loss = yes
	}
	every_county = {
		limit = {
			holder = { is_wastelands = no }
			is_wastelands = no
			holder != root
			holder = { is_in_valinor = no }
		}
		change_title_holder = {
			holder = root
			change = scope:change
		}
	}
	resolve_title_and_vassal_change = scope:change
}

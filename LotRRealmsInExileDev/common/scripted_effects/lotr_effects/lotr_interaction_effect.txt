offer_fealty_ranger_interaction_effect = {
    # Ranger vassalage is a special case, so we use a different interaction effect that mimicks a tributary relationship.
    # Custom Ranger Tributaries LotR
    $TRIBUTARY$ = {
        start_tributary = {
            contract_group = tributary_ranger
            suzerain = $SUZERAIN$
        }
    }
    #Un-exile Rangers from potential target's territory
    $SUZERAIN$ = {
        every_neighboring_and_across_water_top_liege_realm_owner = {
            limit = { opposite_alignment = { TARGET = $SUZERAIN$ } }
            if = {
                limit = {
                    reverse_has_opinion_modifier = {
                        target = root
                        modifier = demanded_eviction_opinion
                    }
                }
                remove_opinion = {
                    target = root
                    modifier = demanded_eviction_opinion
                }
            }
            every_vassal_or_below = {
               limit = {
                    reverse_has_opinion_modifier = {
                        target = root
                        modifier = demanded_eviction_opinion
                    }
                    opposite_alignment = { TARGET = $SUZERAIN$ }
                }
                remove_opinion = {
                    target = root
                    modifier = demanded_eviction_opinion
                }
            }
        }
    }
}

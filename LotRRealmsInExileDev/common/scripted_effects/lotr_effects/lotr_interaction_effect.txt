offer_fealty_ranger_interaction_effect = {
    # Ranger vassalage is a special case, so we use a different interaction effect that mimicks a tributary relationship.
    # Custom Ranger Tributaries LotR
    if = {
        limit = { $TRIBUTARY$ = { has_realm_law = camp_purpose_rangers} }
        $TRIBUTARY$ = {
            start_tributary = {
                contract_group = tributary_ranger
                suzerain = $SUZERAIN$
            }

            hidden_effect = {
                trigger_event = rangers.9200
            }
        }
    }


}

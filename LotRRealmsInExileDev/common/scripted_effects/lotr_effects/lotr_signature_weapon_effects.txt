# Signature Weapon effects

lotr_duel_set_weapon_for_new_char = {
	set_signature_weapon_effect = yes
	if = {
		limit = {
			NOT = { has_character_flag = swordshield_person }
			NOT = { has_character_flag = axeshield_person }
			NOT = { has_character_flag = maceshield_person }
			NOT = { has_character_flag = spearshield_person }
			NOT = { has_character_flag = sword_person }
			NOT = { has_character_flag = axe_person }
			NOT = { has_character_flag = spear_person }
			NOT = { has_character_flag = mace_person }
			NOT = { has_character_flag = longaxe_person }
		}
		random_list = {
			1 = {
				trigger = {
					exists = var:signature_weapon
					var:signature_weapon = flag:sword
				}
				add_character_flag = swordshield_person
			}
			1 = {
				trigger = {
					exists = var:signature_weapon
					var:signature_weapon = flag:sword
				}
				add_character_flag = sword_person
			}
			1 = {
				trigger = {
					exists = var:signature_weapon
					var:signature_weapon = flag:mace
				}
				add_character_flag = maceshield_person
			}
			1 = {
				trigger = {
					exists = var:signature_weapon
					var:signature_weapon = flag:mace
				}
				add_character_flag = mace_person
			}
			1 = {
				trigger = {
					exists = var:signature_weapon
					var:signature_weapon = flag:spear
				}
				add_character_flag = spearshield_person
			}
			1 = {
				trigger = {
					exists = var:signature_weapon
					var:signature_weapon = flag:spear
				}
				add_character_flag = spear_person
			}
			1 = {
				trigger = {
					exists = var:signature_weapon
					var:signature_weapon = flag:axe
				}
				add_character_flag = axe_person
			}
			1 = {
				trigger = {
					exists = var:signature_weapon
					var:signature_weapon = flag:axe
				}
				add_character_flag = axeshield_person
			}
			1 = {
				trigger = {
					exists = var:signature_weapon
					var:signature_weapon = flag:longaxe
				}
				add_character_flag = longaxe_person
			}
		}
	}
}

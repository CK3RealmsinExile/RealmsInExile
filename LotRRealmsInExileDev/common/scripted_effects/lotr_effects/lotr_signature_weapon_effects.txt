# Signature Weapon effects

lotr_duel_set_weapon_for_new_char = {
	set_signature_weapon_effect = yes
	if = {
		limit = {
			NOT = { has_character_flag = swordshield_person }
			NOT = { has_character_flag = axeshield_person }
			NOT = { has_character_flag = maceshield_person }
			NOT = { has_character_flag = spearshield_person }
			NOT = { has_character_flag = sword_person }
			NOT = { has_character_flag = axe_person }
			NOT = { has_character_flag = spear_person }
			NOT = { has_character_flag = mace_person }
			NOT = { has_character_flag = longaxe_person }
		}
		random_list = {
			1 = {
				trigger = {
					exists = var:signature_weapon
					var:signature_weapon = flag:sword
				}
				add_character_flag = swordshield_person
			}
			1 = {
				trigger = {
					exists = var:signature_weapon
					var:signature_weapon = flag:sword
				}
				add_character_flag = sword_person
			}
			1 = {
				trigger = {
					exists = var:signature_weapon
					var:signature_weapon = flag:mace
				}
				add_character_flag = maceshield_person
			}
			1 = {
				trigger = {
					exists = var:signature_weapon
					var:signature_weapon = flag:mace
				}
				add_character_flag = mace_person
			}
			1 = {
				trigger = {
					exists = var:signature_weapon
					var:signature_weapon = flag:spear
				}
				add_character_flag = spearshield_person
			}
			1 = {
				trigger = {
					exists = var:signature_weapon
					var:signature_weapon = flag:spear
				}
				add_character_flag = spear_person
			}
			1 = {
				trigger = {
					exists = var:signature_weapon
					var:signature_weapon = flag:axe
				}
				add_character_flag = axe_person
			}
			1 = {
				trigger = {
					exists = var:signature_weapon
					var:signature_weapon = flag:axe
				}
				add_character_flag = axeshield_person
			}
			1 = {
				trigger = {
					exists = var:signature_weapon
					var:signature_weapon = flag:longaxe
				}
				add_character_flag = longaxe_person
			}
		}
	}
}

lotr_set_signature_weapon_effect = {
	remove_character_flag = swordshield_person
	remove_character_flag = axeshield_person
	remove_character_flag = maceshield_person
	remove_character_flag = spearshield_person
	remove_character_flag = sword_person
	remove_character_flag = axe_person
	remove_character_flag = mace_person
	remove_character_flag = spear_person
	remove_character_flag = longaxe_person

	hidden_effect = {
		set_local_variable = {
			name = signature_weapon
			value = $WEAPON$
		}

		if = {
			limit = { exists = local_var:signature_weapon }

			switch = {
				trigger = local_var:signature_weapon

				flag:sword_and_shield = {
					hidden_effect = {
						add_character_flag = swordshield_person
						set_variable = {
							name = signature_weapon
							value = flag:sword
						}
					}
				}
				flag:axe_and_shield = {
					hidden_effect = {
						add_character_flag = axeshield_person
						set_variable = {
							name = signature_weapon
							value = flag:axe
						}
					}
				}
				flag:mace_and_shield = {
					hidden_effect = {
						add_character_flag = maceshield_person
						set_variable = {
							name = signature_weapon
							value = flag:mace
						}
					}
				}
				flag:spear_and_shield = {
					hidden_effect = {
						add_character_flag = spearshield_person
						set_variable = {
							name = signature_weapon
							value = flag:spear
						}
					}
				}
				flag:sword = {
					hidden_effect = {
						add_character_flag = sword_person
						set_variable = {
							name = signature_weapon
							value = flag:sword
						}
					}
				}
				flag:axe = {
					hidden_effect = {
						add_character_flag = axe_person
						set_variable = {
							name = signature_weapon
							value = flag:axe
						}
					}
				}
				flag:mace = {
					hidden_effect = {
						add_character_flag = mace_person
						set_variable = {
							name = signature_weapon
							value = flag:mace
						}
					}
				}
				flag:spear = {
					hidden_effect = {
						add_character_flag = spear_person
						set_variable = {
							name = signature_weapon
							value = flag:spear
						}
					}
				}
				flag:longaxe = {
					hidden_effect = {
						add_character_flag = longaxe_person
						set_variable = {
							name = signature_weapon
							value = flag:longaxe
						}
					}
				}
			}
		}
	}
}

lotr_set_signature_weapon_history_effect = {
	lotr_set_signature_weapon_effect = { WEAPON = $WEAPON$ }
}

lotr_apply_inactive_traits_on_death = {
	lotr_visible_disfigurement_death_effect = yes

	lotr_one_eyed_death_effect = yes

	if = {
		limit = {
			any_equipped_character_artifact = {
				artifact_slot_type = helmet
			}
		}
		if = {
			limit = {
				lotr_has_artifact_equipped = { ARTIFACT_VARIABLE = crown_woodland_realm }
			}
			make_trait_inactive = equipped_crown_woodland_realm
		}
		else_if = {
			limit = {
				exists = this
				is_alive = no
				OR = {
					this = character:lineofanarion53 # Castamir the Usurper
					this = character:lineofcastamir2 # Castaher
					this = character:lineofcastamir3 # Castamaitë
					this = character:lineofanarion55 # Morlaen
					this = character:lineofanarion56 # Caramir
					this = character:lineofcastamir3 # Castamaitë
					this = character:lineofcastamir4 # Angamaite
					this = character:lineofcastamir5 # Sangahyando
					this = character:lineofcastamir6 # Sangahyandion
					this = character:lineofcastamir7 # Castamir II
					this = character:lineofcastamir8 # Castamir III
				}
			}
			make_trait_inactive = equipped_castamiri_winged_crown
		}
		else_if = {
			limit = {
				lotr_has_artifact_equipped = { ARTIFACT_VARIABLE = crown_of_gondor }
			}
			make_trait_inactive = equipped_crown_of_gondor
		}
		else_if = {
			limit = {
				lotr_has_artifact_equipped = { ARTIFACT_VARIABLE = swan_circlet }
			}
			make_trait_inactive = equipped_swan_circlet
		}
	}
}

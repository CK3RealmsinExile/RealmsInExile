# Marshal Task Jank Effects 
lotr_apply_second_marshal_offensive_effect = { 
	if = {
		limit = {
			NOT = {
				is_target_in_global_variable_list = {
					name = unavailable_unique_decisions
					target = flag:reorganize_the_marshals
				}
			}
		}
		if = {
			limit = { $KINGOFROHAN$ = { second_martial_court_position = { VALUE = 5 } } }
			add_county_modifier = second_marshal_offensive_5_modifier
		}
		else_if = {
			limit = { $KINGOFROHAN$ = { second_martial_court_position = { VALUE = 4 } } }
			add_county_modifier = second_marshal_offensive_4_modifier
		}
		else_if = {
			limit = { $KINGOFROHAN$ = { second_martial_court_position = { VALUE = 3 } } }
			add_county_modifier = second_marshal_offensive_3_modifier
		}
		else_if = {
			limit = { $KINGOFROHAN$ = { second_martial_court_position = { VALUE = 2 } } }
			add_county_modifier = second_marshal_offensive_2_modifier
		}
		else = {
			limit = { $KINGOFROHAN$ = { second_martial_court_position = { VALUE = 1 } } }
			add_county_modifier = second_marshal_offensive_1_modifier
		}
	}
	else = {
		if = {
			limit = { $KINGOFROHAN$ = { eastmark_martial_court_position = { VALUE = 5 } } }
			add_county_modifier = second_marshal_offensive_5_modifier
		}
		else_if = {
			limit = { $KINGOFROHAN$ = { eastmark_martial_court_position = { VALUE = 4 } } }
			add_county_modifier = second_marshal_offensive_4_modifier
		}
		else_if = {
			limit = { $KINGOFROHAN$ = { eastmark_martial_court_position = { VALUE = 3 } } }
			add_county_modifier = second_marshal_offensive_3_modifier
		}
		else_if = {
			limit = { $KINGOFROHAN$ = { eastmark_martial_court_position = { VALUE = 2 } } }
			add_county_modifier = second_marshal_offensive_2_modifier
		}
		else = {
			limit = { $KINGOFROHAN$ = { eastmark_martial_court_position = { VALUE = 1 } } }
			add_county_modifier = second_marshal_offensive_1_modifier
		}
	}
}
lotr_remove_second_marshal_offensive_effect = {
	if = {
		limit = { has_county_modifier = second_marshal_offensive_5_modifier }
		remove_county_modifier = second_marshal_offensive_5_modifier

	if = {

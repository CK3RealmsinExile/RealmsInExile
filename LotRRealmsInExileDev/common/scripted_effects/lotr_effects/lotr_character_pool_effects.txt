# Requires imput of MIN and MAX age values
lotr_character_template_age_traits_effect = { # Adds/Fixes traits and lifespans
	if = {
		limit = { is_elf = yes }
		replace_unelven_traits = yes
	}
	if = {
		limit = { is_dwarf = yes }
		replace_undwarven_traits = yes
	}
	if = {
		limit = {
			is_human = yes
		}
		add_blood_of_numenor_effect = yes # Add Blood of Numenor traits where appropriate
	}
	if = {
	 	limit = { NOT = { additional_life_expectancy = 0 } } #Everyone except normies
		set_age = { $MIN$ $MAX$ }
	}
}

# Assign correct culture, faith and ethnicity to characters created in wilderness
lotr_character_template_wilderness_effect = {
	set_culture = root.culture
	set_character_faith = root.faith
	if = {
		limit = { 
			root = { 
				is_human = no
				cannot_be_killed = no
			}
		}
		if = {
			limit = {
				root = { is_orc = yes }
			}
			set_ethnicity = mordor_uruk_ethnicity
		}
		else_if = {
			limit = {
				root = { is_dwarf = yes }
			}
			set_ethnicity = dwarven_ethnicity
		}
		else_if = {
			limit = {
				root = { is_elf = yes }
			}
			set_ethnicity = elves_ethnicity
		}
		else_if = {
			limit = {
				root = { is_hobbit = yes }
			}
			set_ethnicity = hobbit_ethnicity
		}
		else = {
			set_ethnicity = druedain_ethnicity
		}
	}
}

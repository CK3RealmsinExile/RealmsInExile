lotr_random_visible_disfigurement_effect = {
	if = {
		limit = {
			has_trait = disfigured
			NOT = {
				lotr_has_visible_disfigurement_trigger = yes
			}
		}
		random_list = {
			1 = { make_trait_inactive = visible_disfigurement_1 }
			1 = { make_trait_inactive = visible_disfigurement_2 }
			1 = { make_trait_inactive = visible_disfigurement_3 }
			1 = { make_trait_inactive = visible_disfigurement_4 }
			1 = { make_trait_inactive = visible_disfigurement_5 }
		}
	}
}

lotr_remove_visible_disfigurement_effect = {
	if = {
		limit = {
			has_inactive_trait = visible_disfigurement_1
		}
		make_trait_active = visible_disfigurement_1
		remove_trait = visible_disfigurement_1
	}
	if = {
		limit = {
			has_inactive_trait = visible_disfigurement_2
		}
		make_trait_active = visible_disfigurement_2
		remove_trait = visible_disfigurement_2
	}
	if = {
		limit = {
			has_inactive_trait = visible_disfigurement_3
		}
		make_trait_active = visible_disfigurement_3
		remove_trait = visible_disfigurement_3
	}
	if = {
		limit = {
			has_inactive_trait = visible_disfigurement_4
		}
		make_trait_active = visible_disfigurement_4
		remove_trait = visible_disfigurement_4
	}
	if = {
		limit = {
			has_inactive_trait = visible_disfigurement_5
		}
		make_trait_active = visible_disfigurement_5
		remove_trait = visible_disfigurement_5
	}
}

lotr_visible_disfigurement_death_effect = {
	if = {
		limit = {
			has_character_modifier = wearing_disfigurement_mask_modifier
			NOT = { has_character_flag = disfigurement_fully_masked }
		}
		make_trait_inactive = disfigurement_partially_masked
	}
	else_if = {
		limit = {
			has_character_modifier = wearing_disfigurement_mask_modifier
			has_character_flag = disfigurement_fully_masked
		}
		make_trait_inactive = disfigurement_fully_masked
	}
}

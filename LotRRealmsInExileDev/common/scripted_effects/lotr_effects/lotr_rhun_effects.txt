### Cochineal Industry Effects ###
# Adapted from RICE

cochineal_decision_effect = {
	every_county_in_region = {
		region = special_cochineal_production_region
		custom = RICE_every_cochineal_region_county
		limit = {
			holder = root
		}
		if = {
			limit = {
				duchy = {
					any_de_jure_county = {
						any_county_province = {
							has_building_or_higher = cochineal_industry_03
						}
					}
				}
			}
			add_county_modifier = {
				modifier = cochineal_modifier_4
				years = 10
			}
		}
		else_if = {
			limit = {
				duchy = {
					any_de_jure_county = {
						any_county_province = {
							has_building_or_higher = cochineal_industry_02
						}
					}
				}
			}
			add_county_modifier = {
				modifier = cochineal_modifier_3
				years = 10
			}
		}
		else_if = {
			limit = {
				duchy = {
					any_de_jure_county = {
						any_county_province = {
							has_building_or_higher = cochineal_industry_01
						}
					}
				}
			}
			add_county_modifier = {
				modifier = cochineal_modifier_2
				years = 10
			}
		}
		else = {
			add_county_modifier = {
				modifier = cochineal_modifier_1
				years = 10
			}
		}
	}
}

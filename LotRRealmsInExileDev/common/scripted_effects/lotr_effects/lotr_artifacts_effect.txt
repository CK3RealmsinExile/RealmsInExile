elven_artifact_reclaimed_effect = {
	if = {
		limit = {
			scope:owner = { is_elf = yes }
			has_variable = elven
			NOT = { has_variable = elven_owned_artifact }
		} 
		set_variable = elven_owned_artifact
		if = {
			limit = {
				current_date > game_start_date #Don't give hope for artifacts generated on game start
			}
			if = {
				limit = {
					rarity = illustrious
				}
				scope:owner = {
					add_elven_hope_effect = { VALUE = major_elven_hope_gain HOPE_EFFECT = effect_elven_hope_illustrious_artifact_recovered }
				}
			}
			else = {
				scope:owner = {
					add_elven_hope_effect = { VALUE = medium_elven_hope_gain HOPE_EFFECT = effect_elven_hope_famed_artifact_recovered }
				}
			}
		}
	}
}

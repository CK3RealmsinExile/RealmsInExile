### Effects for sorcery Mechanic ###

sorcery_activity_success_change_effect = {
	scope:activity = {
		if = {
			limit = {
				NOT = { exists = var:sorcery_success_chance_event }
			}
			set_variable = { name = sorcery_success_chance_event value = 0 }
		}
		change_variable = {
			name = sorcery_success_chance_event
			add = sorcery_activity_success_$CHANGE$_value
		}
		set_variable = {
			name = sorcery_success_chance
			value = sorcery_success_chance_value
		}
	}
	custom_tooltip = sorcery_activity_success_$CHANGE$_tt
}

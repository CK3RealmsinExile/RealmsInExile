devastate_county_effect = { # Fires on county scope, takes paramter $SEVERITY$, which quantifies the destruction and ACTOR for logging purposes
	debug_log = "Devestation attempted"
	if = {
		limit = { # Don't devastate Tom
			holder = { is_wastelands = no }
			title_province = { province_is_in_angmar_wars = yes } # You need to be in Angmar Wars
		}

		while = { #First do random raid damage
			count = $SEVERITY$.div2
			devastate_county_secondary_effect = yes # Pass through to selection effect
		}
		while = { # Then systematically ruin the county.
			count = $SEVERITY$
			devastate_county_repeated_effect = yes # Pass through to selection effect
		}
		save_scope_as = devastated_county
		angmar_wars_catalyst_angmar_devastated_effect = yes 
		angmar_wars_catalyst_arnor_devastated_effect = yes 
		holder = { #Fire notification for players associated to devastation
			if = {
				limit = {
					NOT = { has_character_flag = devastation_notifications_supressed }
				}
				send_interface_toast = {
					type = event_toast_effect_bad
					title = county_devastated_toast
					custom_tooltip = county_devastated_toast_desc
				}
			}
			every_liege_or_above = {
				limit = {
					exists = this
					# is_ai = no # Players #Commented out for testing
				}
				if = {
					limit = {
						NOT = { has_character_flag = devastation_notifications_supressed }
					}
					send_interface_toast = {
						type = event_toast_effect_bad
						title = county_devastated_toast
						custom_tooltip = county_devastated_toast_desc
					}
				}
			}
		}
		$ACTOR$ = {
			if = {
				limit = {
					NOT = { has_character_flag = devastation_notifications_supressed }
				}
				send_interface_toast = {
					type = event_toast_effect_bad
					title = county_devastated_toast
					custom_tooltip = county_devastated_toast_desc
				}
			}
		}
		debug_log = "ANGMAR TEST - County Devastated: $SEVERITY$ - $ACTOR$ "
		debug_log_scopes = yes
	}
	else = {
		# Do nothing
		# if = {
		# 	limit = {
		# 		holder = { is_wastelands = yes }
		# 	}
		# 	debug_log = "ANGMAR TEST - Tried to Devastate Tom"
		# }
	}
}

devastate_county_repeated_effect = { # Fires on county scope, removes one level each time fired
	if = { # Don't fire this on Tom
		limit = {
			holder = { is_wastelands = yes }
		}
		# Do nothing
		debug_log = "ANGMAR TEST - Tried to Devastate Tom (Check 2)"
		error_log = "ANGMAR TEST - Tried to Devastate Tom"
	}
	else_if = { # First, reduce development
		limit = {
			development_level > 0
		}
		change_development_level = -1
	}
	else_if = { # If there's not enough dev, start on buildings
		limit = {
			any_county_province = {
				has_holding = yes
				lotr_normal_building_levels > 0
			}
		}
		ordered_county_province = {
			order_by = combined_building_level.inverse # Take least built up place (that has enough levels) first
			limit = {
				has_holding = yes
				lotr_normal_building_levels > 0
			}
			destroy_random_building_variable_effect = yes
			destroy_random_building_effect = yes
			debug_log = "ANGMAR TEST - destroyed building"
		}
	}
	else_if = { # If no province has buildings, start taking holding levels in outer provinces.
		limit = {
			any_county_province = {
				has_holding = yes
				is_county_capital = no # Can't destroy the capital holding yet
			}
		}
		ordered_county_province = {
			order_by = lotr_holding_level.inverse # Take least built up place first
			limit = {
				has_holding = yes
				is_county_capital = no
			}
			if = {
				limit = {
					lotr_holding_level > 1
				}
				lotr_downgrade_holding = yes
				debug_log = "ANGMAR TEST - downgraded holding"
			}
			else = {
				remove_holding = yes
				debug_log = "ANMGAR TEST - removed holding"
			}
		}
	}
	else_if = { #Now start reducing the capital holding
		limit = {
			title_province = {
				lotr_holding_level > 1
			}
		}
		title_province = {
			lotr_downgrade_holding = yes
			debug_log = "ANMGAR TEST - downgraded capital holding"
		}
	}
	else_if = { # If down to just one level of holding in capital, convert to wilderness
		limit = {
			title_province = {
				lotr_holding_level = 1
			}
		}
		make_settlement_county_wilderness = { COUNTY = this }
		debug_log = "ANGMAR TEST - wastelanded county"
	}
}

devastate_county_secondary_effect = {
	random_list = {
		40 = {} # Nothing happens

		5 = { # Rarely take an extra dev
			if = {
				limit = {
					development_level > 0
				}
				change_development_level = -1
			}
		}
		10 = { # Steal some money
			holder = {
				remove_short_term_gold = miniscule_gold_value
			}
		}
		45 = { # Destroy a building
			if = {
				limit = {
					any_county_province = {
						has_holding = yes
						lotr_normal_building_levels > 0
					}
				}
				ordered_county_province = {
					order_by = combined_building_level.inverse # Take least built up place first
					limit = {
						has_holding = yes
						lotr_normal_building_levels > 0
					}
					destroy_random_building_variable_effect = yes
					destroy_random_building_effect = yes
				}
			}
		}
	}
}

devastate_random_sitatuation_county = { # Run from situation scope, effects up to MAXIMUM counties
	ordered_situation_county = {
		limit = {
			holder = {
				save_temporary_scope_as = target_county_holder
			}
			any_neighboring_county = {
				holder = {
					OR = {
						is_wastelands = yes
						angmar_wars_is_opposite_side = { CHARACTER = scope:target_county_holder }
					}
				}
			}
			holder = {
				is_elven_ruler = no
				is_wastelands = no
			}
		}
		order_by = devastation_potential
		max = 2
		check_range_bounds = no
		devastate_county_effect = { SEVERITY = 1 ACTOR = title:k_wastelands.holder }
	}
}

angmar_wars_catalyst_major_epidemic_angmar_effect = { # A major epidemic hits Angmar
	if = {
		limit = {
			county.holder = {
				is_in_angmar_wars = yes
				is_angmar_aligned = yes
			}
			scope:epidemic = {
				outbreak_intensity = major
				NOT = { has_variable = registered_for_catalyst } #No double counting
			}
		}
		situation:angmar_wars_situation = {
			trigger_situation_catalyst = {
				catalyst = catalyst_major_epidemic_angmar
				character = root.county.holder
				county = root.county
			}
		}
		scope:epidemic = {
			set_variable = registered_for_catalyst #No double counting
		}
	}
}

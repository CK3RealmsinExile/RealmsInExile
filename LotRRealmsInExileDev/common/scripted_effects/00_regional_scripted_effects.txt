						faith_hostility_level = {
							target = scope:courtier_to_check.faith
							value >= faith_hostile_level
						}
					}
				}
			}
			random_courtier_or_guest = {
				limit = {

				}

			}
		}
		100 = { # Vassal in your realm with hostile faith
			trigger = {
				top_liege = {
					any_vassal_or_below = {

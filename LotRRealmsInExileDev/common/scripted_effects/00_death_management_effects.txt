silent_disappearance_effect = {
	hidden_effect = {
		if = {
			limit = { 
				is_alive = yes 
				cannot_be_killed = no #LotR
			}

			death = {
		 		death_reason = death_disappearance
		 	}
		}
	}
}

silent_dissapear_ai_if_not_hired = {
	hidden_effect = {
		if = {
			limit = {
				is_alive = yes
				root = { is_ai = yes }
				NOT = { employer = root }
				cannot_be_killed = no #LotR
			}

			death = {
		 		death_reason = death_disappearance
		 	}
		}
		else_if = {
			limit = {
				is_alive = yes
				NOT = { employer = root }
			}
			move_to_pool = yes
		}
	}
}

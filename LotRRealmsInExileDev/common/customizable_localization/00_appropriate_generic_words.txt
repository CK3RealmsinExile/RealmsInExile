		trigger = {

				value < -50
			}
		}
		localization_key = scorn_reaction
	}
	text = {
		trigger = {
			OR = {
				scope:second = {
					has_trait = greedy

				}
				reverse_opinion = {
					target = scope:second
					value < 0
				}
			}

	}
	text = {

			reverse_opinion = {

GetPrideEnvy = {
	type = character
	random_valid = yes
	text = {
		trigger = {
			reverse_opinion = {
				target = scope:second
				value < -50
			}
		}
		localization_key = scorn_reaction
	}
	text = {
		trigger = {
			OR = {
				scope:second = {
					has_trait = greedy
					has_trait = ambitious
				}
				reverse_opinion = {
					target = scope:second
					value < 0
				}
			}
		}
		localization_key = envy_reaction
	}
	text = {
		trigger = {
			reverse_opinion = {
				target = scope:second
				value > 60
			}
		}
		localization_key = pride_reaction
	}
	text = {
		localization_key = surprise_reaction
	}
}

		trigger = {
			opinion = {
				target = scope:second

	type = character

	text = {
		localization_key = house_feud_balance_terrible
		trigger = {
			house.house_head = {
				any_owned_story = {
					story_type = story_cycle_house_feud
					house_feud_balance_value <= -3
				}
			}
		}
	}
	text = {
		localization_key = house_feud_balance_bad
		trigger = {

				}

		}
	}

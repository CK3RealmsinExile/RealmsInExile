
# root = domicile
# scope:owner = <character> owner of domicile
#this on_action happens on every province during a travel plan with moving domicile, so be careful what you put inside here!
on_domicile_moved = {
	effect = {
		if = {
			limit = {
				exists = scope:owner
			}
			scope:owner = { remove_variable = ep3_laamp_decision_1000_is_visiting }
		}
	}
}

# root = domicile
# scope:owner = <character> owner of domicile
on_domicile_building_started = {
	on_actions = { lotr_on_domicile_building_started } #LotR
	effect = {
		scope:owner = {
			send_interface_message = {
				type = msg_domicile_building_started
				left_icon = this
			}
		}
		if = {
			limit = {
				scope:owner = { has_character_modifier = tgp_subsidised_estate_buildings_modifier }
				has_variable = subsidised_estate_var 
				var:subsidised_estate_var > 1
			}
			change_variable = {
				name = subsidised_estate_var
				subtract = 1
			}
			scope:owner = {
				remove_character_modifier = tgp_subsidised_estate_buildings_modifier
				add_character_modifier = tgp_subsidised_estate_buildings_modifier
			}
		}
		else = {
			scope:owner = {
				remove_character_modifier = tgp_subsidised_estate_buildings_modifier
			}
			remove_variable = subsidised_estate_var
		}
	}
}

random_yearly_everyone_pulse = {
	on_actions = {
		lotr_visible_disfigurement_pulse # Used for Flavor and Special Events
	}
}

lotr_visible_disfigurement_pulse = {
	trigger = {
		has_trait = disfigured
		lotr_has_visible_disfigurement_trigger = no
	}
	effect = {
		trigger_event = {
			id = lotr_health_events.3000
		}
	}
}

lotr_one_eyed_pulse = {
	trigger = {
		has_trait = one_eyed
		NOT = { has_character_flag = one_eyed_healed }
	}
	effect = {
		if = {
			limit = { is_ai = yes }
			lotr_random_one_eyed_fashion_effect = yes
			add_character_flag = one_eyed_healed
		}
		else = {
			trigger_event = lotr_health_events.4000
		}
	}
}

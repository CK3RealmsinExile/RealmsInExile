angmar_wars_situation_yearly_pulse = { #Fires on Situtation
	effect = {
		# Spread this pulse out
		every_participant_group = {
			every_situation_group_participant = {
				trigger_event = {
					on_action = angmar_wars_situation_random_events
					days = { 1 300 }
				}
			}
		}
	}
}
angmar_wars_situation_random_events = { #Fires for participants
	trigger = {
		is_in_angmar_wars = yes
	}
	random_events = {
		chance_to_happen = 10

		### Arnor Flavor Events ###
		1 = angmar_wars.2001 #Orc Raid
		1 = angmar_wars.2002 #Bandit Raid
		1 = angmar_wars.2004 #Peasants Abandon the Frontier
	}
}

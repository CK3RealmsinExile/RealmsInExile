# character having its birthday in root scope
# tested after age has increased


on_lotr_old_age_birthday = {
	trigger = {
		additional_life_expectancy > 0
		NOT = { is_elf = yes }
		cannot_be_killed = no
		health_age_equivalent = 50
	}
	effect = { #Removal of health bonus for elderly characters
		if = { #Youthful vigor 1
			limit = { has_character_modifier = youthful_vigor_1_modifier }
			remove_character_modifier = youthful_vigor_1_modifier
		}
		else_if = { #Youthful vigor 1.5
			limit = { has_character_modifier = youthful_vigor_1_5_modifier }
			remove_character_modifier = youthful_vigor_1_5_modifier
		}
		else_if = { #Youthful vigor 2
			limit = { has_character_modifier = youthful_vigor_2_modifier }
			remove_character_modifier = youthful_vigor_2_modifier
		} 
	}
}

on_lotr_nubile_age_birthday = {
	trigger = {
		age = nubile_age
	}
	events = {
		delay = { days = 365 }
		race.0004 #Triggering marriage for betrothed nubiles if AI doesn't manage it
	}
}

on_lotr_majority_birthday = {
	trigger = {
		age = majority_age
	}
	effect = {
		if = { #Heirloom artifact inheritance
			limit = {
				dynasty ?= {
					any_dynasty_member = {
						primary_heir ?= root
						any_character_artifact = {
							has_variable = heirloom_of_title
							prev = var:heirloom_of_title.holder
						}
					}
				}
			}
			dynasty ?= {
				every_dynasty_member = {
					limit = {
						primary_heir ?= root
						any_character_artifact = {
							has_variable = heirloom_of_title
							prev = var:heirloom_of_title.holder
						}
					}
					trigger_event = {
						id = lotr_artifact.0010
						days = 5
					}
				}
			}
		}
	}
}

on_lotr_coronation_age_birthday = {
	trigger = {
		OR = {
			has_character_modifier = dwarven_child_legitimacy_boost_modifier
			has_character_modifier = hobbit_child_legitimacy_boost_modifier
			has_character_modifier = elven_child_legitimacy_boost_modifier
		}
		age = lotr_coronation_minimum_age_value
	}
	remove_character_modifier = dwarven_child_legitimacy_boost_modifier
	remove_character_modifier = hobbit_child_legitimacy_boost_modifier
	remove_character_modifier = elven_child_legitimacy_boost_modifier
}

elven_hope_monthly_pulse = { # Change situation phases according to current hope value
	effect = {
		if = {
			limit = {
				situation_top_has_catalyst = catalyst_elven_very_low_hope
				this.var:elven_hope_value <= 20
			}
			trigger_situation_catalyst = catalyst_elven_very_low_hope
		}
		else_if = {
			limit = {
				situation_top_has_catalyst = catalyst_elven_low_hope
				this.var:elven_hope_value <= 40
			}
			trigger_situation_catalyst = catalyst_elven_low_hope
		}
		else_if = {
			limit = {
				situation_top_has_catalyst = catalyst_elven_medium_hope
				this.var:elven_hope_value <= 60
			}
			trigger_situation_catalyst = catalyst_elven_medium_hope
		}
		else_if = {
			limit = {
				situation_top_has_catalyst = catalyst_elven_high_hope
				this.var:elven_hope_value <= 80
			}
			trigger_situation_catalyst = catalyst_elven_high_hope
		}
		else_if = {
			limit = {
				situation_top_has_catalyst = catalyst_very_elven_high_hope
				this.var:elven_hope_value <= 100
			}
			trigger_situation_catalyst = catalyst_elven_very_high_hope
		}
	}
}

#LOTR Stress Related on_actions

lotr_stress_loss_pipeweed = {
	# Set up the on_action this way so the chance of the generic 'Good Morning' event vs any rarer event is always 85%
	effect = {
		random_list = {
			85 = { # Standard Decision Event
				trigger_event = hobbit.0002 # Good Morning!
			}
			15 = { # Rarer Decision Events
				trigger_event = {
					on_action = lotr_pipeweed_event_pool
				}
			}
		}
	}
}

lotr_pipeweed_event_pool = { # Rarer Pipeweed Events
	random_events = {
		10 = hobbit.0003 # The Munchies
		10 = hobbit.0004 # Cloudgazing
	}
}

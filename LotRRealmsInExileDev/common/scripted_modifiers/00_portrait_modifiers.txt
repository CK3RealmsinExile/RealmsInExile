### Court Amenity triggers get used a lot in the portrait modifiers. We can use these scripted modifiers instead ###

court_fashion_level_4_modifier = {
    modifier = {
        add = 50
        court_fashion_level_4_trigger = {
            CULTURE_INPUT = $INPUT_PARAMETER$
        }
    }
}

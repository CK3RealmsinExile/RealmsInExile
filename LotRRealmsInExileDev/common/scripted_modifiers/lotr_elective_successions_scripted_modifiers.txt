	modifier = {
		add = 15

	modifier = {

		this = scope:candidate
		has_trait = ambitious
		is_adult = yes
		add = {
			add = 40
			if = {
				limit = {
					ai_boldness > 0
				}
				add = this.ai_boldness
			}
			if = {
				limit = {
					ai_greed > 0
				}
				add = this.ai_greed
			}
		}
	}
	modifier = {
		add = 100
		desc = tooltip_feudal_elector_vote_myself_ruler_choice # I have been chosen by the Leader
		this = scope:candidate
		this = scope:holder_candidate
		is_adult = yes
	}
	modifier = {
		desc = tooltip_feudal_elector_vote_myself_ruling_dynasty # I belong to the ruling dynasty
		this = scope:candidate
		exists = this.dynasty
		exists = scope:holder.dynasty
		this.dynasty = scope:holder.dynasty
		is_adult = yes

			value = 40

				limit = {

			}

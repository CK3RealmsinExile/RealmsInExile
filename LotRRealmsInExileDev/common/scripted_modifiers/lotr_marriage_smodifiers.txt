		}

lotr_elf_court_scheme_success_modifier = {
	# LotR - Elven Laamps have a hard time romancing landed rulers and their close family
	modifier = {
		trigger = { 
			scope:owner = {
				is_landed_or_landless_administrative = no 
				is_elf = yes
			}
		}
		add = -150
		desc = "ELVEN_LAAMP_COURT_HANDICAP"
		scope:target = {
			exists = scope:target.liege
			OR = {
				is_spouse_of = liege
				is_close_family_of = liege
			}
		}
	}
}

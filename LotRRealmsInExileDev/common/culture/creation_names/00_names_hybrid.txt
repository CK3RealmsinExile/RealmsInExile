# Broadbeam
broadbeard = { # Broadbeam-Longbeard Fusion
	trigger = {
		OR = { # Ensures it can be formed in either direction
			AND = {
				scope:culture = { has_cultural_pillar = heritage_longbeard }
				scope:other_culture = { has_cultural_pillar = heritage_broadbeam }
			}
			AND = {
				scope:culture = { has_cultural_pillar = heritage_broadbeam }
				scope:other_culture = { has_cultural_pillar = heritage_longbeard }
			}
		}
	}
	hybrid = yes
}

calenardhrim = {
	trigger = {
		capital_county = {
			title_province = { geographical_region = custom_gondor_calenardhon  }
		}
		any_sub_realm_county = {
			scope:other_culture = { has_cultural_pillar = heritage_hadorim	}
			title_province = {
				geographical_region = custom_gondor_calenardhon
			}
		}
		scope:culture = { has_cultural_pillar = heritage_dunedain }
	}
	hybrid = yes
}

hyarmenedain = { # Dúnedain of the Far South
	trigger = {
		capital_county = {
			title_province = { # Capital is in southern Middle-Earth (Umbar or South)
				OR = {
					geographical_region = middleearth_haradwaith_far
					geographical_region = middleearth_haradwaith_near_umbar 
				}
			}
		}
		any_sub_realm_county = { 
			OR = {
				culture = scope:culture
				culture = scope:other_culture
			}
			title_province = {
				OR = {
					geographical_region = middleearth_haradwaith_far
					geographical_region = middleearth_haradwaith_near_umbar 
				}
			}
		} 
		scope:culture = { has_cultural_pillar = heritage_dunedain } # Only formed by Dúnedain - it's "South" only in relation to Arnor+Gondor
		OR = {
			scope:other_culture = { has_cultural_pillar = heritage_northern_adunaic }
			scope:other_culture = { has_cultural_pillar = heritage_bellakari }
			scope:other_culture = { has_cultural_pillar = heritage_southern_adunaic }
		}
	}
	hybrid = yes
}

### Gondorian Provincial Cultures ###

anoredain = { # Dúnedain People of Anorien
	trigger = {
		OR = {
			capital_county = {
				title:k_anorien = {
					is_de_jure_liege_or_above_target = prev
				}
			}
			has_primary_title = title:k_anorien
		}
		scope:culture = { has_cultural_pillar = heritage_dunedain }
		any_sub_realm_county = {
			culture = { has_cultural_pillar = heritage_dunedain }
			kingdom = title:k_anorien
		}
	}
}

tradition_abrakhan_magic_cities = {
	# Replaces Realms 2.2 MaA Innovation and culture's stats.
	category = regional
	layers = {
		0 = learning
		1 = mena
		4 = city2.dds
	}
	is_shown = {
		has_cultural_pillar = heritage_middle_apysaic
	}
	can_pick = {
		OR = {
			this = culture:old_pust
			any_parent_culture_or_above = { this = culture:old_pust }
		}
		custom_description = {
			text = culture_in_red_desert_desc
			any_culture_county = {
				any_county_province = {
					terrain = red_desert
				}
			}
		}
		trigger_if = { #LotR
            limit = {
                scope:replacing ?= culture_tradition:tradition_variag_succession
            }
            custom_tooltip = {
                text = cannot_replace_variag_succession
                NOT = { has_cultural_tradition = tradition_variag_succession } #To prevent stacking after hybridization
            }
        }
	}
	character_modifier = {
		monthly_learning_lifestyle_xp_gain_mult = 0.05
		republic_government_vassal_opinion = 10
	}
	county_modifier = {
		development_growth_factor = 0.05
	}
	parameters = {
		unlock_maa_guard_of_abrakhan = yes
		unlock_maa_risen_carrion = yes
		castle_holding_fort_level_in_red_desert_bonus = yes
		city_holding_fort_level_in_red_desert_bonus = yes
	}
	cost = {
		prestige = {
			add = {
				value = tradition_base_cost
				desc = BASE
				format = "BASE_VALUE_FORMAT"
			}
			if = {
				limit = {
					NOT = {
						any_culture_county = {
							percent >= 0.3
							any_county_province = {
								OR = {
									terrain = dune_sea
									terrain = desert
								}
							}
						}
					}
				}
				add = {
					value = tradition_unfulfilled_criteria_penalty
					desc = dry_percentage_desc
				}
			}
			multiply = tradition_replacement_cost_if_relevant
		}
	}
	ai_will_do = { value = 1 }
}

	is_shown = {

			this = culture:nafarati
			any_parent_culture_or_above = {
				this = culture:nafarati
			} 
		}
	}
	
	can_pick = {
		culture_not_pacifistic_trigger = yes

		# 		any_county_province = {
		# 			geographical_region = middleearth_uttersouth_mumakan

		monthly_county_control_growth_factor = -0.4
		vassal_opinion = -10
		opinion_of_liege = -10
		tolerance_advantage_mod = 5
		
		ai_honor = -20

	}
	
	cost = {
		prestige = {
			add = {
				value = tradition_base_cost
				desc = BASE
				format = "BASE_VALUE_FORMAT"
			}
			if = {
				limit = {
					NOR = {
						culture_pillar:ethos_bellicose = { is_in_list = traits }
						culture_pillar:ethos_spiritual = { is_in_list = traits }
					}
				}
				add = {
					value = tradition_incompatible_ethos_penalty
					desc = not_bellicose_or_spiritual_desc
				}
			}
			if = {
				limit = {
					trigger_if = {
						limit = {
							scope:character = {
								is_ai = no
							}
						}
						NOT = {
							any_ruler = {
								count >= 5
								culture = prev
								primary_title.tier >= tier_county
								is_at_war = yes
							}
						}
					}
					trigger_else = {
						always = no
					}
				}
				add = {
					value = tradition_unfulfilled_criteria_penalty
					desc = rulers_at_war_desc
				}
			}
			
			multiply = tradition_replacement_cost_if_relevant
		}
	}
	
	ai_will_do = {
		value = 100
		if = {
			limit = {
				NOT = {
					scope:character = {
						government_has_flag = government_is_tribal
					}
				}
			}
			multiply = 0.25
		}
		if = {
			limit = {
				NOR = {
					#any_in_list = { list = traits this = flag:heritage_turkic }
					#any_in_list = { list = traits this = flag:heritage_mongolic }
					is_divergent_culture = yes
				}
			}
			multiply = 0
		}
	}
}

tradition_hadorim_forest_wardens = {
	# Replaces Realms 2.2 MaA Innovation.
	# Rohirrim doesn't start with this tradition. Although, Rohirrim can grab the MaA for balance.
	category = regional
	layers = {
		0 = martial
		1 = western
		4 = forest.dds
	}
	is_shown = {
		has_cultural_pillar = heritage_eotheod
	}
	can_pick = {
		culture_not_pacifistic_trigger = yes
		has_cultural_pillar = heritage_eotheod
		trigger_if = { # To prevent stacking.
			limit = {
				OR = {
					NOT = { exists = scope:replacing }
					NOT = { scope:replacing = flag:tradition_forest_fighters }
				}
			}
			NOT = { has_cultural_tradition = tradition_forest_fighters }
		}
		custom_description = {
			text = culture_in_forest_desc
			any_culture_county = {
				any_county_province = {
					OR = {
						terrain = forest
						terrain = taiga
						terrain = mallorn_forest
						terrain = deep_forest
					}
				}
			}
		}
		trigger_if = { #LotR
            limit = {
                scope:replacing ?= culture_tradition:tradition_variag_succession
            }
            custom_tooltip = {
                text = cannot_replace_variag_succession
                NOT = { has_cultural_tradition = tradition_variag_succession } #To prevent stacking after hybridization
            }
        }
	}
	parameters = {
		unlock_maa_hadorim_forest_wardens = yes
		forest_fighter_trait_more_common = yes
		forest_trait_bonuses = yes
		can_recruit_forest_specialist = yes
	}
	province_modifier = {
		forest_levy_size = 0.1
		taiga_levy_size = 0.1
		deep_forest_levy_size = 0.1
		mallorn_forest_levy_size = 0.1
		forest_supply_limit_mult = 0.25
		taiga_supply_limit_mult = 0.25
		deep_forest_supply_limit_mult = 0.25
		mallorn_forest_supply_limit_mult = 0.25
	}
	cost = {
		prestige = {
			add = {
				value = tradition_base_cost
				desc = BASE
				format = "BASE_VALUE_FORMAT"
			}
			# Terrain requirement.
			if = {
				limit = {
					NOT = {
						any_culture_county = {
							percent >= 0.3
							any_county_province = {
								OR = {
									terrain = forest
									terrain = taiga
									terrain = deep_forest
									terrain = mallorn_forest
								}
							}
						}
					}
				}
				add = {
					value = tradition_unfulfilled_criteria_penalty
					desc = lotr_forest_percentage_desc
				}
			}
			# Ethos requirement.
			if = {
				limit = {
					NOR = {
						culture_pillar:ethos_stoic = { is_in_list = traits }
						culture_pillar:ethos_spiritual = { is_in_list = traits }
					}
				}
				add = {
					value = tradition_incompatible_ethos_penalty
					desc = not_stoic_or_spiritual_desc
				}
			}
			# Certain number of archer MaA requirement.
			if = {
				limit = {
					scope:character = {
						number_maa_soldiers_of_base_type = {
							type = archers
							value < 600
						}
					}
				}
				add = {
					value = tradition_unfulfilled_criteria_penalty
					desc = culture_head_lacks_archer_maa_desc
				}
			}
			multiply = tradition_replacement_cost_if_relevant
		}
	}
	ai_will_do = { value = 1 }
}

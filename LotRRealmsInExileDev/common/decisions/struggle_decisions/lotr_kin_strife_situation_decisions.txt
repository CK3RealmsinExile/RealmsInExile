lotr_kin_strife_support_leader_of_situation_as_dynasty = {
	decision_group_type = kin_strife_decisions

	picture = {
		reference = "gfx/interface/illustrations/decisions/struggle_thani_hazad_dissolution.dds"
	}

	cooldown = { years = 20 }

	is_shown = {
		situation:kin_strife = {
			any_participant_group = {
				any_situation_group_participant = { this = root }
			}
		}
	}

	is_valid_showing_failures_only = {
		situation:kin_strife.situation_participant_group:loyalist_leader = { NOT = { participant_group_has_character = root } }
		situation:kin_strife.situation_participant_group:purist_leader = { NOT = { participant_group_has_character = root } }
	}
	
	widget = {
		gui = "decision_view_widget_option_list_generic"
		controller = decision_option_list_controller
		decision_to_second_step_button = "SELECT_ACTION"
		show_from_start = yes

		item = { # Support Loyalist
			value = support_loyalist_decision
			is_valid = {
				NOT = { dynasty = { has_dynasty_modifier = kin_strife_support_loyalist } }
			}
			current_description = support_loyalist_decision_desc
			localization = support_loyalist_decision
			icon = "gfx/interface/icons/modifiers/icon_prestige_05.dds"
			ai_chance = {
				value = 100
			}
		}

		item = { # Support Purist
			value = support_purist_decision
			is_valid = {
				NOT = { dynasty = { has_dynasty_modifier = kin_strife_support_purist } }
			}
			current_description = support_purist_decision_desc
			localization = support_purist_decision
			icon = "gfx/interface/icons/character_interactions/blood_brother.dds"
			ai_chance = {
				value = 100
			}
		}
	}

	effect = {
		if = {
			limit = { scope:support_loyalist_decision = yes }
			dynasty = { add_dynasty_modifier = kin_strife_support_loyalist }
		}
		else_if = {
			limit = { scope:support_purist_decision = yes }
			dynasty = { add_dynasty_modifier = kin_strife_support_purist }
		}
	}

	ai_check_interval_by_tier = {
		barony = 0
		county = 120
		duchy = 120
		kingdom = 120
		empire = 120
		hegemony = 120
	}
}

lotr_kin_strife_support_different_faction = {
	decision_group_type = kin_strife_decisions

	picture = {
		reference = "gfx/interface/illustrations/decisions/struggle_thani_hazad_dissolution.dds"
	}
	is_shown = {
		situation:kin_strife = {
			any_participant_group = {
				any_situation_group_participant = { this = root }
			}
		}
	}

	is_valid_showing_failures_only = {
		situation:kin_strife.situation_participant_group:loyalist_leader = { NOT = { participant_group_has_character = root } }
		situation:kin_strife.situation_participant_group:purist_leader = { NOT = { participant_group_has_character = root } }
	}

	cooldown = { years = 10 }
	
	widget = {
		gui = "decision_view_widget_option_list_generic"
		controller = decision_option_list_controller
		decision_to_second_step_button = "SELECT_ACTION"
		show_from_start = yes

		item = { # Become a Loyalist
			value = become_loyalist_decision
			is_valid = {
				NOT = { has_trait = kin_strife_loyalist }
			}
			current_description = become_loyalist_decision_desc
			localization = become_loyalist_decision
			icon = "gfx/interface/icons/modifiers/icon_prestige_05.dds"
			ai_chance = {
				value = 100
			}
		}

		item = { # Become a Purist
			value = become_purist_decision
			is_valid = {
				NOT = { has_trait = kin_strife_purist }
			}
			current_description = support_purist_decision_desc
			localization = become_purist_decision
			icon = "gfx/interface/icons/character_interactions/blood_brother.dds"
			ai_chance = {
				value = 100
			}
		}

		item = { # Become a Neutral
			value = become_neutral_decision
			is_valid = {
				OR = {
					has_trait = kin_strife_purist
					has_trait = kin_strife_loyalist
				}
			}
			current_description = become_neutral_decision_desc
			localization = become_neutral_decision
			icon = "gfx/interface/icons/character_interactions/blood_brother.dds"
			ai_chance = {
				value = 100
			}
		}
	}

	effect = {
		if = {
			limit = { scope:become_loyalist_decision = yes }
			if = {
				limit = { has_trait = kin_strife_purist }
				remove_trait = kin_strife_purist
			}
			add_trait = kin_strife_loyalist
		}
		else_if = {
			limit = { scope:become_purist_decision = yes }
			if = {
				limit = { has_trait = kin_strife_loyalist }
				remove_trait = kin_strife_loyalist
			}
			add_trait = kin_strife_purist
		}
		else_if = {
			limit = { scope:become_neutral_decision = yes}
			if = {
				limit = { has_trait = kin_strife_loyalist }
				remove_trait = kin_strife_loyalist
			}
			if = {
				limit = { has_trait = kin_strife_purist }
				remove_trait = kin_strife_purist
			}
		}
	}

	ai_check_interval_by_tier = {
		barony = 0
		county = 60
		duchy = 60
		kingdom = 60
		empire = 60
		hegemony = 60
	}
}

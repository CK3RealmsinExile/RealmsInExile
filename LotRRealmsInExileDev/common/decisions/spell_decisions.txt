spell_summon_urukhai = {
	picture = "gfx/interface/illustrations/decisions/decision_urukhai_01.dds"

	desc = spell_summon_urukhai_desc
	selection_tooltip = spell_summon_urukhai_tooltip

	ai_goal = yes
	cooldown = { days = 3000 }

	is_shown = {
		can_cast_spells_trigger = yes
		faith = { religion = { is_in_family = rf_forcesofevil } }
		has_character_flag = saruman_has_urukhai
		NOT = { has_character_modifier = spell_preparing_urukhai_modifier }
	}

	is_valid_showing_failures_only = {
		has_capacity = { COST = 1 }
		can_cast_spells_trigger = yes
	}

	cost = {
#		gold = summon_levy_cost
		gold = {
			value = monumental_gold_value
			if = {
				limit = {
					is_ai = yes
				}
				multiply = 0
			}
			multiply = 2
		}
		piety = {
			value = massive_piety_value
			if = {
				limit = {
					is_ai = yes
				}
				multiply = 0
			}
			multiply = 1
		}
	}

	effect = {
		show_as_tooltip = {
			spawn_uruk_levy = yes
		}
		add_character_modifier = {
			modifier = spell_preparing_urukhai_modifier
			days = 365
		}
		trigger_event = {
			id = spell.0001
			days = 366
		}
	}
	
	ai_potential = {
		always = yes
	}

	ai_will_do = {
		base = 100
	}
}

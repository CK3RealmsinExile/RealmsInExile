decision_revive_gundabad = { # Re-establish your kingdom
	title = decision_revive_gundabad

	major = yes
	ai_goal = yes

	selection_tooltip = decision_revive_gundabad_tooltip
	desc = decision_revive_gundabad_desc

	is_shown = {
		is_landed = yes
		is_orc = yes
		faith_is_evil = yes
		OR = {
			any_held_title = {
				kingdom = title:k_gundabad
			}
			dynasty = dynasty:dynasty_ashmog
		}
		NOT = {
			is_target_in_global_variable_list = {
				name = unavailable_unique_decisions

		}

		is_independent_ruler = yes
		has_title = title:k_gundabad

		is_imprisoned = no
		is_available_adult = yes
	}
	
	effect = {
		add_character_modifier = {
			modifier = gundabad_rising_character_modifier
			years = 10
		}
		add_prestige = 150
		dynasty = { add_dynasty_prestige = 50 }
		custom_tooltip = unlock_decision_gundabad_ignite_the_forges
		custom_tooltip = unlock_decision_gundabad_angmar_mountains
		custom_tooltip = unlock_decision_gundabad_dominion_over_langwell
		
		# trigger_event = {
		# 	id = gundabad.0200
		# }
		if = {
			limit = {
				NOT = { any_artifact = { has_variable = ulukai } }
			}
			create_artifact_ulukai_effect = { OWNER = root }
		}
		else = {
			random_artifact = {
				limit = { has_variable = ulukai }
				save_scope_as = ulukai_artifact
			}
			add_personal_artifact_claim = scope:ulukai_artifact
		}
		add_to_global_variable_list = {
			name = unavailable_unique_decisions
			target = flag:decision_revive_gundabad
		}
	}

	ai_potential = {
		always = yes
	}	

	ai_will_do = {
		base = 100
	}
}

### Convert to Oligarchy (from Feudal/Clan/Admin) ###
decision_adopt_oligarchy_from_feudal = {
	title = decision_adopt_oligarchy_from_feudal
	picture = {
		reference = "gfx/interface/illustrations/decisions/decision_council_03.dds"
	}
	
	ai_goal = no
	ai_potential = { 
		always = no
	}
	ai_will_do = { 
		base = 0
	}
	ai_check_interval_by_tier = {
		barony = 0
		county = 0
		duchy = 0
		kingdom = 0
		empire = 0
		hegemony = 0
	}
	
	selection_tooltip = decision_adopt_oligarchy_from_feudal_tooltip
	desc = decision_adopt_oligarchy_from_feudal_desc
	confirm_text = decision_adopt_oligarchy_from_feudal_confirm
	decision_group_type = major
	
	is_shown = {
		is_ai = no

		is_landed = yes
		is_human = yes # Should cover all above
		primary_title.tier >= tier_county
		highest_held_title_tier <= tier_duchy
		is_independent_ruler = yes
		OR = {
			government_has_flag = government_is_feudal 
			government_has_flag = government_is_clan
			government_has_flag = government_is_administrative
		}
		culture ?= { has_cultural_parameter = lotr_may_adopt_oligarchy }
	}
	
	is_valid = {
		OR = {
			government_has_flag = government_is_feudal 
			government_has_flag = government_is_clan
			government_has_flag = government_is_administrative
		}
		prestige_level >= 3
		# culture ?= { # Covered in is_shown by parameter check
		# 	OR = {
		# 		has_cultural_tradition = tradition_republican_legacy
		# 		has_cultural_tradition = tradition_tribe_unity
		# 		has_cultural_tradition = tradition_collective_lands
		# 		has_cultural_tradition = tradition_parochialism
		# 		has_cultural_tradition = tradition_city_keepers
		# 		# has_cultural_tradition = tradition_dorwinion_vineyards
		# 		has_cultural_tradition = tradition_bellakarean_cities
		# 	}
		# }
		custom_tooltip = { # You'll need to sway your vassals
			text = all_powerful_vassals_required_tt
			any_powerful_vassal = {
				count = all
				opinion = {
					target = root
					value >= 50
				}
			}
		}
	}

	cost = {
		prestige = 500
		gold = {  # Be ready to bride - Calc stolen from admin
			value = 800
			if = {
				limit = { is_ai = no }
				#Realm size
				if = {
					limit = { root.realm_size >= 300 }
					multiply = 3.5
				}
				else_if = {
					limit = { root.realm_size >= 250 }
					multiply = 3
				}
				else_if = {
					limit = { root.realm_size >= 200 }
					multiply = 2.5
				}
				else_if = {
					limit = { root.realm_size >= 150 }
					multiply = 2
				}
				else_if = {
					limit = { root.realm_size >= 100 }
					multiply = 1.5
				}
			}
			#Era
			culture = {
				if = {
					limit = { has_cultural_era_or_later = culture_era_late_medieval }
					multiply = 1.75
				}
				else_if = {
					limit = { has_cultural_era_or_later = culture_era_high_medieval }
					multiply = 1.25
				}
				else_if = {
					limit = { has_cultural_era_or_later = culture_era_early_medieval }
					multiply = 1
				}
				else = {
					multiply = 0.75
				}
			}
			#Income
			if = {
				limit = { root.monthly_character_income >= 100 }
				multiply = 3
			}
			else_if = {
				limit = { root.monthly_character_income >= 80 }
				multiply = 2.5
			}
			else_if = {
				limit = { root.monthly_character_income >= 60 }
				multiply = 2
			}
			else_if = {
				limit = { root.monthly_character_income >= 40 }
				multiply = 1.5
			}
			else_if = {
				limit = { root.monthly_character_income >= 20 }
				multiply = 1.1
			}

			max = 3000
		}
	}
	
	effect = {
		change_government = oligarchy_government
		custom_tooltip = {
			text = TITLES_GAIN_OLIGARCHIC_SUCCESSION
			root.primary_title = {
				add_title_law = oligarchic_elective_succession_law
			}
		}
		custom_tooltip = {
			text = ALL_VASSALS_BECOME_OLIGARCHIC
			every_vassal_or_below = {
				limit = {
					NOT = {
						government_has_flag = government_is_oligarchy
					}
				}
				change_government = oligarchy_government
				this.primary_title = {
					add_title_law = oligarchic_elective_succession_law
				}
			}
		}
	}
}

### Convert to Oligarchy (from Tribal/Wanua) ###
convert_to_oligarchy_decision = {
	title = convert_to_oligarchy_decision
	picture = {
		reference = "gfx/interface/illustrations/decisions/decision_council_03.dds"
	}
	decision_group_type = major
	desc = convert_to_oligarchy_decision_desc
	selection_tooltip = convert_to_oligarchy_decision_tooltip
	confirm_text = convert_to_oligarchy_decision_confirm

	ai_check_interval_by_tier = {
		barony = 0
		county = 0
		duchy = 36
		kingdom = 36
		empire = 36
		hegemony = 36
	}

	is_shown = {
		OR = {
		    government_has_flag = government_is_tribal
			government_has_flag = government_is_wanua
		}
		primary_title.tier >= tier_county
		highest_held_title_tier <= tier_duchy
		is_independent_ruler = yes
		OR = {
			is_independent_ruler = yes
			AND = {
				is_independent_ruler = no
				liege = {
					NOR = {
						government_has_flag = government_is_feudal
						government_has_flag = government_is_clan
						government_has_flag = government_is_elven
						government_has_flag = government_is_oligarchy
					}
				}
			}
		}
		OR = {
			culture ?= { has_cultural_parameter = lotr_may_adopt_oligarchy }
			government_has_flag = government_is_wanua 
		}
	}

	is_valid = {
		has_realm_law = tribal_authority_3
		custom_description = {
			text = faith_is_reformed
			subject = root
			faith = { NOT = { has_doctrine_parameter = unreformed } }
		}
		prestige_level >= 3
		# culture ?= { # Covered in is_shown by parameter check
		# 	OR = {
		# 		has_cultural_tradition = tradition_republican_legacy
		# 		has_cultural_tradition = tradition_tribe_unity
		# 		has_cultural_tradition = tradition_collective_lands
		# 		has_cultural_tradition = tradition_parochialism
		# 		has_cultural_tradition = tradition_city_keepers
		# 		# has_cultural_tradition = tradition_dorwinion_vineyards
		# 		has_cultural_tradition = tradition_bellakarean_cities
		# 	}
		# }
		capital_province.county = {
			development_level >= convert_to_feudalism_development_requirement
		}

		custom_description = { #You need to have 70% of all (non-regional/unique) tribal innovations!
			text = has_all_tribal_innovations
			subject = root.culture
			culture = {
				calc_true_if = {
					amount >= 9
					has_innovation = innovation_motte
					has_innovation = innovation_catapult
					has_innovation = innovation_barracks
					has_innovation = innovation_mustering_grounds
					has_innovation = innovation_bannus
					has_innovation = innovation_quilted_armor
					has_innovation = innovation_gavelkind
					has_innovation = innovation_crop_rotation
					has_innovation = innovation_city_planning
					has_innovation = innovation_casus_belli
					has_innovation = innovation_plenary_assemblies
					has_innovation = innovation_ledger
					has_innovation = innovation_development_01
					has_innovation = innovation_currency_01
				}
			}
		}
		trigger_if = { # Only show if needed
			limit = {
				culture = { has_cultural_parameter = permafrost_cannot_become_feudal }
			}
			custom_description = {
				text = permafrost_feudal_ban
				NOT = { culture = { has_cultural_parameter = permafrost_cannot_become_feudal } }
			}
		}
	}

	cost = {
		prestige = 500
		piety = 150
		gold = monumental_gold_value # Better be ready to bribe
	}

	effect = {
		gain_legitimizing_legend_seed_tooltip_effect = yes
		add_character_flag = ready_to_receive_seed
		trigger_event = lotr_general.0001
		save_scope_as = actor
		hidden_effect = { convert_whole_realm_to_feudalism_effect = yes }
		change_government = oligarchy_government
		custom_tooltip = {
			text = TITLES_GAIN_OLIGARCHIC_SUCCESSION
			root.primary_title = {
				add_title_law = oligarchic_elective_succession_law
			}
		}
		custom_tooltip = {
			text = ALL_VASSALS_BECOME_OLIGARCHIC
			every_vassal_or_below = {
				limit = {
					NOT = {
						government_has_flag = government_is_oligarchy
					}
				}
				change_government = oligarchy_government
				this.primary_title = {
					add_title_law = oligarchic_elective_succession_law
				}
			}
		}
		hidden_effect = {
			every_vassal = {
				trigger_event = {
					id = lotr_general.0002
					days = 3
				}
			}
		}
	}

	ai_potential = {
		always = yes
	}

	ai_will_do = {
		base = 50
	}
}

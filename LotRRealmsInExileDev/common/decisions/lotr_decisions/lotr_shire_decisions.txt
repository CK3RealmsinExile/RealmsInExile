	selection_tooltip = decision_seize_bag_end_tooltip
	picture = {
		reference = "gfx/interface/illustrations/decisions/decision_seize_bag_end.dds"
	}
	sort_order = 2
	

	# Shown
	decision_group_type = realms_in_exile_major
	
	is_shown = {
		any_realm_county = {
			title_province = {
 					geographical_region = middleearth_west_arnor_arthedain

 			}
 		}
		faith_is_evil = yes
		title:c_hobbiton = { 
			title_province ?= { has_building = wonder_bag_end_03 }
			}
	}

	# Requirements
	is_valid = {
		faith_is_evil = yes
		is_landed = yes
		completely_controls = title:c_hobbiton
		title:c_hobbiton = { 
			title_province ?= { has_building = wonder_bag_end_03 }
		}
		OR = {
			is_hobbit = yes
			is_dwarf = yes
			is_orc = yes
		}
	}

	is_valid_showing_failures_only = {
		is_imprisoned = no
		is_lotr_adult = yes
		is_at_war = no
	}

	# Effect
	effect = {
		add_piety = massive_piety_gain
		dynasty = {
			add_dynasty_prestige = medium_dynasty_prestige_value
		}
		title:c_hobbiton = { 
			title_province = {
				remove_building = wonder_bag_end_03
				add_special_building = wonder_bag_end_02
			}
		}
		
		trigger_event = hobbit.0102

	}

	# Cost
	cost = {
		prestige = 200
		gold = 500
	}

	# AI
	ai_goal = yes

	ai_potential = {
		always = yes
	}

	ai_will_do = {
		base = 1000
	}
}

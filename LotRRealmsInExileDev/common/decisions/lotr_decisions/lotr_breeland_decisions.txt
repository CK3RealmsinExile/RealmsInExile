decision_restore_arne = { # Restore the Kingdom of Arne - for custom character runs
	title = decision_restore_arne
	picture = {
		reference = "gfx/interface/illustrations/decisions/decision_breeland.dds"
	}
	
	ai_goal = yes
	selection_tooltip = decision_restore_arne_tooltip
	desc = decision_restore_arne_desc
	confirm_text = decision_restore_arne_confirm
	
	decision_group_type = realms_in_exile_major
	
	is_shown = {	
		culture = culture:eriadorim
        # culture = { any_parent_culture_or_above = { this = culture:eriadorim } } 
		is_landed = yes
		# NOT = { exists = title:k_arne.holder }
		NOT = {
			is_target_in_global_variable_list = {
				name = unavailable_unique_decisions
				target = flag:decision_restore_arne
			}
		}
	}

	is_valid = {
		highest_held_title_tier <= tier_kingdom
		prestige_level >= 4
		completely_controls_region = custom_arne
		any_county_in_region = {
			region = custom_arne
			percent >= 0.5
			culture = { any_parent_culture_or_above = { this = culture:eriadorim } } 
		}
	}

	is_valid_showing_failures_only = {
		is_imprisoned = no
		is_available_adult = yes
		is_at_war = no
		is_independent_ruler = yes
	}
	
	cost = {
		gold = 200
		prestige = 1000
	}

	effect = {
		add_to_global_variable_list = {
			name = unavailable_unique_decisions
			target = flag:decision_restore_arne
		}
		get_title = title:k_arne
		# Creates de-jure Arne kingdom
		title:d_emyn_sul = {
			set_de_jure_liege_title = title:k_arne
		}
		title:d_nan_turath = {
			set_de_jure_liege_title = title:k_arne
		}
		title:d_tyrn_gorthad = {
			set_de_jure_liege_title = title:k_arne
		}
		set_realm_capital = title:c_arhud
		title:c_arhud = {
			change_development_level = 3
		}
		# Expand a bit when the Eradorrim are fleshed out
	}

	ai_potential = {
		always = yes
	}

	ai_will_do = {
		base = 100
	}
}

						culture = { has_cultural_pillar = heritage_southern_apysaic }

				}
			}
		}

			limit = { 
				OR = {
					government_has_flag = government_is_clan
					government_has_flag = government_is_feudal
				}
			}
			root.culture = {
				get_all_innovations_from = culture:bozishnarud
			}
			add_prestige_level = 2
			root.dynasty = {
				add_dynasty_prestige = 500
			}
			give_nickname = nick_honnin_restorer
			add_realm_law_skip_effects = crown_authority_3
			add_piety = 1000
			add_prestige = 1000
			capital_county = {
				change_development_progress_with_overflow = 750
			}
			hidden_effect = {
				every_realm_province = {
					limit = {
						culture = { has_cultural_pillar = heritage_southern_apysaic }
					}
					add_province_modifier = { modifier = honnin_resettlements years = 100 }
				}
			}
		}

			name = unavailable_unique_decisions
			target = flag:reclaim_ray_decision
		}

decision_restore_druedain = {
	sort_order = 35
	picture = {
		reference = "gfx/interface/illustrations/decisions/decision_druadan_02.dds"
	}
	title = decision_restore_druedain
	selection_tooltip = decision_restore_druedain_tooltip
	desc = decision_restore_druedain_desc
	confirm_text = decision_restore_druedain_confirm

	decision_group_type = realms_in_exile_minor

	# SHOWN - The decision appears if conditions are met
	is_shown = {
		is_landed = yes

		# Druedain will not return under a servant of darkness
		faith_is_evil = no

		# Must hold or have vassals hold the Kingdom of Anorien or the Stewardship of Gondor
		OR = {
			has_title = title:k_anorien
			has_title = title:e_gondor_steward
			any_vassal = {
				OR = {
					has_title = title:k_anorien
					has_title = title:e_gondor_steward
				}
			}
		}

		# Must hold or have vassals hold the Druadan Forest duchy counties
		OR = {
			has_title = title:d_druadan_forest
			any_vassal = {
				has_title = title:d_druadan_forest
			}
			AND = {
				OR = {
					has_title = title:c_dru_bhuta
					any_vassal = { has_title = title:c_dru_bhuta }
				}
				OR = {
					has_title = title:c_dhurghud
					any_vassal = { has_title = title:c_dhurghud }
				}
			}
		}

		# The Druedain must no longer live there (culture is not druedain in the counties)
		title:c_dru_bhuta = {
			NOT = { culture = culture:druedain }
		}

		# Must not have already restored the Druedain
		NOT = {
			is_target_in_global_variable_list = {
				name = unavailable_unique_decisions
				target = flag:restore_druedain
			}
		}
	}

	# IS_VALID - Hard requirements
	is_valid = {
		# Must have decent prestige
		prestige_level >= 3

		# Must not be at war
		is_at_war = no

		# Must not be a tyrant - the Druedain must trust you
		custom_tooltip = {
			text = restore_druedain_low_tyranny_requirement
			tyranny < 20
		}

		# Must have a friendly Druedain ruler
		custom_tooltip = {
			text = restore_druedain_need_druedain_friend
			any_relation = {
				type = friend
				culture = culture:druedain
				is_landed = yes
			}
		}
	}

	is_valid_showing_failures_only = {
		is_imprisoned = no
		is_available_adult = yes
	}

	# COST
	cost = {
		gold = 200
		prestige = 500
		piety = 300
	}

	# EFFECT
	effect = {
		add_legitimacy = major_legitimacy_gain
		custom_tooltip = decision_restore_druedain_effect_tooltip
		add_to_global_variable_list = {
			name = unavailable_unique_decisions
			target = flag:restore_druedain
		}
		# Save the befriended Druedain ruler so the event can use them
		random_relation = {
			type = friend
			limit = {
				culture = culture:druedain
				is_landed = yes
			}
			save_scope_as = druedain_contact
		}
		if = {
			limit = { exists = scope:druedain_contact }
			set_variable = {
				name = druedain_contact
				value = scope:druedain_contact
			}
		}
		trigger_event = {
			id = druedain.1000
			days = 5
		}
	}

	# AI
	ai_goal = yes
	ai_potential = {
		always = yes
	}
	ai_will_do = {
		base = 100
	}
}

decision_destroy_angmar = {
	title = decision_destroy_angmar
	desc = decision_destroy_angmar_desc
	confirm_text = decision_destroy_angmar_confirm
	selection_tooltip = decision_destroy_angmar_tooltip
	picture = {
		reference = "gfx/interface/illustrations/decisions/decision_angmar_03.dds"
	}
	decision_group_type = angmar_wars_situation

	# AI
	ai_goal = yes
	ai_potential = {
		always = yes
	}
	ai_will_do = {
		base = 0
		modifier = {
			military_power >= title:k_angmar.holder.military_power.mult3.div2 # 1.5 Angmar's strength
		}
	}

	cooldown = { years = 10 }

	is_shown = {
		is_arnorian_ruler = yes
		title:k_angmar.holder ?= {
			faith_is_evil = yes
		}
		is_landed = yes
	}

	# IS VALID
	is_valid = {
		prestige_level >= very_high_prestige_level

		# Independent Arnorian king
		highest_held_title_tier >= tier_kingdom
		is_independent_ruler = yes

		is_arnor_ascendant = yes

		custom_tooltip = {
			text = angmar_holds_no_land_in_dejure_arnor
			NOT = {
				title:e_arnor = {
					any_de_jure_county = {
						holder.top_liege = title:k_angmar.holder
					}
				}
			}
		}

		title:k_angmar.holder = {
			NOT = { realm_size >= 30 }
		}
	}

	is_valid_showing_failures_only = {
		is_imprisoned = no
		is_available_adult = yes
		is_at_war = no
		gold >= 0
		NOT = { has_truce = title:k_angmar.holder } 
	}

	cost = {
		prestige = 200
		piety = 250
	}

	# EFFECT
	effect = {
		start_war = {
			casus_belli = final_battle_gondor_cb # Replace with relevant later
			target = title:k_angmar.holder
			target_title = title:k_angmar
		}
	}
}

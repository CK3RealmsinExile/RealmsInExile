decision_utter_south_mumakan = {
	title = {
		first_valid = {
 			triggered_desc = {
 				trigger = { has_title = title:d_tamska_main }
				desc = decision_utter_south_mumakan.tamska
 			}
			desc = decision_utter_south_mumakan
		}
	}
	desc = {
 		first_valid = {
 			triggered_desc = {
 				trigger = { has_title = title:d_tamska_main }
				desc = decision_utter_south_mumakan_desc.tamska
 			}
 			desc = decision_utter_south_mumakan_desc
		}
 	}
	
	confirm_text = decision_utter_south_mumakan_confirm
	selection_tooltip = decision_utter_south_mumakan_tooltip
	picture = {
		trigger = {
			has_title = title:d_tamska_main
		}
		reference = "gfx/interface/illustrations/decisions/decision_mag_tumag_02.dds"
	}
	picture = {
		reference = "gfx/interface/illustrations/decisions/decision_mumakan_01.dds"
	}

    # AI
	ai_goal = yes
	ai_potential = {
		always = yes
	}
	ai_will_do = {
		base = 100
	}
	sort_order = 1000

    # IS SHOWN
	decision_group_type = realms_in_exile_major

    is_shown = {
        OR = {
            has_title = title:e_mumakan
            has_title = title:e_shayn
            has_title = title:e_usakan
			has_title = title:e_mag_tumag
        }

        OR = {
            culture = { has_cultural_parameter = heritage_group_mumakani }
            this = character:nazgul4
        }

        NOT = {
			is_target_in_global_variable_list = {
				name = unavailable_unique_decisions
				target = flag:unify_utter_south
			}
			exists = title:h_utter_south.holder
		}
    }

    # IS VALID
	is_valid = {
		prestige_level >= high_prestige_level
		trigger_if = {
			limit = { has_title = title:d_tamska_main }
			has_title = title:e_mag_tumag
		}
		trigger_else = {
			has_title = title:e_shayn 
		}
        has_title = title:e_mumakan
        has_title = title:e_usakan
	}

	is_valid_showing_failures_only = {
		is_imprisoned = no
		is_available_adult = yes
		is_at_war = no
	}

    cost = {
        gold = 1000
		prestige = 5000
	}

    # EFFECT
	effect = {
        #Create the title and make it primary
		create_title_and_vassal_change = {
			type = created
			save_scope_as = title_change
			add_claim_on_loss = yes
		}
		title:h_utter_south = {
			change_title_holder = {
				holder = root
				change = scope:title_change
			}
		}
		resolve_title_and_vassal_change = scope:title_change
		set_primary_title_to = title:h_utter_south

        title:e_mumakan = {
			set_de_jure_liege_title = title:h_utter_south
		}
		if = { # Shayn
			limit = {
				has_title = title:e_shayn
				NOT = { has_title = title:d_tamska_main }
			}
			title:e_shayn = { set_de_jure_liege_title = title:h_utter_south }
		}
		title:e_usakan = {
			set_de_jure_liege_title = title:h_utter_south
		}
		if = { # Mag and Tumag
			limit = {
				has_title = title:e_mag_tumag
				has_title = title:d_tamska_main
			}
			title:e_mag_tumag = { set_de_jure_liege_title = title:h_utter_south }
		}

        root.dynasty = { add_dynasty_prestige = 1000 }

        add_to_global_variable_list = {
			name = unavailable_unique_decisions
			target = flag:unify_utter_south
		}
		
		trigger_event = {
			id = mumakan.0200
			days = 1
		}
    }
}

		hegemony = 0
	}
	selection_tooltip = decision_nafarati_confederation_tooltip
	desc = decision_nafarati_confederation_desc
	confirm_text = decision_nafarati_confederation_confirm

	sort_order = 200

	is_shown = {
		OR = {
			culture =  culture:nafarati
			culture = { any_parent_culture_or_above = { this = culture:nafarati } }
		}
		capital_province = { # Be located at least somewhere close to the Nafarati heartlands
			OR = {
				geographical_region = middleearth_uttersouth_mumakan 
				geographical_region = middleearth_uttersouth_usakan 
				geographical_region = middleearth_uttersouth_shayn
			}
		}
		NOT = { exists = title:k_nafarati.holder }
	}
	is_valid = {
		highest_held_title_tier <= tier_kingdom
		prestige_level >= 4
		legitimacy_level >= 3

		custom_tooltip = {
			text = you_are_the_culture_head
			culture.culture_head ?= root
		}
		
		custom_tooltip = {
			text = own_all_culture_counties
			culture = {
				any_culture_county = {
					percent = 0.8
					holder = {
						any_liege_or_above = {
							this = root
						}
					}
				}
			}
		}

		trigger_if = { #Your liege (and other relevant overlords) needs to like you
			limit = {
				liege != root
			}
			OR = {
				liege = {
					opinion = {
						target = root
						value >= 80
					}
				}
				has_hook = liege
			}
		}
		trigger_if = {
			limit = {
				top_liege != liege
			}
			OR = {
				top_liege = {
					opinion = {
						target = root
						value >= 80
					}
				}
				has_hook = top_liege
			}
		}
		trigger_if = {
			limit = {
				top_overlord != root.top_liege
			}
			OR = {
				top_overlord = {
					opinion = {
						target = root
						value >= 80
					}
				}
				has_hook = top_overlord
			}
		}

	}
	
	is_valid_showing_failures_only = {
		is_at_war = no 
		is_available_adult = yes 
		is_travelling = no
		is_in_debt = no
	}

	cost = {
		prestige = 800
		gold = 100
	}

	effect = {
		# Get Kingdom
		get_title = title:k_nafarati
		set_primary_title_to = title:k_nafarati
		hidden_effect = {
			title:k_nafarati = {
				lotr_move_dejure_capital_effect = { COUNTY = root.capital_county }
				set_de_jure_liege_title = root.capital_county.empire
			}
		}

		add_legitimacy = 500
		root.dynasty = { add_dynasty_prestige = 500 }

		custom_tooltip = nafarati_war_host_tt
		spawn_army = {
			name = nafarati_war_host
			inheritable = no
			uses_supply = no
			levies = 2000
			men_at_arms = {
				type = nafarati_marauders
				stacks = 4
			}
			men_at_arms = {
				type = mahud_headhunters
				stacks = 4
			}
			men_at_arms = {
				type = mubadar_berserkers
				stacks = 4
			}
			location = capital_province
		}

		# Mubadar faith easter egg
		if = {
			limit = {
				faith = faith:faith_mubadar
			}
			hidden_effect = {
				every_sub_realm_county = {
					limit = {
						faith != faith:faith_mubadar
					}
					random = {
						chance = 50
						set_county_faith = faith:faith_mubadar
					}
				}
			}
			custom_tooltip = convert_counties_to_own_faith
		}
	}
}

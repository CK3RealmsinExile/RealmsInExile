decision_destroy_thaurband = {
    title = decision_destroy_thaurband
    picture = {
        reference = "gfx/interface/illustrations/decisions/decision_rhun_1.dds"
    }

    ai_goal = yes
    ai_will_do = {
		base = 100
	}

    selection_tooltip = decision_destroy_thaurband_tooltip
    desc = decision_destroy_thaurband_desc
    confirm_text = decision_destroy_thaurband_confirm

    cost = {
        prestige = 500
    }

    decision_group_type = realms_in_exile_major
	
	is_shown = {
        has_title = title:c_thaurband
        OR = { 
            AND = {
            religion = { is_in_family = rf_eruhini }
            top_liege.religion = { is_in_family = rf_eruhini }
            }
            AND = {
                faith = { has_doctrine = doctrine_echoes_of_the_ainur }
                top_liege.faith = { has_doctrine = doctrine_echoes_of_the_ainur }
            }
        }
        title:c_thaurband.title_province = {
            OR = { 
                has_building = wonder_thaurband_01
                has_building = wonder_thaurband_02
                has_building = wonder_thaurband_03
            }
        }
	}

    effect = { 
        title:c_thaurband = {
			set_county_culture = culture:nurnoth
			set_county_faith = faith:faith_nurn_good
		}
        if = {
			limit = {
				title:c_thaurband.title_province = {
					has_building = wonder_thaurband_01
				}
			}
			title:c_thaurband.title_province = { remove_building = wonder_thaurband_01 }
		}
		else_if = {
			limit = {
				title:c_thaurband.title_province = {
					has_building = wonder_thaurband_02
				}
			}
			title:c_thaurband.title_province = { remove_building = wonder_thaurband_02 }
		}
		else_if = {
			limit = {
				title:c_thaurband.title_province = {
					has_building = wonder_thaurband_03
				}
			}
			title:c_thaurband.title_province = { remove_building = wonder_thaurband_03 }
		}
    }
}

decision_zantaka_marriage_offer = { # Make the marriage offer
	title = decision_zantaka_marriage_offer
	picture = {
        reference = "gfx/interface/illustrations/decisions/fp3/friendly.dds"
    }
	ai_goal = yes
	selection_tooltip = decision_zantaka_marriage_offer_tooltip
	desc = decision_zantaka_marriage_offer_desc
	confirm_text = decision_zantaka_marriage_offer_confirm
	decision_group_type = zantaka_story_decisions
	ai_will_do = {
		base = 100
	}
	sort_order = 80

	is_shown = {
        has_character_flag = zantaka_marriage_route
		NOT = {
			is_target_in_global_variable_list = {
				name = unavailable_unique_decisions
				target = flag:zantaka_marriage_decision
			}
		}
		title:c_emelcyn.holder = {
			dynasty = dynasty:dynasty_pharala
		}
		title:c_emelcyn.holder.player_heir = {
			is_female = no
			is_married = no
			is_landed = no
			is_ai = yes
		}
	}

	is_valid = {
		OR = {
			prestige_level >= 1
        	legitimacy_level >= 1
		}
	}

	cost = {
		prestige = 500
	}

	effect= {
		custom_tooltip = decision_zantaka_marriage_offer_effect_tt
		hidden_effect = { #give free caravan master
			if = {
				limit = { NOT = { employs_court_position = travel_leader_court_position } }
				create_character = {
					template = soldier_friend_character
					employer = root
					gender_female_chance = 25
					save_scope_as = caravan_master
				}
				appoint_court_position = {
					recipient = scope:caravan_master
					court_position = travel_leader_court_position
				}
			}
		}
		start_travel_plan = { #Travel to Emelkyn
			destination = title:c_emelcyn.title_province
			on_arrival_event = zantaka_event_chain.0107
			return_trip = yes
			players_use_planner = no
			on_arrival_destinations = all
		}
		add_to_global_variable_list = {
			name = unavailable_unique_decisions
			target = flag:zantaka_marriage_decision
		}	        
	}

	is_valid_showing_failures_only = {
		is_imprisoned = no
		is_available_adult = yes
		is_at_war = no
	}
}

				type = swear_fealty

			}
			change_liege = {
				liege = root
				change = scope:change
			}
			resolve_title_and_vassal_change = scope:change
		}

decision_zantaka_end_oligarchy = { # End Oligarchy once and for all
	title = decision_zantaka_end_oligarchy
	picture = {
        reference = "gfx/interface/illustrations/decisions/decision_mumakan_04.dds"
    }
	ai_goal = yes
	selection_tooltip = decision_zantaka_end_oligarchy_tooltip
	desc = decision_zantaka_end_oligarchy_desc
	confirm_text = decision_zantaka_end_oligarchy_confirm
	decision_group_type = zantaka_story_decisions
	ai_will_do = {
		base = 100
	}
	sort_order = 110

	is_shown = {
		global_var:zantaka_revolution = yes
		dynasty = dynasty:dynasty_zantaka	
		trigger_if = {
			has_dlc_feature = roads_to_power			
		}
		NOT = {
			is_target_in_global_variable_list = {
				name = unavailable_unique_decisions
				target = flag:zantaka_end_oligarchy
			}
		}
	}

	is_valid = {
		has_title = title:k_lyneria
		prestige_level >= 4
		legitimacy_level >= 3
		trigger_if = {
			limit = {
				top_liege = this
				is_ai = no
			}
			custom_tooltip = {
				text = all_powerful_vassals_required_tt
				any_powerful_vassal = {
					count = all
					opinion = {
						target = root
						value >= 50
					}
				}
			}
		}		
	}

	cost = {
		prestige = 1000
		gold = 750
	}

	effect= {
		#custom_tooltip = decision_zantaka_end_oligarchy_effect_tt	
		trigger_event = zantaka_event_chain.207
		change_government = administrative_government
		title:k_lyneria = {
			if = {
				limit = {
					has_title_law = confederation_elective_succession_law
				}
			}
			remove_title_law = confederation_elective_succession_law			
		}
		if = {
			limit = { NOT = { any_held_title = { is_noble_family_title = yes } } }
			create_noble_family_effect = { GOVERNMENT_GIVER = this }
		}
		tgp_domicile_conversion_when_changing_government_type = {
			NEW_DOMICILE_TYPE = estate
			NEW_GOVERNMENT_TYPE = administrative_government
		}
		hidden_effect = {
			every_vassal_or_below = {
				change_government = administrative_government
				if = {
					limit = { NOT = { any_held_title = { is_noble_family_title = yes } } }
					create_noble_family_effect = { GOVERNMENT_GIVER = this }
				}
				tgp_domicile_conversion_when_changing_government_type = {
					NEW_DOMICILE_TYPE = estate
					NEW_GOVERNMENT_TYPE = administrative_government
				}				
			}				
		}
		add_to_global_variable_list = {
			name = unavailable_unique_decisions
			target = flag:zantaka_end_oligarchy
		}		
	}

	is_valid_showing_failures_only = {
		is_imprisoned = no
		is_available_adult = yes
		is_at_war = no
	}
}

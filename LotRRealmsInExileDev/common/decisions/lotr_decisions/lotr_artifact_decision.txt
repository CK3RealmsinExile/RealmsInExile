make_heirloom_decision = {
	picture = {
		reference = "gfx/interface/illustrations/decisions/decision_dynasty_house.dds"
	}

	decision_group_type = realms_in_exile_minor

	desc = make_heirloom_decision_desc
	
	confirm_text = make_heirloom_decision_confirm

	selection_tooltip = make_heirloom_decision_tooltip

	is_shown = {
		cannot_be_killed = no
		highest_held_title_tier >= tier_kingdom
		player_heir ?= { 
			is_lotr_adult = yes
			house = root.house
		}
		NOT = { #One heirloom per title
			any_artifact = {
				OR = {
					AND = {
						has_variable = heirloom_of_title
						var:heirloom_of_title = root.primary_title
					}
					AND = {
						has_variable = artifact_succession_title
						var:artifact_succession_title = root.primary_title
					}
				}
			}
		}
	}

	is_valid_showing_failures_only = {
		any_character_artifact = {
			OR = { # Cool enough
				rarity = famed
				rarity = illustrious
			}
			ep1_artifact_is_court_artifact_trigger = no # Personal artifacts only
			OR = { #Available to be attributed
				NOT = { has_variable = heirloom_of_title }
				NOT = { has_variable = artifact_succession_title }
				AND = { #To change the title of an heirloom
					has_variable = heirloom_of_title
					var:heirloom_of_title.holder = root
				}
				AND = { #To change the title of an heirloom
					has_variable = artifact_succession_title
					var:artifact_succession_title.holder = root
				}
			}
			NOR = {
				has_variable = ring_of_power
				has_variable = silmaril
			}
		}
	}

	effect = {
		trigger_event = lotr_artifact.0014
	}

	ai_check_interval = 60
	
	ai_potential = {		
		always = no
	}
}

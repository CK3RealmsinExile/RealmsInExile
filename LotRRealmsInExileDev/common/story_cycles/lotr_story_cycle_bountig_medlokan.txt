story_cycle_bountig_medlokan = {
	
	on_setup = {
		title:k_south_rhovanion = { # Start a war between a Ioriag duke and Rohan
			random_de_jure_county_holder = {
				limit = {
					is_wastelands = no
					culture = { has_cultural_pillar = heritage_ioriag }
					highest_held_title_tier > tier_county
				}
				start_war = {
					cb = raid_for_captives_cb
					target = title:k_rohan.holder
				}
				save_scope_as = rhun_attacker
			}
		}
		# scope:rhun_attacker = {
		# 	create_task_contract = {
		# 		task_contract_type = lotr_story
		# 		location = location
		# 		task_contract_employer = location.county.holder
		# 		task_contract_tier = 4
		# 	}
		# }
		story_owner = { 
			create_task_contract = {
				task_contract_type = lotr_story
				location = scope:rhun_attacker.location
				task_contract_employer = scope:rhun_attacker.location.county.holder
				task_contract_tier = 4
			}
		}
			
		set_variable = {
			name = rhun_attacker
			value = scope:rhun_attacker
		}
	}

	on_end = {
		# debug_log = "Harrying of the North ended on:"
		# debug_log_date = yes
	}

	on_owner_death = {
	}

	#End the Story Cycle if there are no longer any Anglo-Saxon lords left, or if Normans no longer rule England
	#Put it on a long timer so William player can't just kick them all out and weather the storm for a month
	effect_group = {
		months = 4
		triggered_effect = {
			trigger = {
				NOT = { story_owner = { has_variable = started_bountig_story } }
			}
			effect = {
				story_owner = { set_variable = started_bountig_story }

				var:rhun_attacker = {
					random_character_war = {
						limit = { using_cb = raid_for_captives_cb }
						save_scope_as = war
					}
				}
				var:rhun_attacker = {
					create_task_contract = {
						task_contract_employer = var:rhun_attacker
						task_contract_type = laamp_join_war_contract
						task_contract_tier = 4
						location = var:rhun_attacker.capital_county.title_province
						save_scope_as = new_contract_defender
					}
				}
				scope:new_contract_defender = {
					set_variable = {
						name = retain_despite_distance
						value = yes
					}
					set_variable = {
						name = task_contract_war
						value = scope:war
					}
					set_variable = {
						name = task_contract_employer
						value = title:k_rohan.holder
					}
					set_variable = {
						name = task_contract_target
						value = scope:war.primary_attacker
					}
				}
			}
		}
	}
}

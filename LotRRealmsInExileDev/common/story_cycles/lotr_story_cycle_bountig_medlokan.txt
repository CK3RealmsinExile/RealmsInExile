story_cycle_bountig_medlokan = {
	
	on_setup = {
		title:k_south_rhovanion = { # Start a war between a Ioriag duke and Rohan
			random_de_jure_county_holder = {
				limit = {
					is_wastelands = no
					culture = { has_cultural_pillar = heritage_ioriag }
					highest_held_title_tier = tier_duchy
				}
				save_scope_as = rhun_attacker
				set_variable = bountig_start_story
			}
		}
		story_owner = { 
			create_task_contract = {
				task_contract_type = lotr_story
				location = scope:rhun_attacker.location
				task_contract_employer = scope:rhun_attacker.location.county.holder
				task_contract_tier = 1
			}
			pan_camera_to_province = scope:rhun_attacker.location
		}
	}

	on_end = {
	}

	on_owner_death = {
		end_story = yes
	}

	effect_group = { # Check every 3 months if contract is still valid
		months = 3
		triggered_effect = {
			trigger = {
				NOT = { story_owner = { has_variable = started_bountig_story } }
				NOT = { any_task_contract = { has_task_contract_type = lotr_story } }
			}
			effect = {
				title:k_south_rhovanion = { # Start a war between a Ioriag duke and Rohan
					random_de_jure_county_holder = {
						limit = {
							is_wastelands = no
							culture = { has_cultural_pillar = heritage_ioriag }
							highest_held_title_tier = tier_duchy
							any_sub_realm_county = { any_neighboring_county = { culture = culture:wastelands } }
						}
						save_scope_as = rhun_attacker
						set_variable = bountig_start_story
					}
				}
				story_owner = { 
					create_task_contract = {
						task_contract_type = lotr_story
						location = scope:rhun_attacker.location
						task_contract_employer = scope:rhun_attacker.location.county.holder
						task_contract_tier = 1
						destination = scope:rhun_attacker.location
						save_scope_as = task_contract
					}
					scope:task_contract = { set_variable = bountig_story }
					trigger_event = bountig_event_chain.0001 # Foce camera to pan to province, in-case of intercepting event
				}
			}
		}
	}

	effect_group = {
		days = 1
		triggered_effect = {
			trigger = {
				story_owner = { has_variable = bountig_story_visit_minas_tirith }
				NOT = { any_task_contract = { has_task_contract_type = lotr_story } }
			}
			effect = {
				story_owner = { save_scope_as = bountig }
				title:c_minas_tirith.title_province = { save_scope_as = main_province }
				scope:bountig = { 
					create_task_contract = {
						task_contract_type = lotr_story
						location = scope:main_province
						task_contract_employer = scope:main_province.county.holder
						task_contract_tier = 1
						save_scope_as = task_contract
					}
					trigger_event = bountig_event_chain.0002 # Foce camera to pan to province, in-case of intercepting event
				}
				scope:bountig = { remove_variable = bountig_story_visit_minas_tirith }
			}
		}
	}
}

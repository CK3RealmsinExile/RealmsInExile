story_gondor_succession = {

	on_setup = {
	}

	on_end = {
		
	}

	on_owner_death = {
		if = {
			limit = {
				story_owner.player_heir = {
					NOT = {
						is_close_or_extended_family_of = story_owner
					}
				}
			}
			make_story_owner ?= title:d_gondor_steward.holder
			trigger_event = {&
				id = gondor.7000
			}
		}
		else = {
			make_story_owner = story_owner.player_heir
		}
	}
}

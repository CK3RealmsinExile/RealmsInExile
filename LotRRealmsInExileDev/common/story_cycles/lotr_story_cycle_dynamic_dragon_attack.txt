lotr_dynamic_dragon_attack_story = {
	on_setup = {
		set_variable = {
			name = capital
			value = story_owner.capital_county
		}
	}

	on_owner_death = {
		if = {
			limit = {
				exists = story_owner.primary_heir
				story_owner.primary_heir = { is_alive = yes }
			}

			if = {
				limit = { var:capital = { exists = var:prepped_1 } }
				story_owner.primary_heir = { add_character_flag = prep_time_1 }
			}
			if = {
				limit = { var:capital = { exists = var:prepped_2 } }
				story_owner.primary_heir = { add_character_flag = prep_time_2 }
			}
			if = {
				limit = { var:capital = { exists = var:prepped_3 } }
				story_owner.primary_heir = { add_character_flag = prep_time_3 }
			}

			# Inherits Dragons
			make_story_owner = story_owner.primary_heir
		}
	}

	on_end = {
		story_owner = {
			remove_character_flag = prep_time_1
			remove_character_flag = prep_time_2
			remove_character_flag = prep_time_3
		}
		var:capital = {
			remove_variable = prepped_1
			remove_variable = prepped_2
			remove_variable = prepped_3
		}
	}

	effect_group = {
		days = { 9 18 }
		triggered_effect = {
			trigger = {
				story_owner = {
					NOT = {
						has_character_flag = prep_time_1
						has_character_flag = prep_time_2
						has_character_flag = prep_time_3
					}
				}
			}
			effect = {
				story_owner = {
					trigger_event = lotr_dynamic_dragon_attack.0100
				}
			}
		}
		triggered_effect = {
			trigger = {
				story_owner = {
					has_character_flag = prep_time_1
					NOT = {
						has_character_flag = prep_time_2
						has_character_flag = prep_time_3
					}
				}
			}
			effect = {
				story_owner = {
					trigger_event = lotr_dynamic_dragon_attack.0101
				}
			}
		}
		triggered_effect = {
			trigger = {
				story_owner = {
					has_character_flag = prep_time_1
					has_character_flag = prep_time_2
					NOT = {
						has_character_flag = prep_time_3
					}
				}
			}
			effect = {
				story_owner = {
					trigger_event = lotr_dynamic_dragon_attack.0102
				}
			}
		}
		triggered_effect = {
			trigger = {
				story_owner = {
					has_character_flag = prep_time_1
					has_character_flag = prep_time_2
					has_character_flag = prep_time_3
				}
			}
			effect = {
				story_owner = {
					trigger_event = lotr_dynamic_dragon_attack.0110 # The Dragon is here
				}
			}
		}
	}
}

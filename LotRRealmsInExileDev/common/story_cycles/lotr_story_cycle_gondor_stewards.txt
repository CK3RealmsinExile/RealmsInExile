story_gondor_stewards = {

	on_setup = {
		set_global_variable = {
			name = lotr_stewards_power_level
			value = 1
		}
	}

	on_end = {
		remove_global_variable = {
			name = lotr_stewards_power_level
		}
	}

	on_owner_death = {
		make_story_owner = story_owner.player_heir
		if = { # If the Stewardship is hereditary, make sure people know that
			limit = {
				global_var:lotr_stewards_power_level = {
					OR = {
						value = 4
						value = 5
					}
				}
			}
			story_owner.player_heir = {
				add_character_flag = flag:rightful_steward_of_gondor
			}
		}
	}

	effect_group = {
		years = 1
		if = {
			limit = {
					is_governor = yes
					NOT = { has_trait = governor }
				}
				add_trait = governor
		}
		else = {
			add_trait_xp = {
				trait = governor
				value = {
					add = 2
				}
			}
		}
	}
}

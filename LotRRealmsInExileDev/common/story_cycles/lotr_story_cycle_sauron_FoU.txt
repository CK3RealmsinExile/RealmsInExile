story_cycle_sauron_FoU = {
	
	on_setup = {
		story_owner = { #End story if Sauron is a player
			if = {
				limit = { 
					is_ai = no
				}
				root = { end_story = yes }
			}
		}
	}

	on_end = { }

	on_owner_death = { end_story = yes } #Check for convinced modifier

	effect_group = { #Set the initial contract in Gondor/Harwan/Umbar
		months = 6
		
		triggered_effect = {  #Set the initial contract in Umbar
			trigger = { story_owner = { has_character_modifier = sauron_strengthened_3_modifier is_ai = yes } }
			effect = {
				story_owner = {
					every_maa_regiment = {
						limit = {
							OR = {
								is_maa_type = onager
								is_maa_type = trebuchet
								is_maa_type = mangonel
							}
						}
						destroy_maa_regiment = yes
					}

					every_maa_regiment = {
						limit = {
							OR = {
								is_maa_type = olog_hai
								is_maa_type = caragor_pack
								is_maa_type = siege_tower
								is_maa_type = mountainorc_warriors
							}
							can_upgrade_maa = yes
						}
						change_maa_regiment_size = {
							size = 1
							reinforce = yes
						}
					}
				}
			}
		}
	}
}

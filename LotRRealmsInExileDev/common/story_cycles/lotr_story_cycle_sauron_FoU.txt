story_cycle_sauron_FoU = {
	
	on_setup = {
		story_owner = { #End story if Sauron is a player
			if = {
				limit = { 
					is_ai = no
				}
				root = { end_story = yes }
			}
		}
	}

	on_end = { }

	on_owner_death = { end_story = yes }

	effect_group = {
		months = 6
		
		triggered_effect = {
			trigger = { story_owner = { has_character_modifier = sauron_strengthened_4_modifier is_ai = yes } }
			effect = {
				story_owner = {
					every_maa_regiment = {
						limit = {
							OR = {
								is_maa_type = onager
								is_maa_type = trebuchet
								is_maa_type = mangonel
							}
						}
						destroy_maa_regiment = yes
					}
					every_maa_regiment = {
						limit = {
							can_upgrade_maa = yes
							is_maa_type = olog_hai
						}
						create_maa_or_upgrade_regiment_effect = { TYPE = olog_hai }
					}
					every_maa_regiment = {
						limit = {
							can_upgrade_maa = yes
							is_maa_type = caragor_pack
						}
						create_maa_or_upgrade_regiment_effect = { TYPE = caragor_pack }
					}
					every_maa_regiment = {
						limit = {
							can_upgrade_maa = yes
							is_maa_type = siege_tower
						}
						create_maa_or_upgrade_regiment_effect = { TYPE = siege_tower }
					}
					every_maa_regiment = {
						limit = {
							can_upgrade_maa = yes
							is_maa_type = mountainorc_warriors
						}
						create_maa_or_upgrade_regiment_effect = { TYPE = mountainorc_warriors }
					}
				}
			}
		}
	}
}

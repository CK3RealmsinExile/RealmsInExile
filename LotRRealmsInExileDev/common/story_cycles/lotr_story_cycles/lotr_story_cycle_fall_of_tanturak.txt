lotr_story_cycle_fall_of_tanturak = {
	
	on_setup = {
#		set_variable = {
#			name = story_owner
#			value = story_owner
#		}
		set_global_variable = {
			name = fall_of_tanturak
			value = this
		}
#		set_global_variable = {
#			name = ongoing_fall_of_tanturak
#			value = yes
#		}
	}

	on_end = {
		debug_log = "Fall of Tantûrak ended on:"
		debug_log_date = yes
	}

	on_owner_death = {
		if = {
			limit = {
				exists = story_owner.player_heir
			}		
			set_global_variable = {
				name = story_owner
				value = story_owner.player_heir
			}
			make_story_owner = story_owner.player_heir
		}

	}


	#End the Story Cycle if the civil war has come to an end
	effect_group = {
		months = 1
		trigger = {
			OR = { #Any of these can be true
				AND = {
					title:k_tanturak = { is_title_created = no }
					title:k_tanturakusurper = { is_title_created = no }
				}
				title:k_tanturak.holder = {
					NOR = {
						culture = culture:tanturaki
						culture = culture:usakani
					}
				}
				story_owner = {
					AND = {
						has_title = title:k_tanturak
						OR = {
							has_title = title:k_tanturakusurper
							title:k_tanturakusurper = { is_title_created = no }
						}
					}
				}
			}
		}

		first_valid = {
			triggered_effect = { #A claimant consolidates Tanturak			
				trigger = {
					title:k_tanturak.holder = {culture = culture:tanturaki}
					OR = {
						title:k_tanturak.holder = {has_title = title:k_tanturakusurper}
						title:k_tanturakusurper = { is_title_created = no }
					}
				}
				effect = {
					remove_global_variable = fall_of_tanturak					
					set_global_variable = {
						name = ongoing_fall_of_tanturak
						value = no
					}						
					every_ruler = {
						limit = {
							capital_province = {
								geographical_region = middleearth_uttersouth_usakan
							}
						}
						trigger_event = fall_of_tanturak.0001
					}									
					end_story = yes
				}
			}
			triggered_effect = { #Succesful Native revolution			
				trigger = {
					OR = {
						title:k_tanturak.holder = {culture = culture:usakani}
						AND = {
							title:k_tanturak = { is_title_created = no }
							title:k_tanturakusurper = { is_title_created = no }
						}
					}					
				}
				effect = {
					remove_global_variable = fall_of_tanturak
					set_global_variable = {
						name = ongoing_fall_of_tanturak
						value = no
					}						
					every_ruler = {
						limit = {
							capital_province = {
								geographical_region = middleearth_uttersouth_usakan
							}
						}
						trigger_event = fall_of_tanturak.0002
					}					
					end_story = yes
				}
			}	
			triggered_effect = { #A foreign power conquers Tanturak		
				trigger = {
					title:k_tanturak.holder = {
						NOR = {
							culture = culture:tanturaki
							culture = culture:usakani
						}
					}
				}
				effect = {
					remove_global_variable = fall_of_tanturak
					set_global_variable = {
						name = ongoing_fall_of_tanturak
						value = no
					}						
					every_ruler = {
						limit = {
							capital_province = {
								geographical_region = middleearth_uttersouth_usakan
							}
						}
						trigger_event = fall_of_tanturak.0003
					}
					end_story = yes
				}
			}			
		}
	}

	effect_group = { # Kill off the Tanturaki if the cycle has been going on for more than 10 years
		years = 10
		trigger = {
			story_owner = {
				is_alive = yes
			}
		}
		first_valid = {
			triggered_effect = {
				trigger = { always = yes }
				effect = {
					every_ruler = {
						limit = {
							capital_province = {
								geographical_region = middleearth_uttersouth_usakan
							}
						}
						trigger_event = fall_of_tanturak.0004
					}					
				}
			}
		}
	}

	# Tanturaki Kings Events
	effect_group = {
		days = { 20 40 }

		trigger = {
			story_owner = {
				is_alive = yes
				culture = culture:tanturaki
			}
		}

		first_valid = {
			triggered_effect = {
				trigger = { always = yes }
				effect = {
					save_scope_as = story_scope
					story_owner = {
						trigger_event = {
							on_action = ongoing_fall_of_tanturak_owner_events
						}
					}
				}
			}
		}
	}

	# Native Rebel Events
	effect_group = {
		days = { 20 40 }

		trigger = {
			story_owner = {
				is_alive = yes
				culture = culture:usakani
			}
		}

		first_valid = {
			triggered_effect = {
				trigger = { always = yes }
				effect = {
					save_scope_as = story_scope
					story_owner = {
						trigger_event = {
							on_action = ongoing_fall_of_tanturak_owner_events
						}
					}
				}
			}
		}
	}

	#Maintenance group
	effect_group = {
		months = 1
		#Give the story cycle to anyone relevant in Tanturak
		triggered_effect = {
			trigger = {
				any_ruler = {
					limit = {
						capital_province = {
							geographical_region = middleearth_uttersouth_usakan
						}
						OR = {
							culture = culture:usakani
							culture = culture:tanturaki
						}
						is_independent_ruler = yes
					}
					NOT = {has_character_flag = involved_in_fall_of_tanturak}
				}
			}
			effect = {
				every_ruler = {
					limit = {
						capital_province = {
							geographical_region = middleearth_uttersouth_usakan
						}
						OR = {
							culture = culture:usakani
							culture = culture:tanturaki
						}
						is_independent_ruler = yes
						NOT = {has_character_flag = involved_in_fall_of_tanturak}
					}
					add_character_flag = involved_in_fall_of_tanturak
					create_story = lotr_story_cycle_fall_of_tanturak
				}
			}
		}
	}
}

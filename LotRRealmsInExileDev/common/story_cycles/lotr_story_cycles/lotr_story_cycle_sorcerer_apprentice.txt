story_cycle_sorcerer_apprentice = {
	
	on_setup = {
		story_owner = {
			random_relation = {
				type = master_sorcerer
				save_scope_as = master_sorcerer
			}
		}
		set_variable = {
			name = master_sorcerer
			value = scope:master_sorcerer
		}
	}

	on_end = { }

	on_owner_death = { end_story = yes }

	effect_group = {
		years = 1
		
		triggered_effect = {
			trigger = { story_owner = { is_available_adult = yes } }
			effect = {
				story_owner = { trigger_event = ongoing_apprentice_sorcerer_events }
			}
		}
	}
}

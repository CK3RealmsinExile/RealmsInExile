macro = {
	description = "Does the current orc ruler have a stable succession?"
	definition = "OrcSuccessionStable(TargetCharacter)"
	replace_with = "EqualTo_CFixedPoint(TargetCharacter.MakeScope.ScriptValue('lotr_disobedient_orc_meet_members'), '(CFixedPoint)0')"
}

macro = {
	description = "Does the current orc ruler have a chaotic succession?"
	definition = "OrcSuccessionChaotic(TargetCharacter)"
	replace_with = "GreaterThan_CFixedPoint(TargetCharacter.MakeScope.ScriptValue('lotr_disobedient_orc_meet_members'), '(CFixedPoint)0')"
}

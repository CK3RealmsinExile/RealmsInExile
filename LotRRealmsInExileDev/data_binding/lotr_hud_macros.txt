macro = {
	description = "Does the current character have the nomad or orcish government"
	definition = "IsNomadOrOrc(TargetCharacter)"
	replace_with = "Or(TargetCharacter.GetGovernment.IsType( 'orcish_government' ), And( TargetCharacter.GetGovernment.IsType( 'nomad_government' ), TargetCharacter.GetDomicile.GetType.IsDomicileType('yurt') ))"
}

macro = {
	description = "Does the current character have the orcish government"
	definition = "IsOrc(TargetCharacter)"
	replace_with = "TargetCharacter.GetGovernment.IsType( 'orcish_government' )"
}
